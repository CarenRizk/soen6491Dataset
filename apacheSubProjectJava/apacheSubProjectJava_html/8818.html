133       private PipelineResult runNaiveQueryTest(SpannerIO.Read readTransform) {
134         readTransform = readTransform.withBatching(false);
135         PCollection<Struct> results = pipeline.apply("read q", readTransform);
136         when(mockBatchTx.executeQuery(
137                 eq(Statement.of(QUERY_STATEMENT)), any(ReadQueryUpdateTransactionOption.class)))
138             .thenReturn(ResultSets.forRows(FAKE_TYPE, FAKE_ROWS));
139     
140         PAssert.that(results).containsInAnyOrder(FAKE_ROWS);
141         PipelineResult result = pipeline.run();
142         verifyQueryRequestMetricWasSet(readTransform.getSpannerConfig(), QUERY_NAME, "ok", 1);
143         return result;
144       }

146       private void runBatchReadTest(SpannerIO.Read readTransform) {
147     
148         PCollection<Struct> results = pipeline.apply("read q", readTransform);
149         when(mockBatchTx.partitionRead(
150                 any(PartitionOptions.class),
151                 eq(TABLE_ID),
152                 eq(KeySet.all()),
153                 eq(Arrays.asList("id", "name")),
154                 any(ReadQueryUpdateTransactionOption.class),
155                 any(ReadAndQueryOption.class)))
156             .thenReturn(Arrays.asList(fakePartition, fakePartition, fakePartition));
157         when(mockBatchTx.execute(any(Partition.class)))
158             .thenReturn(
159                 ResultSets.forRows(FAKE_TYPE, FAKE_ROWS.subList(0, 2)),
160                 ResultSets.forRows(FAKE_TYPE, FAKE_ROWS.subList(2, 4)),
161                 ResultSets.forRows(FAKE_TYPE, FAKE_ROWS.subList(4, 6)));
162     
163         PAssert.that(results).containsInAnyOrder(FAKE_ROWS);
164         pipeline.run();
165         verifyTableRequestMetricWasSet(readTransform.getSpannerConfig(), TABLE_ID, "ok", 4);
166       }

190       private long getRequestMetricCount(HashMap<String, String> baseLabels) {
191         MonitoringInfoMetricName name =
192             MonitoringInfoMetricName.named(MonitoringInfoConstants.Urns.API_REQUEST_COUNT, baseLabels);
193         MetricsContainerImpl container =
194             (MetricsContainerImpl) MetricsEnvironment.getCurrentContainer();
195         return container.getCounter(name).getCumulative();
196       }

198       private long getTableRequestMetric(SpannerConfig config, String table, String status) {
199         HashMap<String, String> baseLabels = getBaseMetricsLabels(config);
200         baseLabels.put(MonitoringInfoConstants.Labels.METHOD, "Read");
201         baseLabels.put(MonitoringInfoConstants.Labels.TABLE_ID, table);
202         baseLabels.put(
203             MonitoringInfoConstants.Labels.RESOURCE,
204             GcpResourceIdentifiers.spannerTable(
205                 baseLabels.get(MonitoringInfoConstants.Labels.SPANNER_PROJECT_ID),
206                 config.getInstanceId().get(),
207                 config.getDatabaseId().get(),
208                 table));
209         baseLabels.put(MonitoringInfoConstants.Labels.STATUS, status);
210         return getRequestMetricCount(baseLabels);
211       }

213       private long getQueryRequestMetric(SpannerConfig config, String queryName, String status) {
214         HashMap<String, String> baseLabels = getBaseMetricsLabels(config);
215         baseLabels.put(MonitoringInfoConstants.Labels.METHOD, "Read");
216         baseLabels.put(MonitoringInfoConstants.Labels.SPANNER_QUERY_NAME, queryName);

221                 config.getInstanceId().get(),
222                 config.getDatabaseId().get(),
223                 queryName));
224         baseLabels.put(MonitoringInfoConstants.Labels.STATUS, status);
225         return getRequestMetricCount(baseLabels);
226       }

238       private HashMap<String, String> getBaseMetricsLabels(SpannerConfig config) {
239         HashMap<String, String> baseLabels = new HashMap<>();
240         baseLabels.put(MonitoringInfoConstants.Labels.PTRANSFORM, "");
241         baseLabels.put(MonitoringInfoConstants.Labels.SERVICE, "Spanner");
242         baseLabels.put(
243             MonitoringInfoConstants.Labels.SPANNER_PROJECT_ID,
244             config.getProjectId() == null || config.getProjectId().get() == null
245                 ? SpannerOptions.getDefaultProjectId()
246                 : config.getProjectId().get());
247         baseLabels.put(
248             MonitoringInfoConstants.Labels.SPANNER_INSTANCE_ID, config.getInstanceId().get());
249         baseLabels.put(
250             MonitoringInfoConstants.Labels.SPANNER_DATABASE_ID, config.getDatabaseId().get());
251         return baseLabels;
252       }
