66     /**
67      * {@link PTransform}s for reading and writing text files.
68      *
69      * <h2>Reading text files</h2>
70      *
71      * <p>To read a {@link PCollection} from one or more text files, use {@code TextIO.read()} to
72      * instantiate a transform and use {@link TextIO.Read#from(String)} to specify the path of the
73      * file(s) to be read. Alternatively, if the filenames to be read are themselves in a {@link
74      * PCollection} you can use {@link FileIO} to match them and {@link TextIO#readFiles} to read them.
75      *
76      * <p>{@link #read} returns a {@link PCollection} of {@link String Strings}, each corresponding to
77      * one line of an input UTF-8 text file (split into lines delimited by '\n', '\r', or '\r\n', or
78      * specified delimiter see {@link TextIO.Read#withDelimiter}).
79      *
80      * <h3>Filepattern expansion and watching</h3>
81      *
82      * <p>By default, the filepatterns are expanded only once. {@link Read#watchForNewFiles} or the
83      * combination of {@link FileIO.Match#continuously(Duration, TerminationCondition)} and {@link
84      * #readFiles()} allow streaming of new files matching the filepattern(s).
85      *
86      * <p>By default, {@link #read} prohibits filepatterns that match no files, and {@link #readFiles()}
87      * allows them in case the filepattern contains a glob wildcard character. Use {@link
88      * Read#withEmptyMatchTreatment} or {@link
89      * FileIO.Match#withEmptyMatchTreatment(EmptyMatchTreatment)} plus {@link #readFiles()} to configure
90      * this behavior.
91      *
92      * <p>Example 1: reading a file or filepattern.
93      *
94      * <pre>{@code
95      * Pipeline p = ...;
96      *
97      * // A simple Read of a local file (only runs locally):
98      * PCollection<String> lines = p.apply(TextIO.read().from("/local/path/to/file.txt"));
99      * }</pre>
100      *
101      * <p>Example 2: reading a PCollection of filenames.
102      *
103      * <pre>{@code
104      * Pipeline p = ...;
105      *
106      * // E.g. the filenames might be computed from other data in the pipeline, or
107      * // read from a data source.
108      * PCollection<String> filenames = ...;
109      *
110      * // Read all files in the collection.
111      * PCollection<String> lines =
112      *     filenames
113      *         .apply(FileIO.matchAll())
114      *         .apply(FileIO.readMatches())
115      *         .apply(TextIO.readFiles());
116      * }</pre>
117      *
118      * <p>Example 3: streaming new files matching a filepattern.
119      *
120      * <pre>{@code
121      * Pipeline p = ...;
122      *
123      * PCollection<String> lines = p.apply(TextIO.read()
124      *     .from("/local/path/to/files/*")
125      *     .watchForNewFiles(
126      *       // Check for new files every minute
127      *       Duration.standardMinutes(1),
128      *       // Stop watching the filepattern if no new files appear within an hour
129      *       afterTimeSinceNewOutput(Duration.standardHours(1))));
130      * }</pre>
131      *
132      * <h3>Reading a very large number of files</h3>
133      *
134      * <p>If it is known that the filepattern will match a very large number of files (e.g. tens of
135      * thousands or more), use {@link Read#withHintMatchesManyFiles} for better performance and
136      * scalability. Note that it may decrease performance if the filepattern matches only a small number
137      * of files.
138      *
139      * <h2>Writing text files</h2>
140      *
141      * <p>To write a {@link PCollection} to one or more text files, use {@code TextIO.write()}, using
142      * {@link TextIO.Write#to(String)} to specify the output prefix of the files to write.
143      *
144      * <p>For example:
145      *
146      * <pre>{@code
147      * // A simple Write to a local file (only runs locally):
148      * PCollection<String> lines = ...;
149      * lines.apply(TextIO.write().to("/path/to/file.txt"));
150      *
151      * // Same as above, only with Gzip compression:
152      * PCollection<String> lines = ...;
153      * lines.apply(TextIO.write().to("/path/to/file.txt"))
154      *      .withSuffix(".txt")
155      *      .withCompression(Compression.GZIP));
156      * }</pre>
157      *
158      * <p>Any existing files with the same names as generated output files will be overwritten.
159      *
160      * <p>If you want better control over how filenames are generated than the default policy allows, a
161      * custom {@link FilenamePolicy} can also be set using {@link TextIO.Write#to(FilenamePolicy)}.
162      *
163      * <h3>Advanced features</h3>
164      *
165      * <p>{@link TextIO} supports all features of {@link FileIO#write} and {@link FileIO#writeDynamic},
166      * such as writing windowed/unbounded data, writing data to multiple destinations, and so on, by
167      * providing a {@link Sink} via {@link #sink()}.
168      *
169      * <p>For example, to write events of different type to different filenames:
170      *
171      * <pre>{@code
172      * PCollection<Event> events = ...;
173      * events.apply(FileIO.<EventType, Event>writeDynamic()
174      *       .by(Event::getTypeName)
175      *       .via(TextIO.sink(), Event::toString)
176      *       .to(type -> nameFilesUsingWindowPaneAndShard(".../events/" + type + "/data", ".txt")));
177      * }</pre>
178      *
179      * <p>For backwards compatibility, {@link TextIO} also supports the legacy {@link
180      * DynamicDestinations} interface for advanced features via {@link Write#to(DynamicDestinations)}.
181      *
182      * <p>Error handling for records that are malformed can be handled by using {@link
183      * TypedWrite#withBadRecordErrorHandler(ErrorHandler)}. See documentation in {@link FileIO} for
184      * details on usage
185      */

417         static boolean isSelfOverlapping(byte[] s) {
418           // s self-overlaps if v exists such as s = vu = wv with u and w non empty
419           for (int i = 1; i < s.length - 1; ++i) {
420             if (ByteBuffer.wrap(s, 0, i).equals(ByteBuffer.wrap(s, s.length - i, i))) {
421               return true;
422             }
423           }
424           return false;
425         }

427         @Override
428         public PCollection<String> expand(PBegin input) {
429           checkNotNull(getFilepattern(), "need to set the filepattern of a TextIO.Read transform");
430           if (getMatchConfiguration().getWatchInterval() == null && !getHintMatchesManyFiles()) {
431             return input.apply("Read", org.apache.beam.sdk.io.Read.from(getSource()));
432           }
433     
434           // All other cases go through FileIO + ReadFiles
435           return input
436               .apply("Create filepattern", Create.ofProvider(getFilepattern(), StringUtf8Coder.of()))
437               .apply("Match All", FileIO.matchAll().withConfiguration(getMatchConfiguration()))
438               .apply(
439                   "Read Matches",
440                   FileIO.readMatches()
441                       .withCompression(getCompression())
442                       .withDirectoryTreatment(DirectoryTreatment.PROHIBIT))
443               .apply(
444                   "Via ReadFiles",
445                   readFiles().withDelimiter(getDelimiter()).withSkipHeaderLines(getSkipHeaderLines()));
446         }

1076         private DynamicDestinations<UserT, DestinationT, String> resolveDynamicDestinations() {
1077           DynamicDestinations<UserT, DestinationT, String> dynamicDestinations =
1078               getDynamicDestinations();
1079           if (dynamicDestinations == null) {
1080             if (getDestinationFunction() != null) {
1081               // In this case, DestinationT == Params
1082               dynamicDestinations =
1083                   (DynamicDestinations)
1084                       DynamicFileDestinations.toDefaultPolicies(
1085                           getDestinationFunction(), getEmptyDestination(), getFormatFunction());
1086             } else {
1087               // In this case, DestinationT == Void
1088               FilenamePolicy usedFilenamePolicy = getFilenamePolicy();
1089               if (usedFilenamePolicy == null) {
1090                 usedFilenamePolicy =
1091                     DefaultFilenamePolicy.fromStandardParameters(
1092                         getFilenamePrefix(),
1093                         getShardTemplate(),
1094                         getFilenameSuffix(),
1095                         getWindowedWrites());
1096               }
1097               dynamicDestinations =
1098                   (DynamicDestinations)
1099                       DynamicFileDestinations.constant(usedFilenamePolicy, getFormatFunction());
1100             }
1101           }
1102           return dynamicDestinations;
1103         }

1105         @Override
1106         public WriteFilesResult<DestinationT> expand(PCollection<UserT> input) {
1107           checkState(
1108               getFilenamePrefix() != null || getTempDirectory() != null,
1109               "Need to set either the filename prefix or the tempDirectory of a TextIO.Write "
1110                   + "transform.");
1111     
1112           List<?> allToArgs =
1113               Lists.newArrayList(
1114                   getFilenamePolicy(),
1115                   getDynamicDestinations(),
1116                   getFilenamePrefix(),
1117                   getDestinationFunction());
1118           checkArgument(
1119               1
1120                   == Iterables.size(
1121                       allToArgs.stream()
1122                           .filter(Predicates.notNull()::apply)
1123                           .collect(Collectors.toList())),
1124               "Exactly one of filename policy, dynamic destinations, filename prefix, or destination "
1125                   + "function must be set");
1126     
1127           if (getDynamicDestinations() != null) {
1128             checkArgument(
1129                 getFormatFunction() == null,
1130                 "A format function should not be specified "
1131                     + "with DynamicDestinations. Use DynamicDestinations.formatRecord instead");
1132           }
1133           if (getFilenamePolicy() != null || getDynamicDestinations() != null) {
1134             checkState(
1135                 getShardTemplate() == null && getFilenameSuffix() == null,
1136                 "shardTemplate and filenameSuffix should only be used with the default "
1137                     + "filename policy");
1138           }
1139           ValueProvider<ResourceId> tempDirectory = getTempDirectory();
1140           if (tempDirectory == null) {
1141             tempDirectory = getFilenamePrefix();
1142           }
1143           WriteFiles<UserT, DestinationT, String> write =
1144               WriteFiles.to(
1145                   new TextSink<>(
1146                       tempDirectory,
1147                       resolveDynamicDestinations(),
1148                       getDelimiter(),
1149                       getHeader(),
1150                       getFooter(),
1151                       getWritableByteChannelFactory()));
1152           if (getNumShards() != null) {
1153             write = write.withNumShards(getNumShards());
1154           }
1155           if (getWindowedWrites()) {
1156             write = write.withWindowedWrites();
1157           }
1158           if (getAutoSharding()) {
1159             write = write.withAutoSharding();
1160           }
1161           if (getNoSpilling()) {
1162             write = write.withNoSpilling();
1163           }
1164           if (getBadRecordErrorHandler() != null) {
1165             write = write.withBadRecordErrorHandler(getBadRecordErrorHandler());
1166           }
1167           if (getSkipIfEmpty()) {
1168             write = write.withSkipIfEmpty();
1169           }
1170           if (getBatchSize() != null) {
1171             write = write.withBatchSize(getBatchSize());
1172           }
1173           if (getBatchSizeBytes() != null) {
1174             write = write.withBatchSizeBytes(getBatchSizeBytes());
1175           }
1176           if (getBatchMaxBufferingDuration() != null) {
1177             write = write.withBatchMaxBufferingDuration(getBatchMaxBufferingDuration());
1178           }
1179           return input.apply("WriteFiles", write);
1180         }
