335       @Test
336       public void testSortedMethodIncludesAllSchemaFields() {
337         // This test is most likely to break when new Schema object attributes are added. It is designed
338         // this way to make sure that the Schema::sorted() method is updated to return a full sorted
339         // copy.
340     
341         // Schema object attributes that are accounted for in Schema::sorted().
342         // Note: Only the appropriate ones are copied over.
343         List<String> attributesAccountedForInSorted =
344             Arrays.asList(
345                 "fieldIndices",
346                 "encodingPositions",
347                 "encodingPositionsOverridden",
348                 "fields",
349                 "hashCode",
350                 "uuid",
351                 "options");
352     
353         // Current attributes in Schema object.
354         List<String> currentAttributes =
355             Arrays.stream(Schema.class.getDeclaredFields())
356                 .filter(field -> !field.isSynthetic())
357                 .map(java.lang.reflect.Field::getName)
358                 .collect(Collectors.toList());
359     
360         List<String> differences = new ArrayList<>(currentAttributes);
361         differences.removeAll(attributesAccountedForInSorted);
362     
363         assertEquals(
364             String.format(
365                 "Detected attributes %s in Schema object that are not accounted for in Schema::sorted(). "
366                     + "If appropriate, sorted() should copy over these attributes as well. Either way, update this test after checking.",
367                 differences.toString()),
368             currentAttributes,
369             attributesAccountedForInSorted);
370       }
