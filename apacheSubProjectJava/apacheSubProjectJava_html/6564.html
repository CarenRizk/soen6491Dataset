32     /**
33      * Triggers control when the elements for a specific key and window are output. As elements arrive,
34      * they are put into one or more windows by a {@link Window} transform and its associated {@link
35      * WindowFn}, and then passed to the associated {@link Trigger} to determine if the {@link
36      * BoundedWindow Window's} contents should be output.
37      *
38      * <p>See {@link GroupByKey} and {@link Window} for more information about how grouping with windows
39      * works.
40      *
41      * <p>The elements that are assigned to a window since the last time it was fired (or since the
42      * window was created) are placed into the current window pane. Triggers are evaluated against the
43      * elements as they are added. When the root trigger fires, the elements in the current pane will be
44      * output. When the root trigger finishes (indicating it will never fire again), the window is
45      * closed and any new elements assigned to that window are discarded.
46      *
47      * <p>Several predefined triggers are provided:
48      *
49      * <ul>
50      *   <li>{@link AfterWatermark} for firing when the watermark passes a timestamp determined from
51      *       either the end of the window or the arrival of the first element in a pane.
52      *   <li>{@link AfterProcessingTime} for firing after some amount of processing time has elapsed
53      *       (typically since the first element in a pane).
54      *   <li>{@link AfterPane} for firing off a property of the elements in the current pane, such as
55      *       the number of elements that have been assigned to the current pane.
56      * </ul>
57      *
58      * <p>In addition, triggers can be combined in a variety of ways:
59      *
60      * <ul>
61      *   <li>{@link Repeatedly#forever} to create a trigger that executes forever. Any time its argument
62      *       finishes it gets reset and starts over. Can be combined with {@link Trigger#orFinally} to
63      *       specify a condition that causes the repetition to stop.
64      *   <li>{@link AfterEach#inOrder} to execute each trigger in sequence, firing each (and every) time
65      *       that a trigger fires, and advancing to the next trigger in the sequence when it finishes.
66      *   <li>{@link AfterFirst#of} to create a trigger that fires after at least one of its arguments
67      *       fires. An {@link AfterFirst} trigger finishes after it fires once.
68      *   <li>{@link AfterAll#of} to create a trigger that fires after all least one of its arguments
69      *       have fired at least once. An {@link AfterAll} trigger finishes after it fires once.
70      * </ul>
71      */
72     @SuppressWarnings({
73       "nullness" // TODO(https://github.com/apache/beam/issues/20497)
74     })
75     public abstract class Trigger implements Serializable {
76     
77       protected final List<Trigger> subTriggers;
78     
79       protected Trigger(List<Trigger> subTriggers) {
80         this.subTriggers = subTriggers;
81       }
82     
83       protected Trigger() {
84         this(Collections.emptyList());
85       }
86     
87       public List<Trigger> subTriggers() {
88         return MoreObjects.firstNonNull(subTriggers, Collections.emptyList());
89       }
90     
91       /**
92        * Return a trigger to use after a {@link GroupByKey} to preserve the intention of this trigger.
93        * Specifically, triggers that are time based and intended to provide speculative results should
94        * continue providing speculative results. Triggers that fire once (or multiple times) should
95        * continue firing once (or multiple times).
96        *
97        * <p>If this method is not overridden, its default implementation delegates its behavior to
98        * {@link #getContinuationTrigger(List)} which is expected to be implemented by subclasses.
99        */
100       public Trigger getContinuationTrigger() {
101         if (subTriggers == null) {
102           return getContinuationTrigger(null);
103         }
104     
105         List<Trigger> subTriggerContinuations = new ArrayList<>();
106         for (Trigger subTrigger : subTriggers) {
107           subTriggerContinuations.add(subTrigger.getContinuationTrigger());
108         }
109         return getContinuationTrigger(subTriggerContinuations);
110       }
111     
112       /**
113        * Subclasses should override this to return the {@link #getContinuationTrigger} of this {@link
114        * Trigger}. For convenience, this is provided the continuation trigger of each of the
115        * sub-triggers in the same order as {@link #subTriggers}.
116        *
117        * @param continuationTriggers contains the result of {@link #getContinuationTrigger()} on each of
118        *     the {@code subTriggers} in the same order.
119        */
120       protected abstract Trigger getContinuationTrigger(List<Trigger> continuationTriggers);
121     
122       /**
123        * <b><i>For internal use only; no backwards-compatibility guarantees.</i></b>
124        *
125        * <p>Returns a bound in event time by which this trigger would have fired at least once for a
126        * given window had there been input data.
127        *
128        * <p>For triggers that do not fire based on the watermark advancing, returns {@link
129        * BoundedWindow#TIMESTAMP_MAX_VALUE}.
130        *
131        * <p>This estimate may be used, for example, to determine that there are no elements in a
132        * side-input window, which causes the default value to be used instead.
133        */
134       @Internal
135       public abstract Instant getWatermarkThatGuaranteesFiring(BoundedWindow window);
136     
137       /**
138        * <b><i>For internal use only; no backwards-compatibility guarantees.</i></b>
139        *
140        * <p>Indicates whether this trigger may "finish". A top level trigger that finishes can cause
141        * data loss, so is rejected by GroupByKey validation.
142        */
143       @Internal
144       public abstract boolean mayFinish();
145     
146       /**
147        * <b><i>For internal use only; no backwards-compatibility guarantees.</i></b>
148        *
149        * <p>Returns whether this performs the same triggering as the given {@link Trigger}.
150        */
151       @Internal
152       public boolean isCompatible(Trigger other) {
153         if (!getClass().equals(other.getClass())) {
154           return false;
155         }
156     
157         if (subTriggers == null) {
158           return other.subTriggers == null;
159         } else if (other.subTriggers == null) {
160           return false;
161         } else if (subTriggers.size() != other.subTriggers.size()) {
162           return false;
163         }
164     
165         for (int i = 0; i < subTriggers.size(); i++) {
166           if (!subTriggers.get(i).isCompatible(other.subTriggers.get(i))) {
167             return false;
168           }
169         }
170     
171         return true;
172       }
173     
174       @Override
175       public String toString() {
176         String simpleName = getClass().getSimpleName();
177         if (getClass().getEnclosingClass() != null) {
178           simpleName = getClass().getEnclosingClass().getSimpleName() + "." + simpleName;
179         }
180         if (subTriggers == null || subTriggers.isEmpty()) {
181           return simpleName;
182         } else {
183           return simpleName + "(" + Joiner.on(", ").join(subTriggers) + ")";
184         }
185       }
186     
187       @Override
188       public boolean equals(@Nullable Object obj) {
189         if (this == obj) {
190           return true;
191         }
192         if (!(obj instanceof Trigger)) {
193           return false;
194         }
195         Trigger that = (Trigger) obj;
196         return Objects.equals(getClass(), that.getClass())
197             && Objects.equals(subTriggers, that.subTriggers);
198       }
199     
200       @Override
201       public int hashCode() {
202         return Objects.hash(getClass(), subTriggers);
203       }
204     
205       /**
206        * Specify an ending condition for this trigger. If the {@code until} {@link Trigger} fires then
207        * the combination fires.
208        *
209        * <p>The expression {@code t1.orFinally(t2)} fires every time {@code t1} fires, and finishes as
210        * soon as either {@code t1} finishes or {@code t2} fires, in which case it fires one last time
211        * for {@code t2}. Both {@code t1} and {@code t2} are executed in parallel. This means that {@code
212        * t1} may have fired since {@code t2} started, so not all of the elements that {@code t2} has
213        * seen are necessarily in the current pane.
214        *
215        * <p>For example the final firing of the following trigger may only have 1 element:
216        *
217        * <pre>{@code
218        * Repeatedly.forever(AfterPane.elementCountAtLeast(2))
219        *     .orFinally(AfterPane.elementCountAtLeast(5))
220        * }</pre>
221        *
222        * <p>Note that if {@code t1} is {@link OnceTrigger}, then {@code t1.orFinally(t2)} is the same as
223        * {@code AfterFirst.of(t1, t2)}.
224        */
225       public OrFinallyTrigger orFinally(OnceTrigger until) {
226         return new OrFinallyTrigger(this, until);
227       }
228     
229       /**
230        * <b><i>For internal use only; no backwards-compatibility guarantees.</i></b>
231        *
232        * <p>Triggers that are guaranteed to fire at most once should extend {@link OnceTrigger} rather
233        * than the general {@link Trigger} class to indicate that behavior.
234        */
235       @Internal
236       public abstract static class OnceTrigger extends Trigger {
237         protected OnceTrigger(List<Trigger> subTriggers) {
238           super(subTriggers);
239         }
240     
241         @Override
242         public final boolean mayFinish() {
243           return true;
244         }
245     
246         @Override
247         public final OnceTrigger getContinuationTrigger() {
248           Trigger continuation = super.getContinuationTrigger();
249           if (!(continuation instanceof OnceTrigger)) {
250             throw new IllegalStateException("Continuation of a OnceTrigger must be a OnceTrigger");
251           }
252           return (OnceTrigger) continuation;
253         }
254       }
255     }
