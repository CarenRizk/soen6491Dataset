87     /**
88      * {@link PTransform}s for reading and writing Avro files.
89      *
90      * <h2>Reading Avro files</h2>
91      *
92      * <p>To read a {@link PCollection} from one or more Avro files with the same schema known at
93      * pipeline construction time, use {@link #read}, using {@link Read#from} to specify the filename or
94      * filepattern to read from. If the filepatterns to be read are themselves in a {@link PCollection}
95      * you can use {@link FileIO} to match them and {@link AvroIO#readFiles} to read them. If the schema
96      * is unknown at pipeline construction time, use {@link #parseGenericRecords} or {@link
97      * #parseFilesGenericRecords}.
98      *
99      * <p>Many configuration options below apply to several or all of these transforms.
100      *
101      * <p>See {@link FileSystems} for information on supported file systems and filepatterns.
102      *
103      * <h3>Filepattern expansion and watching</h3>
104      *
105      * <p>By default, the filepatterns are expanded only once. {@link Read#watchForNewFiles} or the
106      * combination of {@link FileIO.Match#continuously(Duration, TerminationCondition)} and {@link
107      * AvroIO#readFiles(Class)} allow streaming of new files matching the filepattern(s).
108      *
109      * <p>By default, {@link #read} prohibits filepatterns that match no files, and {@link
110      * AvroIO#readFiles(Class)} allows them in case the filepattern contains a glob wildcard character.
111      * Use {@link Read#withEmptyMatchTreatment} or {@link
112      * FileIO.Match#withEmptyMatchTreatment(EmptyMatchTreatment)} plus {@link AvroIO#readFiles(Class)}
113      * to configure this behavior.
114      *
115      * <h3>Reading records of a known schema</h3>
116      *
117      * <p>To read specific records, such as Avro-generated classes, use {@link #read(Class)}. To read
118      * {@link GenericRecord GenericRecords}, use {@link #readGenericRecords(Schema)} which takes a
119      * {@link Schema} object, or {@link #readGenericRecords(String)} which takes an Avro schema in a
120      * JSON-encoded string form. An exception will be thrown if a record doesn't match the specified
121      * schema. Likewise, to read a {@link PCollection} of filepatterns, apply {@link FileIO} matching
122      * plus {@link #readFilesGenericRecords}.
123      *
124      * <p>For example:
125      *
126      * <pre>{@code
127      * Pipeline p = ...;
128      *
129      * // Read Avro-generated classes from files on GCS
130      * PCollection<AvroAutoGenClass> records =
131      *     p.apply(AvroIO.read(AvroAutoGenClass.class).from("gs://my_bucket/path/to/records-*.avro"));
132      *
133      * // Read GenericRecord's of the given schema from files on GCS
134      * Schema schema = new Schema.Parser().parse(new File("schema.avsc"));
135      * PCollection<GenericRecord> records =
136      *     p.apply(AvroIO.readGenericRecords(schema)
137      *                .from("gs://my_bucket/path/to/records-*.avro"));
138      * }</pre>
139      *
140      * <h3>Reading records of an unknown schema</h3>
141      *
142      * <p>To read records from files whose schema is unknown at pipeline construction time or differs
143      * between files, use {@link #parseGenericRecords} - in this case, you will need to specify a
144      * parsing function for converting each {@link GenericRecord} into a value of your custom type.
145      * Likewise, to read a {@link PCollection} of filepatterns with unknown schema, use {@link FileIO}
146      * matching plus {@link #parseFilesGenericRecords(SerializableFunction)}.
147      *
148      * <p>For example:
149      *
150      * <pre>{@code
151      * Pipeline p = ...;
152      *
153      * PCollection<Foo> records =
154      *     p.apply(AvroIO.parseGenericRecords(new SerializableFunction<GenericRecord, Foo>() {
155      *       public Foo apply(GenericRecord record) {
156      *         // If needed, access the schema of the record using record.getSchema()
157      *         return ...;
158      *       }
159      *     }));
160      * }</pre>
161      *
162      * <h3>Reading from a {@link PCollection} of filepatterns</h3>
163      *
164      * <pre>{@code
165      * Pipeline p = ...;
166      *
167      * PCollection<String> filepatterns = p.apply(...);
168      * PCollection<AvroAutoGenClass> records =
169      *     filepatterns.apply(AvroIO.readAll(AvroAutoGenClass.class));
170      * PCollection<AvroAutoGenClass> records =
171      *     filepatterns
172      *         .apply(FileIO.matchAll())
173      *         .apply(FileIO.readMatches())
174      *         .apply(AvroIO.readFiles(AvroAutoGenClass.class));
175      * PCollection<GenericRecord> genericRecords =
176      *     filepatterns.apply(AvroIO.readGenericRecords(schema));
177      * PCollection<Foo> records =
178      *     filepatterns
179      *         .apply(FileIO.matchAll())
180      *         .apply(FileIO.readMatches())
181      *         .apply(AvroIO.parseFilesGenericRecords(new SerializableFunction...);
182      * }</pre>
183      *
184      * <h3>Streaming new files matching a filepattern</h3>
185      *
186      * <pre>{@code
187      * Pipeline p = ...;
188      *
189      * PCollection<AvroAutoGenClass> lines = p.apply(AvroIO
190      *     .read(AvroAutoGenClass.class)
191      *     .from("gs://my_bucket/path/to/records-*.avro")
192      *     .watchForNewFiles(
193      *       // Check for new files every minute
194      *       Duration.standardMinutes(1),
195      *       // Stop watching the filepattern if no new files appear within an hour
196      *       afterTimeSinceNewOutput(Duration.standardHours(1))));
197      * }</pre>
198      *
199      * <h3>Reading a very large number of files</h3>
200      *
201      * <p>If it is known that the filepattern will match a very large number of files (e.g. tens of
202      * thousands or more), use {@link Read#withHintMatchesManyFiles} for better performance and
203      * scalability. Note that it may decrease performance if the filepattern matches only a small number
204      * of files.
205      *
206      * <h3>Inferring Beam schemas from Avro files</h3>
207      *
208      * <p>If you want to use SQL or schema based operations on an Avro-based PCollection, you must
209      * configure the read transform to infer the Beam schema and automatically setup the Beam related
210      * coders by doing:
211      *
212      * <pre>{@code
213      * PCollection<AvroAutoGenClass> records =
214      *     p.apply(AvroIO.read(...).from(...).withBeamSchemas(true));
215      * }</pre>
216      *
217      * <h3>Inferring Beam schemas from Avro PCollections</h3>
218      *
219      * <p>If you created an Avro-based PCollection by other means e.g. reading records from Kafka or as
220      * the output of another PTransform, you may be interested on making your PCollection schema-aware
221      * so you can use the Schema-based APIs or Beam's SqlTransform.
222      *
223      * <p>If you are using Avro specific records (generated classes from an Avro schema), you can
224      * register a schema provider for the specific Avro class to make any PCollection of these objects
225      * schema-aware.
226      *
227      * <pre>{@code
228      * pipeline.getSchemaRegistry().registerSchemaProvider(AvroAutoGenClass.class, AvroAutoGenClass.getClassSchema());
229      * }</pre>
230      *
231      * You can also manually set an Avro-backed Schema coder for a PCollection using {@link
232      * AvroUtils#schemaCoder(Class, Schema)} to make it schema-aware.
233      *
234      * <pre>{@code
235      * PCollection<AvroAutoGenClass> records = ...
236      * AvroCoder<AvroAutoGenClass> coder = (AvroCoder<AvroAutoGenClass>) users.getCoder();
237      * records.setCoder(AvroUtils.schemaCoder(coder.getType(), coder.getSchema()));
238      * }</pre>
239      *
240      * <p>If you are using GenericRecords you may need to set a specific Beam schema coder for each
241      * PCollection to match their internal Avro schema.
242      *
243      * <pre>{@code
244      * org.apache.avro.Schema avroSchema = ...
245      * PCollection<GenericRecord> records = ...
246      * records.setCoder(AvroUtils.schemaCoder(avroSchema));
247      * }</pre>
248      *
249      * <h2>Writing Avro files</h2>
250      *
251      * <p>To write a {@link PCollection} to one or more Avro files, use {@link Write}, using {@code
252      * AvroIO.write().to(String)} to specify the output filename prefix. The default {@link
253      * DefaultFilenamePolicy} will use this prefix, in conjunction with a {@link ShardNameTemplate} (set
254      * via {@link Write#withShardNameTemplate(String)}) and optional filename suffix (set via {@link
255      * Write#withSuffix(String)}, to generate output filenames in a sharded way. You can override this
256      * default write filename policy using {@link Write#to(FilenamePolicy)} to specify a custom file
257      * naming policy.
258      *
259      * <p>By default, {@link Write} produces output files that are compressed using the {@link
260      * org.apache.avro.file.Codec CodecFactory.snappyCodec()}. This default can be changed or overridden
261      * using {@link Write#withCodec}.
262      *
263      * <h3>Writing specific or generic records</h3>
264      *
265      * <p>To write specific records, such as Avro-generated classes, use {@link #write(Class)}. To write
266      * {@link GenericRecord GenericRecords}, use either {@link #writeGenericRecords(Schema)} which takes
267      * a {@link Schema} object, or {@link #writeGenericRecords(String)} which takes a schema in a
268      * JSON-encoded string form. An exception will be thrown if a record doesn't match the specified
269      * schema.
270      *
271      * <p>For example:
272      *
273      * <pre>{@code
274      * // A simple Write to a local file (only runs locally):
275      * PCollection<AvroAutoGenClass> records = ...;
276      * records.apply(AvroIO.write(AvroAutoGenClass.class).to("/path/to/file.avro"));
277      *
278      * // A Write to a sharded GCS file (runs locally and using remote execution):
279      * Schema schema = new Schema.Parser().parse(new File("schema.avsc"));
280      * PCollection<GenericRecord> records = ...;
281      * records.apply("WriteToAvro", AvroIO.writeGenericRecords(schema)
282      *     .to("gs://my_bucket/path/to/numbers")
283      *     .withSuffix(".avro"));
284      * }</pre>
285      *
286      * <h3>Writing windowed or unbounded data</h3>
287      *
288      * <p>By default, all input is put into the global window before writing. If per-window writes are
289      * desired - for example, when using a streaming runner - {@link Write#withWindowedWrites()} will
290      * cause windowing and triggering to be preserved. When producing windowed writes with a streaming
291      * runner that supports triggers, the number of output shards must be set explicitly using {@link
292      * Write#withNumShards(int)}; some runners may set this for you to a runner-chosen value, so you may
293      * need not set it yourself. A {@link FilenamePolicy} must be set, and unique windows and triggers
294      * must produce unique filenames.
295      *
296      * <h3>Writing data to multiple destinations</h3>
297      *
298      * <p>The following shows a more-complex example of AvroIO.Write usage, generating dynamic file
299      * destinations as well as a dynamic Avro schema per file. In this example, a PCollection of user
300      * events (e.g. actions on a website) is written out to Avro files. Each event contains the user id
301      * as an integer field. We want events for each user to go into a specific directory for that user,
302      * and each user's data should be written with a specific schema for that user; a side input is
303      * used, so the schema can be calculated in a different stage.
304      *
305      * <pre>{@code
306      * // This is the user class that controls dynamic destinations for this avro write. The input to
307      * // AvroIO.Write will be UserEvent, and we will be writing GenericRecords to the file (in order
308      * // to have dynamic schemas). Everything is per userid, so we define a dynamic destination type
309      * // of Integer.
310      * class UserDynamicAvroDestinations
311      *     extends DynamicAvroDestinations<UserEvent, Integer, GenericRecord> {
312      *   private final PCollectionView<Map<Integer, String>> userToSchemaMap;
313      *   public UserDynamicAvroDestinations( PCollectionView<Map<Integer, String>> userToSchemaMap) {
314      *     this.userToSchemaMap = userToSchemaMap;
315      *   }
316      *   public GenericRecord formatRecord(UserEvent record) {
317      *     return formatUserRecord(record, getSchema(record.getUserId()));
318      *   }
319      *   public Schema getSchema(Integer userId) {
320      *     return new Schema.Parser().parse(sideInput(userToSchemaMap).get(userId));
321      *   }
322      *   public Integer getDestination(UserEvent record) {
323      *     return record.getUserId();
324      *   }
325      *   public Integer getDefaultDestination() {
326      *     return 0;
327      *   }
328      *   public FilenamePolicy getFilenamePolicy(Integer userId) {
329      *     return DefaultFilenamePolicy.fromParams(new Params().withBaseFilename(baseDir + "/user-"
330      *     + userId + "/events"));
331      *   }
332      *   public List<PCollectionView<?>> getSideInputs() {
333      *     return ImmutableList.<PCollectionView<?>>of(userToSchemaMap);
334      *   }
335      * }
336      * PCollection<UserEvents> events = ...;
337      * PCollectionView<Map<Integer, String>> userToSchemaMap = events.apply(
338      *     "ComputePerUserSchemas", new ComputePerUserSchemas());
339      * events.apply("WriteAvros", AvroIO.<Integer>writeCustomTypeToGenericRecords()
340      *     .to(new UserDynamicAvroDestinations(userToSchemaMap)));
341      * }</pre>
342      *
343      * <p>Error handling for writing records that are malformed can be handled by using {@link
344      * TypedWrite#withBadRecordErrorHandler(ErrorHandler)}. See documentation in {@link FileIO} for
345      * details on usage
346      */
