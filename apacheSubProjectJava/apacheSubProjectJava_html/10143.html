98       @Test
99       @Category(ValidatesRunner.class)
100       public void testFlattenInputMultipleCopies() {
101         int count = 5;
102         PCollection<Long> longs = p.apply("mkLines", GenerateSequence.from(0).to(count));
103         PCollection<Long> biggerLongs =
104             p.apply("mkOtherLines", GenerateSequence.from(0).to(count))
105                 .apply(
106                     MapElements.via(
107                         new SimpleFunction<Long, Long>() {
108                           @Override
109                           public Long apply(Long input) {
110                             return input + 10L;
111                           }
112                         }));
113     
114         PCollection<Long> flattened =
115             PCollectionList.of(longs).and(longs).and(biggerLongs).apply(Flatten.pCollections());
116     
117         List<Long> expectedLongs = new ArrayList<>();
118         for (int i = 0; i < count; i++) {
119           // The duplicated input
120           expectedLongs.add((long) i);
121           expectedLongs.add((long) i);
122           // The bigger longs
123           expectedLongs.add(i + 10L);
124         }
125         PAssert.that(flattened).containsInAnyOrder(expectedLongs);
126     
127         p.run();
128       }

206       @Test
207       @Category(ValidatesRunner.class)
208       public void testFlattenMultiplePCollectionsHavingMultipleConsumers() {
209         PCollection<String> input = p.apply(Create.of("AA", "BBB", "CC"));
210         final TupleTag<String> outputEvenLengthTag = new TupleTag<String>() {};
211         final TupleTag<String> outputOddLengthTag = new TupleTag<String>() {};
212     
213         PCollectionTuple tuple =
214             input.apply(
215                 ParDo.of(
216                         new DoFn<String, String>() {
217                           @ProcessElement
218                           public void processElement(ProcessContext c) {
219                             if (c.element().length() % 2 == 0) {
220                               c.output(c.element());
221                             } else {
222                               c.output(outputOddLengthTag, c.element());
223                             }
224                           }
225                         })
226                     .withOutputTags(outputEvenLengthTag, TupleTagList.of(outputOddLengthTag)));
227     
228         PCollection<String> outputEvenLength = tuple.get(outputEvenLengthTag);
229         PCollection<String> outputOddLength = tuple.get(outputOddLengthTag);
230     
231         PCollection<String> outputMerged =
232             PCollectionList.of(outputEvenLength).and(outputOddLength).apply(Flatten.pCollections());
233     
234         PAssert.that(outputMerged).containsInAnyOrder("AA", "BBB", "CC");
235         PAssert.that(outputEvenLength).containsInAnyOrder("AA", "CC");
236         PAssert.that(outputOddLength).containsInAnyOrder("BBB");
237     
238         p.run();
239       }

315       private static class IdentityFn<T> extends DoFn<T, T> {
316         @ProcessElement
317         public void processElement(ProcessContext c) {
318           c.output(c.element());
319         }
320       }

322       private PCollectionList<String> makePCollectionListOfStrings(
323           Pipeline p, List<List<String>> lists) {
324         return makePCollectionList(p, StringUtf8Coder.of(), lists);
325       }
326     
327       private <T> PCollectionList<T> makePCollectionList(
328           Pipeline p, Coder<T> coder, List<List<T>> lists) {
329         List<PCollection<T>> pcs = new ArrayList<>();
330         int index = 0;
331         for (List<T> list : lists) {
332           PCollection<T> pc = p.apply("Create" + index++, Create.of(list).withCoder(coder));
333           pcs.add(pc);
334         }
335         return PCollectionList.of(pcs);
336       }
337     
338       private <T> List<T> flattenLists(List<List<T>> lists) {
339         List<T> flattened = new ArrayList<>();
340         for (List<T> list : lists) {
341           flattened.addAll(list);
342         }
343         return flattened;
344       }
