97     /**
98      * Read and Write {@link PTransform}s for Cloud Pub/Sub streams. These transforms create and consume
99      * unbounded {@link PCollection PCollections}.
100      *
101      * <h3>Using local emulator</h3>
102      *
103      * <p>In order to use local emulator for Pubsub you should use {@code
104      * PubsubOptions#setPubsubRootUrl(String)} method to set host and port of your local emulator.
105      *
106      * <h3>Permissions</h3>
107      *
108      * <p>Permission requirements depend on the {@link PipelineRunner} that is used to execute the Beam
109      * pipeline. Please refer to the documentation of corresponding {@link PipelineRunner
110      * PipelineRunners} for more details.
111      *
112      * <h3>Updates to the I/O connector code</h3>
113      *
114      * For any significant updates to this I/O connector, please consider involving corresponding code
115      * reviewers mentioned <a
116      * href="https://github.com/apache/beam/blob/master/sdks/java/io/google-cloud-platform/OWNERS">
117      * here</a>.
118      *
119      * <h3>Example PubsubIO read usage</h3>
120      *
121      * <pre>{@code
122      * // Read from a specific topic; a subscription will be created at pipeline start time.
123      * PCollection<PubsubMessage> messages = PubsubIO.readMessages().fromTopic(topic);
124      *
125      * // Read from a subscription.
126      * PCollection<PubsubMessage> messages = PubsubIO.readMessages().fromSubscription(subscription);
127      *
128      * // Read messages including attributes. All PubSub attributes will be included in the PubsubMessage.
129      * PCollection<PubsubMessage> messages = PubsubIO.readMessagesWithAttributes().fromTopic(topic);
130      *
131      * // Examples of reading different types from PubSub.
132      * PCollection<String> strings = PubsubIO.readStrings().fromTopic(topic);
133      * PCollection<MyProto> protos = PubsubIO.readProtos(MyProto.class).fromTopic(topic);
134      * PCollection<MyType> avros = PubsubIO.readAvros(MyType.class).fromTopic(topic);
135      *
136      * }</pre>
137      *
138      * <h3>Example PubsubIO write usage</h3>
139      *
140      * Data can be written to a single topic or to a dynamic set of topics. In order to write to a
141      * single topic, the {@link PubsubIO.Write#to(String)} method can be used. For example:
142      *
143      * <pre>{@code
144      * avros.apply(PubsubIO.writeAvros(MyType.class).to(topic));
145      * protos.apply(PubsubIO.writeProtos(MyProto.class).to(topic));
146      * strings.apply(PubsubIO.writeStrings().to(topic));
147      * }</pre>
148      *
149      * Dynamic topic destinations can be accomplished by specifying a function to extract the topic from
150      * the record using the {@link PubsubIO.Write#to(SerializableFunction)} method. For example:
151      *
152      * <pre>{@code
153      * avros.apply(PubsubIO.writeAvros(MyType.class).
154      *      to((ValueInSingleWindow<Event> quote) -> {
155      *               String country = quote.getCountry();
156      *               return "projects/myproject/topics/events_" + country;
157      *              });
158      * }</pre>
159      *
160      * Dynamic topics can also be specified by writing {@link PubsubMessage} objects containing the
161      * topic and writing using the {@link PubsubIO#writeMessagesDynamic()} method. For example:
162      *
163      * <pre>{@code
164      * events.apply(MapElements.into(new TypeDescriptor<PubsubMessage>() {})
165      *                         .via(e -> new PubsubMessage(
166      *                             e.toByteString(), Collections.emptyMap()).withTopic(e.getCountry())))
167      * .apply(PubsubIO.writeMessagesDynamic());
168      * }</pre>
169      *
170      * <h3>Custom timestamps</h3>
171      *
172      * All messages read from PubSub have a stable publish timestamp that is independent of when the
173      * message is read from the PubSub topic. By default, the publish time is used as the timestamp for
174      * all messages read and the watermark is based on that. If there is a different logical timestamp
175      * to be used, that timestamp must be published in a PubSub attribute and specified using {@link
176      * PubsubIO.Read#withTimestampAttribute}. See the Javadoc for that method for the timestamp format.
177      */

284          * Creates a class representing a Pub/Sub subscription from the specified subscription path.
285          *
286          * <p>Cloud Pub/Sub subscription names should be of the form {@code
287          * projects/<project>/subscriptions/<subscription>}, where {@code <project>} is the name of the
288          * project the subscription belongs to. The {@code <subscription>} component must comply with
289          * the following requirements:
290          *
291          * <ul>
292          *   <li>Can only contain lowercase letters, numbers, dashes ('-'), underscores ('_') and
293          *       periods ('.').
294          *   <li>Must be between 3 and 255 characters.
295          *   <li>Must begin with a letter.
296          *   <li>Must end with a letter or a number.
297          *   <li>Cannot begin with {@code 'goog'} prefix.
298          * </ul>
299          */
300         public static PubsubSubscription fromPath(String path) {
301           if (path.startsWith(SUBSCRIPTION_RANDOM_TEST_PREFIX)
302               || path.startsWith(SUBSCRIPTION_STARTING_SIGNAL)) {
303             return new PubsubSubscription(PubsubSubscription.Type.FAKE, "", path);
304           }
305     
306           String projectName, subscriptionName;
307     
308           Matcher v1beta1Match = V1BETA1_SUBSCRIPTION_REGEXP.matcher(path);
309           if (v1beta1Match.matches()) {
310             LOG.warn(
311                 "Saw subscription in v1beta1 format. Subscriptions should be in the format "
312                     + "projects/<project_id>/subscriptions/<subscription_name>");
313             projectName = v1beta1Match.group(1);
314             subscriptionName = v1beta1Match.group(2);
315           } else {
316             Matcher match = SUBSCRIPTION_REGEXP.matcher(path);
317             if (!match.matches()) {
318               throw new IllegalArgumentException(
319                   "Pubsub subscription is not in "
320                       + "projects/<project_id>/subscriptions/<subscription_name> format: "
321                       + path);
322             }
323             projectName = match.group(1);
324             subscriptionName = match.group(2);
325           }
326     
327           validateProjectName(projectName);
328           validatePubsubName(subscriptionName);
329           return new PubsubSubscription(PubsubSubscription.Type.NORMAL, projectName, subscriptionName);
330         }

1110         @Override
1111         public PCollection<T> expand(PBegin input) {
1112           if (getTopicProvider() == null && getSubscriptionProvider() == null) {
1113             throw new IllegalStateException(
1114                 "Need to set either the topic or the subscription for " + "a PubsubIO.Read transform");
1115           }
1116           if (getTopicProvider() != null && getSubscriptionProvider() != null) {
1117             throw new IllegalStateException(
1118                 "Can't set both the topic and the subscription for " + "a PubsubIO.Read transform");
1119           }
1120     
1121           if (getDeadLetterTopicProvider() != null
1122               && !(getBadRecordRouter() instanceof ThrowingBadRecordRouter)) {
1123             throw new IllegalArgumentException(
1124                 "PubSubIO cannot be configured with both a dead letter topic and a bad record router");
1125           }
1126     
1127           @Nullable
1128           ValueProvider<TopicPath> topicPath =
1129               getTopicProvider() == null
1130                   ? null
1131                   : NestedValueProvider.of(getTopicProvider(), new TopicPathTranslator());
1132           @Nullable
1133           ValueProvider<SubscriptionPath> subscriptionPath =
1134               getSubscriptionProvider() == null
1135                   ? null
1136                   : NestedValueProvider.of(getSubscriptionProvider(), new SubscriptionPathTranslator());
1137           PubsubUnboundedSource source =
1138               new PubsubUnboundedSource(
1139                   getClock(),
1140                   getPubsubClientFactory(),
1141                   null /* always get project from runtime PipelineOptions */,
1142                   topicPath,
1143                   subscriptionPath,
1144                   getTimestampAttribute(),
1145                   getIdAttribute(),
1146                   getNeedsAttributes(),
1147                   getNeedsMessageId(),
1148                   getNeedsOrderingKey());
1149     
1150           PCollection<PubsubMessage> preParse = input.apply(source);
1151           return expandReadContinued(preParse, topicPath, subscriptionPath);
1152         }

1168                 // flag that reported metrics
1169                 private final SerializableFunction<PubsubMessage, T> underlying =

1171                 private transient boolean reportedMetrics = false;
1172     
1173                 // public
1174                 @Override
1175                 public T apply(PubsubMessage input) {
1176                   if (!reportedMetrics) {
1177                     // report Lineage once
1178                     if (topicPath != null) {
1179                       TopicPath topic = topicPath.get();
1180                       if (topic != null) {
1181                         Lineage.getSources().add("pubsub", "topic", topic.getDataCatalogSegments());
1182                       }
1183                     }
1184                     if (subscriptionPath != null) {
1185                       SubscriptionPath sub = subscriptionPath.get();
1186                       if (sub != null) {
1187                         Lineage.getSources()
1188                             .add("pubsub", "subscription", sub.getDataCatalogSegments());
1189                       }
1190                     }
1191                     reportedMetrics = true;
1192                   }
1193                   return underlying.apply(input);
1194                 }
1195               };
1196           PCollection<T> read;
1197           if (getDeadLetterTopicProvider() == null
1198               && (getBadRecordRouter() instanceof ThrowingBadRecordRouter)) {
1199             read = preParse.apply(MapElements.into(typeDescriptor).via(parseFnWrapped));
1200           } else {
1201             // parse PubSub messages, separating out exceptions
1202             Result<PCollection<T>, KV<PubsubMessage, EncodableThrowable>> result =
1203                 preParse.apply(
1204                     "PubsubIO.Read/Map/Parse-Incoming-Messages",
1205                     MapElements.into(typeDescriptor)
1206                         .via(parseFnWrapped)
1207                         .exceptionsVia(new WithFailures.ThrowableHandler<PubsubMessage>() {}));
1208     
1209             // Emit parsed records
1210             read = result.output();
1211     
1212             // Send exceptions to either the bad record router or the dead letter topic
1213             if (!(getBadRecordRouter() instanceof ThrowingBadRecordRouter)) {
1214               PCollection<BadRecord> badRecords =
1215                   result
1216                       .failures()
1217                       .apply(
1218                           "Map Failures To BadRecords",
1219                           ParDo.of(new ParseReadFailuresToBadRecords(preParse.getCoder())));
1220               getBadRecordErrorHandler()
1221                   .addErrorCollection(badRecords.setCoder(BadRecord.getCoder(preParse.getPipeline())));
1222             } else {
1223               // Write out failures to the provided dead-letter topic.
1224               result
1225                   .failures()
1226                   // Since the stack trace could easily exceed Pub/Sub limits, we need to remove it from
1227                   // the attributes.
1228                   .apply(
1229                       "PubsubIO.Read/Map/Remove-Stack-Trace-Attribute",
1230                       MapElements.into(new TypeDescriptor<KV<PubsubMessage, Map<String, String>>>() {})
1231                           .via(
1232                               kv -> {
1233                                 PubsubMessage message = kv.getKey();
1234                                 String messageId =
1235                                     message.getMessageId() == null ? "<null>" : message.getMessageId();
1236                                 Throwable throwable = kv.getValue().throwable();
1237     
1238                                 // In order to stay within Pub/Sub limits, we aren't adding the stack
1239                                 // trace to the attributes. Therefore, we need to log the throwable.
1240                                 LOG.error(
1241                                     "Error parsing Pub/Sub message with id '{}'", messageId, throwable);
1242     
1243                                 ImmutableMap<String, String> attributes =
1244                                     ImmutableMap.<String, String>builder()
1245                                         .put("exceptionClassName", throwable.getClass().getName())
1246                                         .put("exceptionMessage", throwable.getMessage())
1247                                         .put("pubsubMessageId", messageId)
1248                                         .build();
1249     
1250                                 return KV.of(kv.getKey(), attributes);
1251                               }))
1252                   .apply(
1253                       "PubsubIO.Read/Map/Create-Dead-Letter-Payload",
1254                       MapElements.into(TypeDescriptor.of(PubsubMessage.class))
1255                           .via(kv -> new PubsubMessage(kv.getKey().getPayload(), kv.getValue())))
1256                   .apply(
1257                       writeMessages()
1258                           .to(getDeadLetterTopicProvider().get().asPath())
1259                           .withClientFactory(getPubsubClientFactory()));
1260             }
1261           }
1262           return read.setCoder(getCoder());
1263         }

1500         @Override
1501         public PDone expand(PCollection<T> input) {
1502           if (getTopicProvider() == null && !getDynamicDestinations()) {
1503             throw new IllegalStateException(
1504                 "need to set the topic of a PubsubIO.Write transform if not using "
1505                     + "dynamic topic destinations.");
1506           }
1507     
1508           SerializableFunction<ValueInSingleWindow<T>, PubsubIO.PubsubTopic> topicFunction =
1509               getTopicFunction();
1510           if (topicFunction == null && getTopicProvider() != null) {
1511             topicFunction = v -> getTopicProvider().get();
1512           }
1513           int maxMessageSize = PUBSUB_MESSAGE_MAX_TOTAL_SIZE;
1514           if (input.isBounded() == PCollection.IsBounded.BOUNDED) {
1515             maxMessageSize =
1516                 Math.min(
1517                     maxMessageSize,
1518                     MoreObjects.firstNonNull(
1519                         getMaxBatchBytesSize(), MAX_PUBLISH_BATCH_BYTE_SIZE_DEFAULT));
1520           }
1521           TupleTag<PubsubMessage> pubsubMessageTupleTag = new TupleTag<>();
1522           PCollectionTuple pubsubMessageTuple =
1523               input.apply(
1524                   ParDo.of(
1525                           new PreparePubsubWriteDoFn<>(
1526                               getFormatFn(),
1527                               topicFunction,
1528                               maxMessageSize,
1529                               getBadRecordRouter(),
1530                               input.getCoder(),
1531                               pubsubMessageTupleTag))
1532                       .withOutputTags(pubsubMessageTupleTag, TupleTagList.of(BAD_RECORD_TAG)));
1533     
1534           getBadRecordErrorHandler()
1535               .addErrorCollection(
1536                   pubsubMessageTuple
1537                       .get(BAD_RECORD_TAG)
1538                       .setCoder(BadRecord.getCoder(input.getPipeline())));
1539           PCollection<PubsubMessage> pubsubMessages =
1540               pubsubMessageTuple.get(pubsubMessageTupleTag).setCoder(PubsubMessageWithTopicCoder.of());
1541           switch (input.isBounded()) {
1542             case BOUNDED:
1543               pubsubMessages.apply(
1544                   ParDo.of(
1545                       new PubsubBoundedWriter(
1546                           MoreObjects.firstNonNull(getMaxBatchSize(), MAX_PUBLISH_BATCH_SIZE),
1547                           MoreObjects.firstNonNull(
1548                               getMaxBatchBytesSize(), MAX_PUBLISH_BATCH_BYTE_SIZE_DEFAULT))));
1549               return PDone.in(input.getPipeline());
1550             case UNBOUNDED:
1551               return pubsubMessages.apply(
1552                   new PubsubUnboundedSink(
1553                       getPubsubClientFactory(),
1554                       getTopicProvider() != null
1555                           ? NestedValueProvider.of(getTopicProvider(), new TopicPathTranslator())
1556                           : null,
1557                       getTimestampAttribute(),
1558                       getIdAttribute(),
1559                       100 /* numShards */,
1560                       MoreObjects.firstNonNull(
1561                           getMaxBatchSize(), PubsubUnboundedSink.DEFAULT_PUBLISH_BATCH_SIZE),
1562                       MoreObjects.firstNonNull(
1563                           getMaxBatchBytesSize(), PubsubUnboundedSink.DEFAULT_PUBLISH_BATCH_BYTES),
1564                       getPubsubRootUrl()));
1565           }
1566           throw new RuntimeException(); // cases are exhaustive.
1567         }

1576         /**
1577          * Writer to Pubsub which batches messages from bounded collections.
1578          *
1579          * <p>Public so can be suppressed by runners.
1580          */
1581         public class PubsubBoundedWriter extends DoFn<PubsubMessage, Void> {
1582           private class OutgoingData {
1583             List<OutgoingMessage> messages;
1584             long bytes;
1585     
1586             OutgoingData() {
1587               this.messages = Lists.newArrayList();
1588               this.bytes = 0;
1589             }
1590           }
1591     
1592           private transient Map<PubsubTopic, OutgoingData> output;
1593     
1594           private transient PubsubClient pubsubClient;
1595     
1596           private int maxPublishBatchByteSize;
1597           private int maxPublishBatchSize;
1598     
1599           PubsubBoundedWriter(int maxPublishBatchSize, int maxPublishBatchByteSize) {
1600             this.maxPublishBatchSize = maxPublishBatchSize;
1601             this.maxPublishBatchByteSize = maxPublishBatchByteSize;
1602           }
1603     
1604           PubsubBoundedWriter() {
1605             this(MAX_PUBLISH_BATCH_SIZE, MAX_PUBLISH_BATCH_BYTE_SIZE_DEFAULT);
1606           }
1607     
1608           @StartBundle
1609           public void startBundle(StartBundleContext c) throws IOException {
1610             this.output = Maps.newHashMap();
1611     
1612             // NOTE: idAttribute is ignored.
1613             this.pubsubClient =
1614                 getPubsubClientFactory()
1615                     .newClient(
1616                         getTimestampAttribute(), null, c.getPipelineOptions().as(PubsubOptions.class));
1617           }
1618     
1619           @ProcessElement
1620           public void processElement(@Element PubsubMessage message, @Timestamp Instant timestamp)
1621               throws IOException, SizeLimitExceededException {
1622             // Validate again here just as a sanity check.
1623             PreparePubsubWriteDoFn.validatePubsubMessageSize(message, maxPublishBatchByteSize);
1624             byte[] payload = message.getPayload();
1625             int messageSize = payload.length;
1626     
1627             PubsubTopic pubsubTopic;
1628             if (getTopicProvider() != null) {
1629               pubsubTopic = getTopicProvider().get();
1630             } else {
1631               pubsubTopic =
1632                   PubsubTopic.fromPath(Preconditions.checkArgumentNotNull(message.getTopic()));
1633             }
1634             // Checking before adding the message stops us from violating max batch size or bytes
1635             OutgoingData currentTopicOutput =
1636                 output.computeIfAbsent(pubsubTopic, t -> new OutgoingData());
1637             if (currentTopicOutput.messages.size() >= maxPublishBatchSize
1638                 || (!currentTopicOutput.messages.isEmpty()
1639                     && (currentTopicOutput.bytes + messageSize) >= maxPublishBatchByteSize)) {
1640               publish(pubsubTopic, currentTopicOutput.messages);
1641               currentTopicOutput.messages.clear();
1642               currentTopicOutput.bytes = 0;
1643             }
1644     
1645             Map<String, String> attributes = message.getAttributeMap();
1646             String orderingKey = message.getOrderingKey();
1647     
1648             com.google.pubsub.v1.PubsubMessage.Builder msgBuilder =
1649                 com.google.pubsub.v1.PubsubMessage.newBuilder()
1650                     .setData(ByteString.copyFrom(payload))
1651                     .putAllAttributes(attributes);
1652     
1653             if (orderingKey != null) {
1654               msgBuilder.setOrderingKey(orderingKey);
1655             }
1656     
1657             // NOTE: The record id is always null.
1658             currentTopicOutput.messages.add(
1659                 OutgoingMessage.of(
1660                     msgBuilder.build(), timestamp.getMillis(), null, message.getTopic()));
1661             currentTopicOutput.bytes += messageSize;
1662           }
1663     
1664           @FinishBundle
1665           public void finishBundle() throws IOException {
1666             for (Map.Entry<PubsubTopic, OutgoingData> entry : output.entrySet()) {
1667               publish(entry.getKey(), entry.getValue().messages);
1668             }
1669             output = null;
1670             pubsubClient.close();
1671             pubsubClient = null;
1672           }
1673     
1674           private void publish(PubsubTopic topic, List<OutgoingMessage> messages) throws IOException {
1675             int n =
1676                 pubsubClient.publish(
1677                     PubsubClient.topicPathFromName(topic.project, topic.topic), messages);
1678             checkState(n == messages.size());
1679           }
