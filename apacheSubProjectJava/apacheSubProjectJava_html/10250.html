59       private static class WindowedCount extends PTransform<PCollection<String>, PCollection<String>> {
60     
61         private static final class FormatCountsDoFn extends DoFn<KV<String, Long>, String> {
62           @ProcessElement
63           public void processElement(ProcessContext c, BoundedWindow window) {
64             c.output(
65                 c.element().getKey()
66                     + ":"
67                     + c.element().getValue()
68                     + ":"
69                     + c.timestamp().getMillis()
70                     + ":"
71                     + window);
72           }
73         }
74     
75         private WindowFn<? super String, ?> windowFn;
76     
77         public WindowedCount(WindowFn<? super String, ?> windowFn) {
78           this.windowFn = windowFn;
79         }
80     
81         @Override
82         public PCollection<String> expand(PCollection<String> in) {
83           return in.apply(
84                   "Window",
85                   Window.<String>into(windowFn).withTimestampCombiner(TimestampCombiner.EARLIEST))
86               .apply(Count.perElement())
87               .apply("FormatCounts", ParDo.of(new FormatCountsDoFn()))
88               .setCoder(StringUtf8Coder.of());
89         }
90       }
91     
92       private String output(String value, int count, int timestamp, int windowStart, int windowEnd) {
93         return value
94             + ":"
95             + count
96             + ":"
97             + timestamp
98             + ":["
99             + new Instant(windowStart)
100             + ".."
101             + new Instant(windowEnd)
102             + ")";
103       }

105       @Test
106       @Category(ValidatesRunner.class)
107       public void testPartitioningWindowing() {
108         PCollection<String> input =
109             p.apply(
110                 Create.timestamped(
111                     TimestampedValue.of("a", new Instant(1)),
112                     TimestampedValue.of("b", new Instant(2)),
113                     TimestampedValue.of("b", new Instant(3)),
114                     TimestampedValue.of("c", new Instant(11)),
115                     TimestampedValue.of("d", new Instant(11))));
116     
117         PCollection<String> output =
118             input.apply(new WindowedCount(FixedWindows.of(Duration.millis(10))));
119     
120         PAssert.that(output)
121             .containsInAnyOrder(
122                 output("a", 1, 1, 0, 10),
123                 output("b", 2, 2, 0, 10),
124                 output("c", 1, 11, 10, 20),
125                 output("d", 1, 11, 10, 20));
126     
127         p.run();
128       }

216       static class ExtractWordsWithTimestampsFn extends DoFn<String, String> {
217         @ProcessElement
