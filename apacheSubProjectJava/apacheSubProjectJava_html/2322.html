93       /**
94        * This is the entry point to convert user-provided specs to {@link ResourceId ResourceIds}.
95        * Callers should use {@link #match} to resolve users specs ambiguities before calling other
96        * methods.
97        *
98        * <p>Implementation handles the following ambiguities of a user-provided spec:
99        *
100        * <ol>
101        *   <li>{@code spec} could be a glob or a uri. {@link #match} should be able to tell and choose
102        *       efficient implementations.
103        *   <li>The user-provided {@code spec} might refer to files or directories. It is common that
104        *       users that wish to indicate a directory will omit the trailing path delimiter, such as
105        *       {@code "/tmp/dir"} in Linux. The {@link FileSystem} should be able to recognize a
106        *       directory with the trailing path delimiter omitted, but should always return a correct
107        *       {@link ResourceId} (e.g., {@code "/tmp/dir/"} inside the returned {@link MatchResult}.
108        * </ol>
109        *
110        * <p>All {@link FileSystem} implementations should support glob in the final hierarchical path
111        * component of {@link ResourceId}. This allows SDK libraries to construct file system agnostic
112        * spec. {@link FileSystem FileSystems} can support additional patterns for user-provided specs.
113        *
114        * <p>In case the spec schemes don't match any known {@link FileSystem} implementations,
115        * FileSystems will attempt to use {@link LocalFileSystem} to resolve a path.
116        *
117        * <p>Specs that do not match any resources are treated according to {@link
118        * EmptyMatchTreatment#DISALLOW}.
119        *
120        * @return {@code List<MatchResult>} in the same order of the input specs.
121        * @throws IllegalArgumentException if specs are invalid -- empty or have different schemes.
122        * @throws IOException if all specs failed to match due to issues like: network connection,
123        *     authorization. Exception for individual spec is deferred until callers retrieve metadata
124        *     with {@link MatchResult#metadata()}.
125        */
126       public static List<MatchResult> match(List<String> specs) throws IOException {
127         return getFileSystemInternal(getOnlyScheme(specs)).match(specs);
128       }
129     
130       /** Like {@link #match(List)}, but with a configurable {@link EmptyMatchTreatment}. */
131       public static List<MatchResult> match(List<String> specs, EmptyMatchTreatment emptyMatchTreatment)
132           throws IOException {
133         List<MatchResult> matches = getFileSystemInternal(getOnlyScheme(specs)).match(specs);
134         List<MatchResult> res = Lists.newArrayListWithExpectedSize(matches.size());
135         for (int i = 0; i < matches.size(); i++) {
136           res.add(maybeAdjustEmptyMatchResult(specs.get(i), matches.get(i), emptyMatchTreatment));
137         }
138         return res;
139       }

144        * <p>The function {@link #match(List)} is preferred when matching multiple patterns, as it allows

147       public static MatchResult match(String spec) throws IOException {

149         verify(

164       private static MatchResult maybeAdjustEmptyMatchResult(
165           String spec, MatchResult res, EmptyMatchTreatment emptyMatchTreatment) throws IOException {
166         if (res.status() == Status.NOT_FOUND
167             || (res.status() == Status.OK && res.metadata().isEmpty())) {
168           boolean notFoundAllowed =
169               emptyMatchTreatment == EmptyMatchTreatment.ALLOW
170                   || (hasGlobWildcard(spec)
171                       && emptyMatchTreatment == EmptyMatchTreatment.ALLOW_IF_WILDCARD);
172           return notFoundAllowed
173               ? MatchResult.create(Status.OK, Collections.emptyList())
174               : MatchResult.create(
175                   Status.NOT_FOUND, new FileNotFoundException("No files matched spec: " + spec));
176         }
177         return res;
178       }

180       /**
181        * Returns the {@link Metadata} for a single file resource. Expects a resource specification
182        * {@code spec} that matches a single result.
183        *
184        * @param spec a resource specification that matches exactly one result.
185        * @return the {@link Metadata} for the specified resource.
186        * @throws FileNotFoundException if the file resource is not found.
187        * @throws IOException in the event of an error in the inner call to {@link #match}, or if the
188        *     given spec does not match exactly 1 result.
189        */
190       public static Metadata matchSingleFileSpec(String spec) throws IOException {
191         List<MatchResult> matches = FileSystems.match(Collections.singletonList(spec));
192         MatchResult matchResult = Iterables.getOnlyElement(matches);
193         if (matchResult.status() == Status.NOT_FOUND) {
194           throw new FileNotFoundException(String.format("File spec %s not found", spec));
195         } else if (matchResult.status() != Status.OK) {
196           throw new IOException(
197               String.format("Error matching file spec %s: status %s", spec, matchResult.status()));
198         } else {
199           List<Metadata> metadata = matchResult.metadata();
200           if (metadata.size() != 1) {
201             throw new IOException(
202                 String.format(
203                     "Expecting spec %s to match exactly one file, but matched %s: %s",
204                     spec, metadata.size(), metadata));
205           }
206           return metadata.get(0);
207         }
208       }

262       /**
263        * Copies a {@link List} of file-like resources from one location to another.
264        *
265        * <p>The number of source resources must equal the number of destination resources. Destination
266        * resources will be created recursively.
267        *
268        * <p>{@code srcResourceIds} and {@code destResourceIds} must have the same scheme.
269        *
270        * <p>It doesn't support copying globs.
271        *
272        * @param srcResourceIds the references of the source resources
273        * @param destResourceIds the references of the destination resources
274        */
275       public static void copy(
276           List<ResourceId> srcResourceIds, List<ResourceId> destResourceIds, MoveOptions... moveOptions)
277           throws IOException {
278         validateSrcDestLists(srcResourceIds, destResourceIds);
279         if (srcResourceIds.isEmpty()) {
280           return;
281         }
282         FileSystem fileSystem = getFileSystemInternal(srcResourceIds.iterator().next().getScheme());
283         FilterResult filtered = filterFiles(fileSystem, srcResourceIds, destResourceIds, moveOptions);
284         if (!filtered.resultSources.isEmpty()) {
285           fileSystem.copy(filtered.resultSources, filtered.resultDestinations);
286         }
287       }

289       /**
290        * Renames a {@link List} of file-like resources from one location to another.
291        *
292        * <p>The number of source resources must equal the number of destination resources. Destination
293        * resources will be created recursively.
294        *
295        * <p>{@code srcResourceIds} and {@code destResourceIds} must have the same scheme.

298        *
299        * <p>Src files will be removed, even if the copy is skipped due to specified move options.
300        *

302        * @param destResourceIds the references of the destination resources

305           List<ResourceId> srcResourceIds, List<ResourceId> destResourceIds, MoveOptions... moveOptions)
306           throws IOException {
307         validateSrcDestLists(srcResourceIds, destResourceIds);
308         if (srcResourceIds.isEmpty()) {
309           return;
310         }
311         renameInternal(
312             getFileSystemInternal(srcResourceIds.iterator().next().getScheme()),
313             srcResourceIds,
314             destResourceIds,
315             moveOptions);
316       }

318       @VisibleForTesting
319       static void renameInternal(
320           FileSystem fileSystem,
321           List<ResourceId> srcResourceIds,
322           List<ResourceId> destResourceIds,
323           MoveOptions... moveOptions)
324           throws IOException {
325         try {
326           fileSystem.rename(srcResourceIds, destResourceIds, moveOptions);
327         } catch (UnsupportedOperationException e) {
328           // Some file systems do not yet support specifying the move options. Instead we
329           // perform filtering using match calls before renaming.
330           FilterResult filtered = filterFiles(fileSystem, srcResourceIds, destResourceIds, moveOptions);
331           if (!filtered.resultSources.isEmpty()) {
332             fileSystem.rename(filtered.resultSources, filtered.resultDestinations);
333           }
334           if (!filtered.filteredExistingSrcs.isEmpty()) {
335             fileSystem.delete(filtered.filteredExistingSrcs);
336           }
337         }
338       }

340       /**
341        * Deletes a collection of resources.
342        *
343        * <p>{@code resourceIds} must have the same scheme.
344        *
345        * @param resourceIds the references of the resources to delete.
346        */
347       public static void delete(Collection<ResourceId> resourceIds, MoveOptions... moveOptions)
348           throws IOException {
349         if (resourceIds.isEmpty()) {
350           // Short-circuit.
351           return;
352         }
353     
354         Collection<ResourceId> resourceIdsToDelete;
355         if (Sets.newHashSet(moveOptions)
356             .contains(MoveOptions.StandardMoveOptions.IGNORE_MISSING_FILES)) {
357           resourceIdsToDelete =
358               FluentIterable.from(matchResources(Lists.newArrayList(resourceIds)))
359                   .filter(matchResult -> !matchResult.status().equals(Status.NOT_FOUND))
360                   .transformAndConcat(
361                       new Function<MatchResult, Iterable<Metadata>>() {
362                         @SuppressFBWarnings(
363                             value = "NP_METHOD_PARAMETER_TIGHTENS_ANNOTATION",
364                             justification = "https://github.com/google/guava/issues/920")
365                         @Nonnull
366                         @Override
367                         public Iterable<Metadata> apply(@Nonnull MatchResult input) {
368                           try {
369                             return Lists.newArrayList(input.metadata());
370                           } catch (IOException e) {
371                             throw new RuntimeException(
372                                 String.format("Failed to get metadata from MatchResult: %s.", input),
373                                 e);
374                           }
375                         }
376                       })
377                   .transform(
378                       new Function<Metadata, ResourceId>() {
379                         @SuppressFBWarnings(
380                             value = "NP_METHOD_PARAMETER_TIGHTENS_ANNOTATION",
381                             justification = "https://github.com/google/guava/issues/920")
382                         @Nonnull
383                         @Override
384                         public ResourceId apply(@Nonnull Metadata input) {
385                           return input.resourceId();
386                         }
387                       })
388                   .toList();
389         } else {
390           resourceIdsToDelete = resourceIds;
391         }
392         if (resourceIdsToDelete.isEmpty()) {
393           return;
394         }
395         getFileSystemInternal(resourceIdsToDelete.iterator().next().getScheme())
396             .delete(resourceIdsToDelete);
397       }

399       private static class FilterResult {
400         public List<ResourceId> resultSources = new ArrayList();
401         public List<ResourceId> resultDestinations = new ArrayList();
402         public List<ResourceId> filteredExistingSrcs = new ArrayList();
403       };
404     
405       private static FilterResult filterFiles(
406           FileSystem fileSystem,
407           List<ResourceId> srcResourceIds,
408           List<ResourceId> destResourceIds,
409           MoveOptions... moveOptions)
410           throws IOException {
411         FilterResult result = new FilterResult();
412         if (moveOptions.length == 0 || srcResourceIds.isEmpty()) {
413           // Nothing will be filtered.
414           result.resultSources = srcResourceIds;
415           result.resultDestinations = destResourceIds;
416           return result;
417         }
418         Set<MoveOptions> moveOptionSet = Sets.newHashSet(moveOptions);
419         final boolean ignoreMissingSrc =
420             moveOptionSet.contains(StandardMoveOptions.IGNORE_MISSING_FILES);
421         final boolean skipExistingDest =
422             moveOptionSet.contains(StandardMoveOptions.SKIP_IF_DESTINATION_EXISTS);
423         final int size = srcResourceIds.size();
424     
425         // Match necessary srcs and dests with a single match call.
426         List<ResourceId> matchResources = new ArrayList<>();
427         if (ignoreMissingSrc) {
428           matchResources.addAll(srcResourceIds);
429         }
430         if (skipExistingDest) {
431           matchResources.addAll(destResourceIds);
432         }
433         List<MatchResult> matchResults =
434             fileSystem.match(
435                 FluentIterable.from(matchResources).transform(ResourceId::toString).toList());
436         List<MatchResult> matchSrcResults = ignoreMissingSrc ? matchResults.subList(0, size) : null;
437         List<MatchResult> matchDestResults =
438             skipExistingDest
439                 ? matchResults.subList(matchResults.size() - size, matchResults.size())
440                 : null;
441     
442         for (int i = 0; i < size; ++i) {
443           if (matchSrcResults != null && matchSrcResults.get(i).status().equals(Status.NOT_FOUND)) {
444             // If the source is not found, and we are ignoring missing source files, then we skip it.
445             continue;
446           }
447           if (matchDestResults != null
448               && matchSrcResults != null
449               && matchDestResults.get(i).status().equals(Status.OK)
450               && checksumMatch(
451                   matchDestResults.get(i).metadata().get(0),
452                   matchSrcResults.get(i).metadata().get(0))) {
453             // If the destination exists, and we are skipping when destinations exist, then we skip
454             // the copy but note that the source exists in case it should be deleted.
455             result.filteredExistingSrcs.add(srcResourceIds.get(i));
456             continue;
457           }
458           result.resultSources.add(srcResourceIds.get(i));
459           result.resultDestinations.add(destResourceIds.get(i));
460         }
461         return result;
462       }
463     
464       private static boolean checksumMatch(MatchResult.Metadata first, MatchResult.Metadata second) {
465         return first.checksum() != null && first.checksum().equals(second.checksum());
466       }

468       private static void validateSrcDestLists(
469           List<ResourceId> srcResourceIds, List<ResourceId> destResourceIds) {
470         checkArgument(
471             srcResourceIds.size() == destResourceIds.size(),
472             "Number of source resource ids %s must equal number of destination resource ids %s",
473             srcResourceIds.size(),
474             destResourceIds.size());
475     
476         if (srcResourceIds.isEmpty()) {
477           // nothing more to validate.
478           return;
479         }
480     
481         Set<String> schemes =
482             FluentIterable.from(srcResourceIds)
483                 .append(destResourceIds)
484                 .transform(ResourceId::getScheme)
485                 .toSet();
486         checkArgument(
487             schemes.size() == 1,
488             String.format(
489                 "Expect srcResourceIds and destResourceIds have the same scheme, but received %s.",
490                 Joiner.on(", ").join(schemes)));
491       }

493       private static String getOnlyScheme(List<String> specs) {
494         checkArgument(!specs.isEmpty(), "Expect specs are not empty.");
495         Set<String> schemes = FluentIterable.from(specs).transform(FileSystems::parseScheme).toSet();
496         return Iterables.getOnlyElement(schemes);
497       }

499       private static String parseScheme(String spec) {
500         // The spec is almost, but not quite, a URI. In particular,
501         // the reserved characters '[', ']', and '?' have meanings that differ
502         // from their use in the URI spec. ('*' is not reserved).
503         // Here, we just need the scheme, which is so circumscribed as to be
504         // very easy to extract with a regex.
505         Matcher matcher = FILE_SCHEME_PATTERN.matcher(spec);
506     
507         if (!matcher.matches()) {
508           return DEFAULT_SCHEME;
509         } else {
510           return matcher.group("scheme").toLowerCase();
511         }
512       }

514       /** Internal method to get {@link FileSystem} for {@code scheme}. */
515       @VisibleForTesting
516       static FileSystem getFileSystemInternal(String scheme) {
517         String lowerCaseScheme = scheme.toLowerCase();
518         Map<String, FileSystem> schemeToFileSystem = SCHEME_TO_FILESYSTEM.get();
519         FileSystem rval = schemeToFileSystem.get(lowerCaseScheme);
520         if (rval == null) {
521           throw new IllegalArgumentException("No filesystem found for scheme " + scheme);
522         }
523         return rval;
524       }

528       /**
529        * Sets the default configuration in workers.
530        *
531        * <p>It will be used in {@link FileSystemRegistrar FileSystemRegistrars} for all schemes.
532        *
533        * <p>This is expected only to be used by runners after {@code Pipeline.run}, or in tests.
534        */
535       @Internal
536       public static void setDefaultPipelineOptions(PipelineOptions options) {
537         checkNotNull(options, "options");
538         long id = options.getOptionsId();
539         int nextRevision = options.revision();
540     
541         while (true) {
542           KV<Long, Integer> revision = FILESYSTEM_REVISION.get();
543           // only update file systems if the pipeline changed or the options revision increased
544           if (revision != null && revision.getKey().equals(id) && revision.getValue() >= nextRevision) {
545             return;
546           }
547     
548           if (FILESYSTEM_REVISION.compareAndSet(revision, KV.of(id, nextRevision))) {
549             Set<FileSystemRegistrar> registrars =
550                 Sets.newTreeSet(ReflectHelpers.ObjectsClassComparator.INSTANCE);
551             registrars.addAll(
552                 Lists.newArrayList(
553                     ServiceLoader.load(FileSystemRegistrar.class, ReflectHelpers.findClassLoader())));
554     
555             SCHEME_TO_FILESYSTEM.set(verifySchemesAreUnique(options, registrars));
556             return;
557           }
558         }
559       }

561       @VisibleForTesting
562       static Map<String, FileSystem> verifySchemesAreUnique(
563           PipelineOptions options, Set<FileSystemRegistrar> registrars) {
564         Multimap<String, FileSystem> fileSystemsBySchemes =
565             TreeMultimap.create(Ordering.<String>natural(), Ordering.arbitrary());
566     
567         for (FileSystemRegistrar registrar : registrars) {
568           for (FileSystem fileSystem : registrar.fromOptions(options)) {
569             fileSystemsBySchemes.put(fileSystem.getScheme(), fileSystem);
570           }
571         }
572         for (Entry<String, Collection<FileSystem>> entry : fileSystemsBySchemes.asMap().entrySet()) {
573           if (entry.getValue().size() > 1) {
574             String conflictingFileSystems =
575                 Joiner.on(", ")
576                     .join(
577                         FluentIterable.from(entry.getValue())
578                             .transform(input -> input.getClass().getName())
579                             .toSortedList(Ordering.natural()));
580             throw new IllegalStateException(
581                 String.format(
582                     "Scheme: [%s] has conflicting filesystems: [%s]",
583                     entry.getKey(), conflictingFileSystems));
584           }
585         }
586     
587         ImmutableMap.Builder<String, FileSystem> schemeToFileSystem = ImmutableMap.builder();
588         for (Entry<String, FileSystem> entry : fileSystemsBySchemes.entries()) {
589           schemeToFileSystem.put(entry.getKey(), entry.getValue());
590         }
591         return schemeToFileSystem.build();
592       }
593     
594       /**
595        * Returns a new {@link ResourceId} that represents the named resource of a type corresponding to
596        * the resource type.
597        *
598        * <p>The supplied {@code singleResourceSpec} is expected to be in a proper format, including any
599        * necessary escaping, for the underlying {@link FileSystem}.
600        *
601        * <p>This function may throw an {@link IllegalArgumentException} if given an invalid argument,
602        * such as when the specified {@code singleResourceSpec} is not a valid resource name.
603        */
604       public static ResourceId matchNewResource(String singleResourceSpec, boolean isDirectory) {
605         return getFileSystemInternal(parseScheme(singleResourceSpec))
606             .matchNewResource(singleResourceSpec, isDirectory);
607       }
608     
609       /**
610        * Returns a new {@link ResourceId} that represents the named directory resource.
611        *
612        * @param singleResourceSpec the root directory, for example "/abc"
613        * @param baseNames a list of named directory, for example ["d", "e", "f"]
614        * @return the ResourceId for the resolved directory. In same example as above, it corresponds to
615        *     "/abc/d/e/f".
616        */
617       public static ResourceId matchNewDirectory(String singleResourceSpec, String... baseNames) {
618         ResourceId currentDir = matchNewResource(singleResourceSpec, true);
619         for (String dir : baseNames) {
620           currentDir = currentDir.resolve(dir, StandardResolveOptions.RESOLVE_DIRECTORY);
621         }
622         return currentDir;
623       }
