134         @Test
135         public void testAvroIOGetName() {
136           assertEquals("AvroIO.Read", AvroIO.read(String.class).from("/tmp/foo*/baz").getName());
137           assertEquals("AvroIO.Write", AvroIO.write(String.class).to("/tmp/foo/baz").getName());
138         }
139     
140         @Test
141         public void testWriteWithDefaultCodec() {
142           AvroIO.Write<String> write = AvroIO.write(String.class).to("/tmp/foo/baz");
143           assertEquals(CodecFactory.snappyCodec().toString(), write.inner.getCodec().toString());
144         }
145     
146         @Test
147         public void testWriteWithCustomCodec() {
148           AvroIO.Write<String> write =
149               AvroIO.write(String.class).to("/tmp/foo/baz").withCodec(CodecFactory.snappyCodec());
150           assertEquals(SNAPPY_CODEC, write.inner.getCodec().toString());
151         }
152     
153         @Test
154         public void testWriteWithSerDeCustomDeflateCodec() {
155           AvroIO.Write<String> write =
156               AvroIO.write(String.class).to("/tmp/foo/baz").withCodec(CodecFactory.deflateCodec(9));
157     
158           assertEquals(
159               CodecFactory.deflateCodec(9).toString(),
160               SerializableUtils.clone(write.inner.getCodec()).getCodec().toString());
161         }
162     
163         @Test
164         public void testWriteWithSerDeCustomXZCodec() {
165           AvroIO.Write<String> write =
166               AvroIO.write(String.class).to("/tmp/foo/baz").withCodec(CodecFactory.xzCodec(9));
167     
168           assertEquals(
169               CodecFactory.xzCodec(9).toString(),
170               SerializableUtils.clone(write.inner.getCodec()).getCodec().toString());
171         }
