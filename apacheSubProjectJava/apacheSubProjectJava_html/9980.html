216         private Matcher<Iterable<? extends Integer>> quantileMatcher(
217             int size, int numQuantiles, int absoluteError) {
218           List<Matcher<? super Integer>> quantiles = new ArrayList<>();
219           quantiles.add(CoreMatchers.is(0));
220           for (int k = 1; k < numQuantiles - 1; k++) {
221             int expected = (int) (((double) (size - 1)) * k / (numQuantiles - 1));
222             quantiles.add(new Between<>(expected - absoluteError, expected + absoluteError));
223           }
224           quantiles.add(CoreMatchers.is(size - 1));
225           return contains(quantiles);
226         }
227     
228         private static class Between<T extends Comparable<T>> extends TypeSafeDiagnosingMatcher<T> {
229           private final T min;
230           private final T max;
231     
232           private Between(T min, T max) {
233             this.min = min;
234             this.max = max;
235           }
236     
237           @Override
238           public void describeTo(Description description) {
239             description.appendText("is between " + min + " and " + max);
240           }
241     
242           @Override
243           protected boolean matchesSafely(T item, Description mismatchDescription) {
244             return min.compareTo(item) <= 0 && item.compareTo(max) <= 0;
245           }
246         }
247     
248         private static class DescendingIntComparator implements SerializableComparator<Integer> {
249           @Override
250           public int compare(Integer o1, Integer o2) {
251             return o2.compareTo(o1);
252           }
253         }
254     
255         private static class OrderByLength implements Comparator<String>, Serializable {
256           @Override
257           public int compare(String a, String b) {
258             if (a.length() != b.length()) {
259               return a.length() - b.length();
260             } else {
261               return a.compareTo(b);
262             }
263           }
264         }

279       /** Tests to ensure we are calculating the optimal buffers. */
280       @RunWith(Parameterized.class)
281       public static class BufferTests {
282     
283         private final double epsilon;
284         private final long maxInputSize;
285         private final int expectedNumBuffers;
286         private final int expectedBufferSize;
287         private final ApproximateQuantilesCombineFn<?, ?> combineFn;
288     
289         /**
290          * Test data taken from "Munro-Paterson Algorithm" reference values table of "Approximate
291          * Medians and other Quantiles in One Pass and with Limited Memory" paper.
292          *
293          * @see ApproximateQuantilesCombineFn for paper reference.
294          */
295         private static final double[] epsilons = new double[] {0.1, 0.05, 0.01, 0.005, 0.001};
296     
297         private static final int[] maxElementExponents = new int[] {5, 6, 7, 8, 9};
298     
299         private static final int[][] expectedNumBuffersValues =
300             new int[][] {
301               {11, 14, 17, 21, 24},
302               {11, 14, 17, 20, 23},
303               {9, 11, 14, 17, 21},
304               {8, 11, 14, 17, 20},
305               {6, 9, 11, 14, 17},
306             };
307     
308         private static final int[][] expectedBufferSizeValues =
309             new int[][] {
310               {98, 123, 153, 96, 120},
311               {98, 123, 153, 191, 239},
312               {391, 977, 1221, 1526, 954},
313               {782, 977, 1221, 1526, 1908},
314               {3125, 3907, 9766, 12208, 15259},
315             };
316     
317         @Parameterized.Parameters(name = "{index}: epsilon = {0}, maxInputSize = {1}")
318         public static Collection<Object[]> data() {
319           Collection<Object[]> testData = Lists.newArrayList();
320           for (int i = 0; i < epsilons.length; i++) {
321             for (int j = 0; j < maxElementExponents.length; j++) {
322               testData.add(
323                   new Object[] {
324                     epsilons[i],
325                     (long) Math.pow(10, maxElementExponents[j]),
326                     expectedNumBuffersValues[i][j],
327                     expectedBufferSizeValues[i][j]
328                   });
329             }
330           }
331     
332           return testData;
333         }
334     
335         public BufferTests(
336             Double epsilon, Long maxInputSize, Integer expectedNumBuffers, Integer expectedBufferSize) {
337           this.epsilon = epsilon;
338           this.maxInputSize = maxInputSize;
339           this.expectedNumBuffers = expectedNumBuffers;
340           this.expectedBufferSize = expectedBufferSize;
341     
342           this.combineFn =
343               ApproximateQuantilesCombineFn.create(10, new Top.Natural<Long>(), maxInputSize, epsilon);
344         }
345     
346         /** Verify the buffers are efficiently calculated according to the reference table values. */
347         @Test
348         public void testEfficiency() {
349           assertEquals("Number of buffers", expectedNumBuffers, combineFn.getNumBuffers());
350           assertEquals("Buffer size", expectedBufferSize, combineFn.getBufferSize());
351         }
352     
353         /** Verify that buffers are correct according to the two constraint equations. */
354         @Test
355         public void testCorrectness() {
356           int b = combineFn.getNumBuffers();
357           int k = combineFn.getBufferSize();
358           long n = this.maxInputSize;
359     
360           assertThat(
361               "(b-2)2^(b-2) + 1/2 <= eN",
362               (b - 2) * (1 << (b - 2)) + 0.5,
363               Matchers.lessThanOrEqualTo(this.epsilon * n));
364           assertThat(
365               "k2^(b-1) >= N", Math.pow(k * 2, b - 1), Matchers.greaterThanOrEqualTo((double) n));
366         }
367       }
