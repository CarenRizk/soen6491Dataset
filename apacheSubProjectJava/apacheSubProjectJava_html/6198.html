66       /**
67        * Returns a {@code PTransform} that takes a {@code PCollection<T>} and returns a {@code
68        * PCollection<List<T>>} whose single value is a {@code List} of the approximate {@code N}-tiles
69        * of the elements of the input {@code PCollection}. This gives an idea of the distribution of the
70        * input elements.
71        *
72        * <p>The computed {@code List} is of size {@code numQuantiles}, and contains the input elements'
73        * minimum value, {@code numQuantiles-2} intermediate values, and maximum value, in sorted order,
74        * using the given {@code Comparator} to order values. To compute traditional {@code N}-tiles, one
75        * should use {@code ApproximateQuantiles.globally(N+1, compareFn)}.
76        *
77        * <p>If there are fewer input elements than {@code numQuantiles}, then the result {@code List}
78        * will contain all the input elements, in sorted order.
79        *
80        * <p>The argument {@code Comparator} must be {@code Serializable}.
81        *
82        * <p>Example of use:
83        *
84        * <pre>{@code
85        * PCollection<String> pc = ...;
86        * PCollection<List<String>> quantiles =
87        *     pc.apply(ApproximateQuantiles.globally(11, stringCompareFn));
88        * }</pre>
89        *
90        * @param <T> the type of the elements in the input {@code PCollection}
91        * @param numQuantiles the number of elements in the resulting quantile values {@code List}
92        * @param compareFn the function to use to order the elements
93        */
94       public static <T, ComparatorT extends Comparator<T> & Serializable>
95           PTransform<PCollection<T>, PCollection<List<T>>> globally(
96               int numQuantiles, ComparatorT compareFn) {
97         return Combine.globally(ApproximateQuantilesCombineFn.create(numQuantiles, compareFn));
98       }

163       /**
164        * The {@code ApproximateQuantilesCombineFn} combiner gives an idea of the distribution of a
165        * collection of values using approximate {@code N}-tiles. The output of this combiner is a {@code
166        * List} of size {@code numQuantiles}, containing the input values' minimum value, {@code
167        * numQuantiles-2} intermediate values, and maximum value, in sorted order, so for traditional
168        * {@code N}-tiles, one should use {@code ApproximateQuantilesCombineFn#create(N+1)}.
169        *
170        * <p>If there are fewer values to combine than {@code numQuantiles}, then the result {@code List}
171        * will contain all the values being combined, in sorted order.
172        *
173        * <p>Values are ordered using either a specified {@code Comparator} or the values' natural
174        * ordering.
175        *
176        * <p>To evaluate the quantiles we use the "New Algorithm" described here:
177        *
178        * <pre>
179        *   [MRL98] Manku, Rajagopalan &amp; Lindsay, "Approximate Medians and other
180        *   Quantiles in One Pass and with Limited Memory", Proc. 1998 ACM
181        *   SIGMOD, Vol 27, No 2, p 426-435, June 1998.
182        *   http://citeseerx.ist.psu.edu/viewdoc/download?doi=10.1.1.6.6513&amp;rep=rep1&amp;type=pdf
183        * </pre>
184        *
185        * <p>The default error bound is {@code 1 / N}, though in practice the accuracy tends to be much
186        * better.
187        *
188        * <p>See {@link #create(int, Comparator, long, double)} for more information about the meaning of
189        * {@code epsilon}, and {@link #withEpsilon} for a convenient way to adjust it.
190        *
191        * @param <T> the type of the values being combined
192        */
193       public static class ApproximateQuantilesCombineFn<
194               T, ComparatorT extends Comparator<T> & Serializable>
195           extends AccumulatingCombineFn<T, QuantileState<T, ComparatorT>, List<T>> {
196     
197         /**
198          * The cost (in time and space) to compute quantiles to a given accuracy is a function of the
199          * total number of elements in the data set. If an estimate is not known or specified, we use
200          * this as an upper bound. If this is too low, errors may exceed the requested tolerance; if too
201          * high, efficiency may be non-optimal. The impact is logarithmic with respect to this value, so
202          * this default should be fine for most uses.
203          */
204         public static final long DEFAULT_MAX_NUM_ELEMENTS = (long) 1e9;
205     
206         /** The comparison function to use. */
207         private final ComparatorT compareFn;
208     
209         /**
210          * Number of quantiles to produce. The size of the final output list, including the minimum and
211          * maximum, is numQuantiles.
212          */
213         private final int numQuantiles;
214     
215         /** The size of the buffers, corresponding to k in the referenced paper. */
216         private final int bufferSize;
217     
218         /** The number of buffers, corresponding to b in the referenced paper. */
219         private final int numBuffers;
220     
221         private final long maxNumElements;

233           this.compareFn = compareFn;
234           this.bufferSize = bufferSize;
235           this.numBuffers = numBuffers;
236           this.maxNumElements = maxNumElements;

244          * <p>The {@code Comparator} must be {@code Serializable}.

250             ApproximateQuantilesCombineFn<T, ComparatorT> create(
251                 int numQuantiles, ComparatorT compareFn) {

283         /**
284          * Creates an approximate quantiles combiner with the given {@code compareFn} and desired number
285          * of quantiles. A total of {@code numQuantiles} elements will appear in the output list,
286          * including the minimum and maximum.
287          *
288          * <p>The {@code Comparator} must be {@code Serializable}.
289          *
290          * <p>The default error bound is {@code epsilon}, which holds as long as the number of elements
291          * is less than {@code maxNumElements}. Specifically, if one considers the input as a sorted
292          * list x_1, ..., x_N, then the distance between the each exact quantile x_c and its
293          * approximation x_c' is bounded by {@code |c - c'| < epsilon * N}. Note that these errors are
294          * worst-case scenarios; in practice the accuracy tends to be much better.
295          */
296         public static <T, ComparatorT extends Comparator<T> & Serializable>
297             ApproximateQuantilesCombineFn<T, ComparatorT> create(
298                 int numQuantiles, ComparatorT compareFn, long maxNumElements, double epsilon) {
299           // Compute optimal b and k.
300           int b = 2;
301           while ((b - 2) * (1 << (b - 2)) < epsilon * maxNumElements) {
302             b++;
303           }
304           b--;
305           int k = Math.max(2, (int) Math.ceil(maxNumElements / (float) (1 << (b - 1))));
306           return new ApproximateQuantilesCombineFn<>(numQuantiles, compareFn, k, b, maxNumElements);
307         }

337       /** Compact summarization of a collection on which quantiles can be estimated. */
338       static class QuantileState<T, ComparatorT extends Comparator<T> & Serializable>
339           implements Accumulator<T, QuantileState<T, ComparatorT>, List<T>> {
340     
341         private ComparatorT compareFn;
342         private int numQuantiles;
343         private int numBuffers;
344         private int bufferSize;
345     
346         private @Nullable T min;
347     
348         private @Nullable T max;
349     
350         /** The set of buffers, ordered by level from smallest to largest. */
351         private PriorityQueue<QuantileBuffer<T>> buffers;
352     
353         /**
354          * The algorithm requires that the manipulated buffers always be filled to capacity to perform
355          * the collapse operation. This operation can be extended to buffers of varying sizes by
356          * introducing the notion of fractional weights, but it's easier to simply combine the
357          * remainders from all shards into new, full buffers and then take them into account when
358          * computing the final output.
359          */
360         private List<T> unbufferedElements = Lists.newArrayList();
361     
362         private QuantileState(
363             ComparatorT compareFn,
364             int numQuantiles,
365             @Nullable T min,
366             @Nullable T max,
367             int numBuffers,
368             int bufferSize,
369             Collection<T> unbufferedElements,
370             Collection<QuantileBuffer<T>> buffers) {
371           this.compareFn = compareFn;
372           this.numQuantiles = numQuantiles;
373           this.numBuffers = numBuffers;
374           this.bufferSize = bufferSize;
375           this.buffers =
376               new PriorityQueue<>(numBuffers + 1, (q1, q2) -> Integer.compare(q1.level, q2.level));
377           this.min = min;
378           this.max = max;
379           this.unbufferedElements.addAll(unbufferedElements);
380           this.buffers.addAll(buffers);
381         }
382     
383         public static <T, ComparatorT extends Comparator<T> & Serializable>
384             QuantileState<T, ComparatorT> empty(
385                 ComparatorT compareFn, int numQuantiles, int numBuffers, int bufferSize) {
386           return new QuantileState<>(
387               compareFn,
388               numQuantiles,
389               null, /* min */
390               null, /* max */
391               numBuffers,
392               bufferSize,
393               Collections.emptyList(),
394               Collections.emptyList());
395         }
396     
397         public static <T, ComparatorT extends Comparator<T> & Serializable>
398             QuantileState<T, ComparatorT> singleton(
399                 ComparatorT compareFn, int numQuantiles, T elem, int numBuffers, int bufferSize) {
400           return new QuantileState<>(
401               compareFn,
402               numQuantiles,
403               elem, /* min */
404               elem, /* max */
405               numBuffers,
406               bufferSize,
407               Collections.singletonList(elem),
408               Collections.emptyList());
409         }
410     
411         /** Add a new element to the collection being summarized by this state. */
412         @Override
413         public void addInput(T elem) {
414           if (isEmpty()) {
415             min = max = elem;
416           } else if (compareFn.compare(elem, min) < 0) {
417             min = elem;
418           } else if (compareFn.compare(elem, max) > 0) {
419             max = elem;
420           }
421           addUnbuffered(elem);
422         }
423     
424         /** Add a new buffer to the unbuffered list, creating a new buffer and collapsing if needed. */
425         private void addUnbuffered(T elem) {
426           unbufferedElements.add(elem);
427           if (unbufferedElements.size() == bufferSize) {
428             unbufferedElements.sort(compareFn);
429             buffers.add(new QuantileBuffer<>(unbufferedElements));
430             unbufferedElements = Lists.newArrayListWithCapacity(bufferSize);
431             collapseIfNeeded();
432           }
433         }
434     
435         /**
436          * Updates this as if adding all elements seen by other.
437          *
438          * <p>Note that this ignores the {@code Comparator} of the other {@link QuantileState}. In
439          * practice, they should generally be equal, but this method tolerates a mismatch.
440          */
441         @Override
442         public void mergeAccumulator(QuantileState<T, ComparatorT> other) {
443           if (other.isEmpty()) {
444             return;
445           }
446           if (min == null || compareFn.compare(other.min, min) < 0) {
447             min = other.min;
448           }
449           if (max == null || compareFn.compare(other.max, max) > 0) {
450             max = other.max;
451           }
452           for (T elem : other.unbufferedElements) {
453             addUnbuffered(elem);
454           }
455           buffers.addAll(other.buffers);
456           collapseIfNeeded();
457         }
458     
459         public boolean isEmpty() {
460           return unbufferedElements.isEmpty() && buffers.isEmpty();
461         }
462     
463         private void collapseIfNeeded() {
464           while (buffers.size() > numBuffers) {
465             List<QuantileBuffer<T>> toCollapse = Lists.newArrayList();
466             toCollapse.add(buffers.poll());
467             toCollapse.add(buffers.poll());
468             int minLevel = toCollapse.get(1).level;
469             while (!buffers.isEmpty() && buffers.peek().level == minLevel) {
470               toCollapse.add(buffers.poll());
471             }
472             buffers.add(collapse(toCollapse));
473           }
474         }
475     
476         private QuantileBuffer<T> collapse(Iterable<QuantileBuffer<T>> buffers) {
477           int newLevel = 0;
478           long newWeight = 0;
479           for (QuantileBuffer<T> buffer : buffers) {
480             // As presented in the paper, there should always be at least two
481             // buffers of the same (minimal) level to collapse, but it is possible
482             // to violate this condition when combining buffers from independently
483             // computed shards.  If they differ we take the max.
484             newLevel = Math.max(newLevel, buffer.level + 1);
485             newWeight += buffer.weight;
486           }
487           List<T> newElements = interpolate(buffers, bufferSize, newWeight, offset(newWeight));
488           return new QuantileBuffer<>(newLevel, newWeight, newElements);
489         }
490     
491         /**
492          * If the weight is even, we must round up or down. Alternate between these two options to avoid
493          * a bias.
494          */
495         private long offset(long newWeight) {
496           if (newWeight % 2 == 1) {
497             return (newWeight + 1) / 2;
498           } else {
499             offsetJitter = 2 - offsetJitter;
500             return (newWeight + offsetJitter) / 2;
501           }
502         }
503     
504         /** For alternating between biasing up and down in the above even weight collapse operation. */
505         private int offsetJitter = 0;
506     
507         /**
508          * Emulates taking the ordered union of all elements in buffers, repeated according to their
509          * weight, and picking out the (k * step + offset)-th elements of this list for {@code 0 &lt;= k
510          * &lt; count}.
511          */
512         private List<T> interpolate(
513             Iterable<QuantileBuffer<T>> buffers, int count, double step, double offset) {
514           List<Iterator<WeightedValue<T>>> iterators = Lists.newArrayList();
515           for (QuantileBuffer<T> buffer : buffers) {
516             iterators.add(buffer.sizedIterator());
517           }
518           // Each of the buffers is already sorted by element.
519           Iterator<WeightedValue<T>> sorted =
520               Iterators.mergeSorted(iterators, (a, b) -> compareFn.compare(a.getValue(), b.getValue()));
521     
522           List<T> newElements = Lists.newArrayListWithCapacity(count);
523           WeightedValue<T> weightedElement = sorted.next();
524           double current = weightedElement.getWeight();
525           for (int j = 0; j < count; j++) {
526             double target = j * step + offset;
527             while (current <= target && sorted.hasNext()) {
528               weightedElement = sorted.next();
529               current += weightedElement.getWeight();
530             }
531             newElements.add(weightedElement.getValue());
532           }
533           return newElements;
534         }
535     
536         /**
537          * Outputs numQuantiles elements consisting of the minimum, maximum, and numQuantiles - 2 evenly
538          * spaced intermediate elements.
539          *
540          * <p>Returns the empty list if no elements have been added.
541          */
542         @Override
543         public List<T> extractOutput() {
544           if (isEmpty()) {
545             return Lists.newArrayList();
546           }
547           long totalCount = unbufferedElements.size();
548           for (QuantileBuffer<T> buffer : buffers) {
549             totalCount += bufferSize * buffer.weight;
550           }
551           List<QuantileBuffer<T>> all = Lists.newArrayList(buffers);
552           if (!unbufferedElements.isEmpty()) {
553             unbufferedElements.sort(compareFn);
554             all.add(new QuantileBuffer<>(unbufferedElements));
555           }
556           double step = 1.0 * totalCount / (numQuantiles - 1);
557           double offset = (1.0 * totalCount - 1) / (numQuantiles - 1);
558           List<T> quantiles = interpolate(all, numQuantiles - 2, step, offset);
559           quantiles.add(0, min);
560           quantiles.add(max);
561           return quantiles;
562         }
563       }

566       private static class QuantileBuffer<T> {
567         private int level;
568         private long weight;
569         private List<T> elements;
570

572           this(0, 1, elements);
573         }

575         public QuantileBuffer(int level, long weight, List<T> elements) {
576           this.level = level;
577           this.weight = weight;
578           this.elements = elements;
579         }

581         @Override
582         public String toString() {
583           return "QuantileBuffer["
584               + "level="
585               + level
586               + ", weight="
587               + weight
588               + ", elements="
589               + elements
590               + "]";
591         }
592     
593         public Iterator<WeightedValue<T>> sizedIterator() {
594           return new UnmodifiableIterator<WeightedValue<T>>() {
595             Iterator<T> iter = elements.iterator();
596     
597             @Override
598             public boolean hasNext() {
599               return iter.hasNext();
600             }
601     
602             @Override
603             public WeightedValue<T> next() {
604               return WeightedValue.of(iter.next(), weight);
605             }
606           };
607         }

610       /** Coder for QuantileState. */
611       private static class QuantileStateCoder<T, ComparatorT extends Comparator<T> & Serializable>
612           extends CustomCoder<QuantileState<T, ComparatorT>> {
613         private final ComparatorT compareFn;
614         private final Coder<T> elementCoder;
615         private final Coder<List<T>> elementListCoder;
616         private final Coder<Integer> intCoder = BigEndianIntegerCoder.of();
617     
618         public QuantileStateCoder(ComparatorT compareFn, Coder<T> elementCoder) {
619           this.compareFn = compareFn;
620           this.elementCoder = elementCoder;
621           this.elementListCoder = ListCoder.of(elementCoder);
622         }
623     
624         @Override
625         public void encode(QuantileState<T, ComparatorT> state, OutputStream outStream)
626             throws CoderException, IOException {
627           intCoder.encode(state.numQuantiles, outStream);
628           intCoder.encode(state.bufferSize, outStream);
629           elementCoder.encode(state.min, outStream);
630           elementCoder.encode(state.max, outStream);
631           elementListCoder.encode(state.unbufferedElements, outStream);
632           BigEndianIntegerCoder.of().encode(state.buffers.size(), outStream);
633           for (QuantileBuffer<T> buffer : state.buffers) {
634             encodeBuffer(buffer, outStream);
635           }
636         }
637     
638         @Override
639         public QuantileState<T, ComparatorT> decode(InputStream inStream)
640             throws CoderException, IOException {
641           int numQuantiles = intCoder.decode(inStream);
642           int bufferSize = intCoder.decode(inStream);
643           T min = elementCoder.decode(inStream);
644           T max = elementCoder.decode(inStream);
645           List<T> unbufferedElements = elementListCoder.decode(inStream);
646           int numBuffers = BigEndianIntegerCoder.of().decode(inStream);
647           List<QuantileBuffer<T>> buffers = new ArrayList<>(numBuffers);
648           for (int i = 0; i < numBuffers; i++) {
649             buffers.add(decodeBuffer(inStream));
650           }
651           return new QuantileState<>(
652               compareFn, numQuantiles, min, max, numBuffers, bufferSize, unbufferedElements, buffers);
653         }
654     
655         private void encodeBuffer(QuantileBuffer<T> buffer, OutputStream outStream)
656             throws CoderException, IOException {
657           DataOutputStream outData = new DataOutputStream(outStream);
658           outData.writeInt(buffer.level);
659           outData.writeLong(buffer.weight);
660           elementListCoder.encode(buffer.elements, outStream);
661         }
662     
663         private QuantileBuffer<T> decodeBuffer(InputStream inStream)
664             throws IOException, CoderException {
665           DataInputStream inData = new DataInputStream(inStream);
666           return new QuantileBuffer<>(
667               inData.readInt(), inData.readLong(), elementListCoder.decode(inStream));
668         }
669     
670         /**
671          * Notifies ElementByteSizeObserver about the byte size of the encoded value using this coder.
672          */
673         @Override
674         public void registerByteSizeObserver(
675             QuantileState<T, ComparatorT> state, ElementByteSizeObserver observer) throws Exception {
676           elementCoder.registerByteSizeObserver(state.min, observer);
677           elementCoder.registerByteSizeObserver(state.max, observer);
678           elementListCoder.registerByteSizeObserver(state.unbufferedElements, observer);
679     
680           BigEndianIntegerCoder.of().registerByteSizeObserver(state.buffers.size(), observer);
681           for (QuantileBuffer<T> buffer : state.buffers) {
682             observer.update(4L + 8);
683             elementListCoder.registerByteSizeObserver(buffer.elements, observer);
684           }
685         }

687         @Override
688         public boolean equals(@Nullable Object other) {
689           if (other == this) {
690             return true;
691           }
692           if (!(other instanceof QuantileStateCoder)) {
693             return false;
694           }
695           QuantileStateCoder<?, ?> that = (QuantileStateCoder<?, ?>) other;
696           return Objects.equals(this.elementCoder, that.elementCoder)
697               && Objects.equals(this.compareFn, that.compareFn);
698         }
699     
700         @Override
701         public int hashCode() {
702           return Objects.hash(elementCoder, compareFn);
703         }
704     
705         @Override
706         public void verifyDeterministic() throws NonDeterministicException {
707           verifyDeterministic(this, "QuantileState.ElementCoder must be deterministic", elementCoder);
708           verifyDeterministic(
709               this, "QuantileState.ElementListCoder must be deterministic", elementListCoder);
710         }
