94       @Test
95       @Category(NeedsRunner.class)
96       public void testMatchAndMatchAll() throws IOException {
97         Path firstPath = tmpFolder.newFile("first").toPath();
98         Path secondPath = tmpFolder.newFile("second").toPath();
99         int firstSize = 37;
100         int secondSize = 42;
101         long firstModified = 1541097000L;
102         long secondModified = 1541098000L;
103         Files.write(firstPath, new byte[firstSize]);
104         Files.write(secondPath, new byte[secondSize]);
105         Files.setLastModifiedTime(firstPath, FileTime.fromMillis(firstModified));
106         Files.setLastModifiedTime(secondPath, FileTime.fromMillis(secondModified));
107         MatchResult.Metadata firstMetadata = metadata(firstPath, firstSize, firstModified);
108         MatchResult.Metadata secondMetadata = metadata(secondPath, secondSize, secondModified);
109     
110         PAssert.that(
111                 p.apply(
112                     "Match existing",
113                     FileIO.match().filepattern(tmpFolder.getRoot().getAbsolutePath() + "/*")))
114             .containsInAnyOrder(firstMetadata, secondMetadata);
115         PAssert.that(
116                 p.apply(
117                     "Match existing with provider",
118                     FileIO.match()
119                         .filepattern(p.newProvider(tmpFolder.getRoot().getAbsolutePath() + "/*"))))
120             .containsInAnyOrder(firstMetadata, secondMetadata);
121         PAssert.that(
122                 p.apply("Create existing", Create.of(tmpFolder.getRoot().getAbsolutePath() + "/*"))
123                     .apply("MatchAll existing", FileIO.matchAll()))
124             .containsInAnyOrder(firstMetadata, secondMetadata);
125     
126         PAssert.that(
127                 p.apply(
128                     "Match non-existing ALLOW",
129                     FileIO.match()
130                         .filepattern(tmpFolder.getRoot().getAbsolutePath() + "/blah")
131                         .withEmptyMatchTreatment(EmptyMatchTreatment.ALLOW)))
132             .empty();
133         PAssert.that(
134                 p.apply(
135                         "Create non-existing",
136                         Create.of(tmpFolder.getRoot().getAbsolutePath() + "/blah"))
137                     .apply(
138                         "MatchAll non-existing ALLOW",
139                         FileIO.matchAll().withEmptyMatchTreatment(EmptyMatchTreatment.ALLOW)))
140             .empty();
141     
142         PAssert.that(
143                 p.apply(
144                     "Match non-existing ALLOW_IF_WILDCARD",
145                     FileIO.match()
146                         .filepattern(tmpFolder.getRoot().getAbsolutePath() + "/blah*")
147                         .withEmptyMatchTreatment(EmptyMatchTreatment.ALLOW_IF_WILDCARD)))
148             .empty();
149         PAssert.that(
150                 p.apply(
151                         "Create non-existing wildcard + explicit",
152                         Create.of(tmpFolder.getRoot().getAbsolutePath() + "/blah*"))
153                     .apply(
154                         "MatchAll non-existing ALLOW_IF_WILDCARD",
155                         FileIO.matchAll()
156                             .withEmptyMatchTreatment(EmptyMatchTreatment.ALLOW_IF_WILDCARD)))
157             .empty();
158         PAssert.that(
159                 p.apply(
160                         "Create non-existing wildcard + default",
161                         Create.of(tmpFolder.getRoot().getAbsolutePath() + "/blah*"))
162                     .apply("MatchAll non-existing default", FileIO.matchAll()))
163             .empty();
164     
165         p.run();
166       }

219       /** DoFn that copy test files from source to watch path. */
220       private static class CopyFilesFn
221           extends DoFn<KV<String, MatchResult.Metadata>, MatchResult.Metadata> {
222         public CopyFilesFn(Path sourcePath, Path watchPath) {
223           this.sourcePathStr = sourcePath.toString();
224           this.watchPathStr = watchPath.toString();
225         }
226     
227         @StateId("count")
228         @SuppressWarnings("unused")
229         private final StateSpec<ValueState<Integer>> countSpec = StateSpecs.value(VarIntCoder.of());
230     
231         @ProcessElement
232         public void processElement(ProcessContext context, @StateId("count") ValueState<Integer> count)
233             throws IOException, InterruptedException {
234           int current = firstNonNull(count.read(), 0);
235           // unpack value as output
236           context.output(Objects.requireNonNull(context.element()).getValue());
237     
238           CopyOption[] cpOptions = {StandardCopyOption.COPY_ATTRIBUTES};
239           CopyOption[] updOptions = {StandardCopyOption.REPLACE_EXISTING};
240           final Path sourcePath = Paths.get(sourcePathStr);
241           final Path watchPath = Paths.get(watchPathStr);
242     
243           if (0 == current) {
244             Thread.sleep(100);
245             Files.copy(sourcePath.resolve("first"), watchPath.resolve("first"), updOptions);
246             Files.copy(sourcePath.resolve("second"), watchPath.resolve("second"), cpOptions);
247           } else if (1 == current) {
248             Thread.sleep(100);
249             Files.copy(sourcePath.resolve("first"), watchPath.resolve("first"), updOptions);
250             Files.copy(sourcePath.resolve("second"), watchPath.resolve("second"), updOptions);
251             Files.copy(sourcePath.resolve("third"), watchPath.resolve("third"), cpOptions);
252           }
253           count.write(current + 1);
254         }
255     
256         // Member variables need to be serializable.
257         private final String sourcePathStr;
258         private final String watchPathStr;
259       }

261       @Test
262       @Category(NeedsRunner.class)
263       public void testRead() throws IOException {
264         final String path = tmpFolder.newFile("file").getAbsolutePath();
265         final String pathGZ = tmpFolder.newFile("file.gz").getAbsolutePath();
266         Files.write(new File(path).toPath(), "Hello world".getBytes(StandardCharsets.UTF_8));
267         try (Writer writer =
268             new OutputStreamWriter(
269                 new GZIPOutputStream(new FileOutputStream(pathGZ)), StandardCharsets.UTF_8)) {
270           writer.write("Hello world");
271         }
272     
273         PCollection<MatchResult.Metadata> matches = p.apply("Match", FileIO.match().filepattern(path));
274         PCollection<FileIO.ReadableFile> decompressedAuto =
275             matches.apply("Read AUTO", FileIO.readMatches().withCompression(Compression.AUTO));
276         PCollection<FileIO.ReadableFile> decompressedDefault =
277             matches.apply("Read default", FileIO.readMatches());
278         PCollection<FileIO.ReadableFile> decompressedUncompressed =
279             matches.apply(
280                 "Read UNCOMPRESSED", FileIO.readMatches().withCompression(Compression.UNCOMPRESSED));
281         for (PCollection<FileIO.ReadableFile> c :
282             Arrays.asList(decompressedAuto, decompressedDefault, decompressedUncompressed)) {
283           PAssert.thatSingleton(c)
284               .satisfies(
285                   input -> {
286                     assertEquals(path, input.getMetadata().resourceId().toString());
287                     assertEquals("Hello world".length(), input.getMetadata().sizeBytes());
288                     assertEquals(Compression.UNCOMPRESSED, input.getCompression());
289                     assertTrue(input.getMetadata().isReadSeekEfficient());
290                     try {
291                       assertEquals("Hello world", input.readFullyAsUTF8String());
292                     } catch (IOException e) {
293                       throw new RuntimeException(e);
294                     }
295                     return null;
296                   });
297         }
298     
299         PCollection<MatchResult.Metadata> matchesGZ =
300             p.apply("Match GZ", FileIO.match().filepattern(pathGZ));
301         PCollection<FileIO.ReadableFile> compressionAuto =
302             matchesGZ.apply("Read GZ AUTO", FileIO.readMatches().withCompression(Compression.AUTO));
303         PCollection<FileIO.ReadableFile> compressionDefault =
304             matchesGZ.apply("Read GZ default", FileIO.readMatches());
305         PCollection<FileIO.ReadableFile> compressionGzip =
306             matchesGZ.apply("Read GZ GZIP", FileIO.readMatches().withCompression(Compression.GZIP));
307         for (PCollection<FileIO.ReadableFile> c :
308             Arrays.asList(compressionAuto, compressionDefault, compressionGzip)) {
309           PAssert.thatSingleton(c)
310               .satisfies(
311                   input -> {
312                     assertEquals(pathGZ, input.getMetadata().resourceId().toString());
313                     assertFalse(input.getMetadata().sizeBytes() == "Hello world".length());
314                     assertEquals(Compression.GZIP, input.getCompression());
315                     assertFalse(input.getMetadata().isReadSeekEfficient());
316                     try {
317                       assertEquals("Hello world", input.readFullyAsUTF8String());
318                     } catch (IOException e) {
319                       throw new RuntimeException(e);
320                     }
321                     return null;
322                   });
323         }
324     
325         p.run();
326       }

328       private static MatchResult.Metadata metadata(Path path, int size, long lastModifiedMillis) {
329         return MatchResult.Metadata.builder()
330             .setResourceId(FileSystems.matchNewResource(path.toString(), false /* isDirectory */))
331             .setIsReadSeekEfficient(true)

350       @Test
351       public void testFilenameFnResolution() throws Exception {
352         FileIO.Write.FileNaming foo = (window, pane, numShards, shardIndex, compression) -> "foo";
353     
354         String expected =
355             FileSystems.matchNewResource("test", true).resolve("foo", RESOLVE_FILE).toString();
356         assertEquals(
357             "Filenames should be resolved within a relative directory if '.to' is invoked",
358             expected,
359             getDefaultFileName(FileIO.writeDynamic().to("test").withNaming(o -> foo)));
360         assertEquals(
361             "Filenames should be resolved within a relative directory if '.to' is invoked",
362             expected,
363             getDefaultFileName(FileIO.write().to("test").withNaming(foo)));
364     
365         assertEquals(
366             "Filenames should be resolved as the direct result of the filenaming function if '.to' "
367                 + "is not invoked",
368             "foo",
369             getDefaultFileName(FileIO.writeDynamic().withNaming(o -> foo)));
370         assertEquals(
371             "Filenames should be resolved as the direct result of the filenaming function if '.to' "
372                 + "is not invoked",
373             "foo",
374             getDefaultFileName(FileIO.write().withNaming(foo)));
375     
376         assertEquals(
377             "Default to the defaultNaming if a filenaming isn't provided for a non-dynamic write",
378             "output-00000-of-00000",
379             resolveFileNaming(FileIO.write())
380                 .getFilename(
381                     GlobalWindow.INSTANCE,
382                     PaneInfo.ON_TIME_AND_ONLY_FIRING,
383                     0,
384                     0,
385                     Compression.UNCOMPRESSED));
386     
387         assertEquals(
388             "Default Naming should take prefix and suffix into account if provided",
389             "foo-00000-of-00000.bar",
390             resolveFileNaming(FileIO.write().withPrefix("foo").withSuffix(".bar"))
391                 .getFilename(
392                     GlobalWindow.INSTANCE,
393                     PaneInfo.ON_TIME_AND_ONLY_FIRING,
394                     0,
395                     0,
396                     Compression.UNCOMPRESSED));
397     
398         assertEquals(
399             "Filenames should be resolved within a relative directory if '.to' is invoked, "
400                 + "even with default naming",
401             FileSystems.matchNewResource("test", true)
402                 .resolve("output-00000-of-00000", RESOLVE_FILE)
403                 .toString(),
404             resolveFileNaming(FileIO.write().to("test"))
405                 .getFilename(
406                     GlobalWindow.INSTANCE,
407                     PaneInfo.ON_TIME_AND_ONLY_FIRING,
408                     0,
409                     0,
410                     Compression.UNCOMPRESSED));
411       }
