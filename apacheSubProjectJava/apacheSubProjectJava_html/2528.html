110     /**
111      * {@link PTransform Transforms} for reading from and writing to Google Cloud Bigtable.
112      *
113      * <p>Please note the Cloud BigTable HBase connector available <a
114      * href="https://github.com/googleapis/java-bigtable-hbase/tree/master/bigtable-dataflow-parent/bigtable-hbase-beam">here</a>.
115      * We recommend using that connector over this one if <a
116      * href="https://cloud.google.com/bigtable/docs/hbase-bigtable">HBase API</a></> works for your
117      * needs.
118      *
119      * <p>For more information about Cloud Bigtable, see the online documentation at <a
120      * href="https://cloud.google.com/bigtable/">Google Cloud Bigtable</a>.
121      *
122      * <h3>Reading from Cloud Bigtable</h3>
123      *
124      * <p>The Bigtable source returns a set of rows from a single table, returning a {@code
125      * PCollection<Row>}.
126      *
127      * <p>To configure a Cloud Bigtable source, you must supply a table id, a project id, an instance id
128      * and optionally a {@link BigtableOptions} to provide more specific connection configuration. By
129      * default, {@link BigtableIO.Read} will read all rows in the table. The row ranges to be read can
130      * optionally be restricted using {@link BigtableIO.Read#withKeyRanges}, and a {@link RowFilter} can
131      * be specified using {@link BigtableIO.Read#withRowFilter}. For example:
132      *
133      * <pre>{@code
134      * Pipeline p = ...;
135      *
136      * // Scan the entire table.
137      * p.apply("read",
138      *     BigtableIO.read()
139      *         .withProjectId(projectId)
140      *         .withInstanceId(instanceId)
141      *         .withTableId("table"));
142      *
143      * // Scan a prefix of the table.
144      * ByteKeyRange keyRange = ...;
145      * p.apply("read",
146      *     BigtableIO.read()
147      *         .withProjectId(projectId)
148      *         .withInstanceId(instanceId)
149      *         .withTableId("table")
150      *         .withKeyRange(keyRange));
151      *
152      * // Scan a subset of rows that match the specified row filter.
153      * p.apply("filtered read",
154      *     BigtableIO.read()
155      *         .withProjectId(projectId)
156      *         .withInstanceId(instanceId)
157      *         .withTableId("table")
158      *         .withRowFilter(filter));
159      *
160      * // Configure timeouts for reads.
161      * // Let each attempt run for 1 second, retry if the attempt failed.
162      * // Give up after the request is retried for 60 seconds.
163      * Duration attemptTimeout = Duration.millis(1000);
164      * Duration operationTimeout = Duration.millis(60 * 1000);
165      * p.apply("read",
166      *     BigtableIO.read()
167      *         .withProjectId(projectId)
168      *         .withInstanceId(instanceId)
169      *         .withTableId("table")
170      *         .withKeyRange(keyRange)
171      *         .withAttemptTimeout(attemptTimeout)
172      *         .withOperationTimeout(operationTimeout);
173      * }</pre>
174      *
175      * <h3>Writing to Cloud Bigtable</h3>
176      *
177      * <p>The Bigtable sink executes a set of row mutations on a single table. It takes as input a
178      * {@link PCollection PCollection&lt;KV&lt;ByteString, Iterable&lt;Mutation&gt;&gt;&gt;}, where the
179      * {@link ByteString} is the key of the row being mutated, and each {@link Mutation} represents an
180      * idempotent transformation to that row.
181      *
182      * <p>To configure a Cloud Bigtable sink, you must supply a table id, a project id, an instance id
183      * and optionally a configuration function for {@link BigtableOptions} to provide more specific
184      * connection configuration, for example:
185      *
186      * <pre>{@code
187      * PCollection<KV<ByteString, Iterable<Mutation>>> data = ...;
188      *
189      * data.apply("write",
190      *     BigtableIO.write()
191      *         .withProjectId("project")
192      *         .withInstanceId("instance")
193      *         .withTableId("table"));
194      * }
195      *
196      * // Configure batch size for writes
197      * data.apply("write",
198      *     BigtableIO.write()
199      *         .withProjectId("project")
200      *         .withInstanceId("instance")
201      *         .withTableId("table")
202      *         .withBatchElements(100)); // every batch will have 100 elements
203      * }</pre>
204      *
205      * <p>Configure timeout for writes:
206      *
207      * <pre>{@code
208      * // Let each attempt run for 1 second, retry if the attempt failed.
209      * // Give up after the request is retried for 60 seconds.
210      * Duration attemptTimeout = Duration.millis(1000);
211      * Duration operationTimeout = Duration.millis(60 * 1000);
212      * data.apply("write",
213      *     BigtableIO.write()
214      *         .withProjectId("project")
215      *         .withInstanceId("instance")
216      *         .withTableId("table")
217      *         .withAttemptTimeout(attemptTimeout)
218      *         .withOperationTimeout(operationTimeout));
219      * }</pre>
220      *
221      * <p>You can also limit the wait time in the finish bundle step by setting the
222      * bigtable_writer_wait_timeout_ms experimental flag when you run the pipeline. For example,
223      * --experiments=bigtable_writer_wait_timeout_ms=60000 will limit the wait time in finish bundle to
224      * be 10 minutes.
225      *
226      * <p>Optionally, BigtableIO.write() may be configured to emit {@link BigtableWriteResult} elements
227      * after each group of inputs is written to Bigtable. These can be used to then trigger user code
228      * after writes have completed. See {@link org.apache.beam.sdk.transforms.Wait} for details on the
229      * windowing requirements of the signal and input PCollections.
230      *
231      * <pre>{@code
232      * // See Wait.on
233      * PCollection<KV<ByteString, Iterable<Mutation>>> data = ...;
234      *
235      * PCollection<BigtableWriteResult> writeResults =
236      *     data.apply("write",
237      *         BigtableIO.write()
238      *             .withProjectId("project")
239      *             .withInstanceId("instance")
240      *             .withTableId("table"))
241      *             .withWriteResults();
242      *
243      * // The windowing of `moreData` must be compatible with `data`, see {@link org.apache.beam.sdk.transforms.Wait#on}
244      * // for details.
245      * PCollection<...> moreData = ...;
246      *
247      * moreData
248      *     .apply("wait for writes", Wait.on(writeResults))
249      *     .apply("do something", ParDo.of(...))
250      *
251      * }</pre>
252      *
253      * <h3>Streaming Changes from Cloud Bigtable</h3>
254      *
255      * <p>Cloud Bigtable change streams enable users to capture and stream out mutations from their
256      * Cloud Bigtable tables in real-time. Cloud Bigtable change streams enable many use cases including
257      * integrating with a user's data analytics pipelines, support audit and archival requirements as
258      * well as triggering downstream application logic on specific database changes.
259      *
260      * <p>Change stream connector creates and manages a metadata table to manage the state of the
261      * connector. By default, the table is created in the same instance as the table being streamed.
262      * However, it can be overridden with {@link
263      * BigtableIO.ReadChangeStream#withMetadataTableProjectId}, {@link
264      * BigtableIO.ReadChangeStream#withMetadataTableInstanceId}, {@link
265      * BigtableIO.ReadChangeStream#withMetadataTableTableId}, and {@link
266      * BigtableIO.ReadChangeStream#withMetadataTableAppProfileId}. The app profile for the metadata
267      * table must be a single cluster app profile with single row transaction enabled.
268      *
269      * <p>Note - To prevent unforeseen stream stalls, the BigtableIO connector outputs all data with an
270      * output timestamp of zero, making all data late, which will ensure that the stream will not stall.
271      * However, it means that you may have to deal with all data as late data, and features that depend
272      * on watermarks will not function. This means that Windowing functions and States and Timers are no
273      * longer effectively usable. Example use cases that are not possible because of this include:
274      *
275      * <ul>
276      *   <li>Completeness in a replicated cluster with writes to a row on multiple clusters.
277      *   <li>Ordering at the row level in a replicated cluster where the row is being written through
278      *       multiple-clusters.
279      * </ul>
280      *
281      * Users can use GlobalWindows with (non-event time) Triggers to group this late data into Panes.
282      * You can see an example of this in the pipeline below.
283      *
284      * <pre>{@code
285      * Pipeline pipeline = ...;
286      * pipeline
287      *    .apply(
288      *        BigtableIO.readChangeStream()
289      *            .withProjectId(projectId)
290      *            .withInstanceId(instanceId)
291      *            .withTableId(tableId)
292      *            .withAppProfileId(appProfileId)
293      *            .withStartTime(startTime));
294      * }</pre>
295      *

1370         private BiFunction<MutateRowResponse, Throwable, Void> handleMutationException(

1372           return (MutateRowResponse result, Throwable exception) -> {

1384         private void retryIndividualRecord(
1385             KV<ByteString, Iterable<Mutation>> record, BoundedWindow window) {
1386           try {
1387             bigtableWriter.writeSingleRecord(record);
1388           } catch (Throwable e) {
1389             if (isDataException(e)) {
1390               // if we get another NotFoundException, we know this is the bad record.
1391               badRecords.add(KV.of(new BigtableWriteException(record, e), window));
1392             } else {
1393               failures.add(new BigtableWriteException(record, e));
1394             }
1395           }
1396         }

1398         // This method checks if an exception is the result of an error in the data.
1399         // We first check if the exception is retryable, because if it is, we want to retry it via the
1400         // runner. If the method is retryable, we check if it is a NotFoundException, or if it's an
1401         // InvalidArgumentException. A NotFoundException likely means that the mutation is trying to
1402         // write to a column family that doesn't exist. An InvalidArgumentException means that the
1403         // mutation itself is invalid, with either an empty row key, invalid timestamp (ts <= -2),
1404         // an empty mutation, or a column qualifier that is too large.
1405         private static boolean isDataException(Throwable e) {
1406           if (e instanceof ApiException && !((ApiException) e).isRetryable()) {
1407             return e instanceof NotFoundException || e instanceof InvalidArgumentException;
1408           }
1409           return false;
1410         }

1412         @FinishBundle
1413         public void finishBundle(FinishBundleContext c) throws Exception {
1414           if (bigtableWriter != null) {
1415             try {
1416               bigtableWriter.close();
1417             } catch (IOException e) {
1418               // If the writer fails due to a batching exception, but no failures were detected
1419               // it means that error handling was enabled, and that errors were detected and routed
1420               // to the error queue. Bigtable will successfully write other failures in the batch,
1421               // so this exception should be ignored
1422               if (!(e.getCause() instanceof BatchingException)) {
1423                 throw e;
1424               }
1425             }
1426     
1427             // Sanity check: ensure that all element futures are resolved. This should be already be the
1428             // case once bigtableWriter.close() finishes.
1429             try {
1430               CompletableFuture.allOf(outstandingWrites.toArray(new CompletableFuture<?>[0]))
1431                   .get(1, TimeUnit.MINUTES);
1432             } catch (TimeoutException e) {
1433               throw new IllegalStateException(
1434                   "Unexpected timeout waiting for element future to resolve after the writer was closed",
1435                   e);
1436             }
1437     
1438             if (!reportedLineage) {
1439               bigtableWriter.reportLineage();
1440               reportedLineage = true;
1441             }
1442             bigtableWriter = null;
1443           }
1444     
1445           for (KV<BigtableWriteException, BoundedWindow> badRecord : badRecords) {
1446             try {
1447               badRecordRouter.route(
1448                   c,
1449                   badRecord.getKey().getRecord(),
1450                   inputCoder,
1451                   (Exception) badRecord.getKey().getCause(),
1452                   "Failed to write malformed mutation to Bigtable",
1453                   badRecord.getValue());
1454             } catch (Exception e) {
1455               failures.add(badRecord.getKey());
1456             }
1457           }
1458     
1459           checkForFailures();
1460     
1461           LOG.debug("Wrote {} records", recordsWritten);
1462     
1463           for (Map.Entry<BoundedWindow, Long> entry : seenWindows.entrySet()) {
1464             c.output(
1465                 BigtableWriteResult.create(entry.getValue()),
1466                 entry.getKey().maxTimestamp(),
1467                 entry.getKey());
1468           }
1469         }

1501         /** If any write has asynchronously failed, fail the bundle with a useful error. */
1502         private void checkForFailures() throws IOException {
1503           // Note that this function is never called by multiple threads and is the only place that
1504           // we remove from failures, so this code is safe.
1505           if (failures.isEmpty()) {
1506             return;
1507           }
1508     
1509           StringBuilder logEntry = new StringBuilder();
1510           int i = 0;
1511           List<BigtableWriteException> suppressed = Lists.newArrayList();
1512           for (; i < 10 && !failures.isEmpty(); ++i) {
1513             BigtableWriteException exc = failures.remove();
1514             logEntry.append("\n").append(exc.getMessage());
1515             if (exc.getCause() != null) {
1516               logEntry.append(": ").append(exc.getCause().getMessage());
1517             }
1518             suppressed.add(exc);
1519           }
1520           String message =
1521               String.format(
1522                   "At least %d errors occurred writing to Bigtable. First %d errors: %s",
1523                   i + failures.size(), i, logEntry.toString());
1524           LOG.error(message);
1525           IOException exception = new IOException(message);
1526           for (BigtableWriteException e : suppressed) {
1527             exception.addSuppressed(e);
1528           }
1529           throw exception;
1530         }

1600         @Override
1601         public List<BigtableSource> split(long desiredBundleSizeBytes, PipelineOptions options)
1602             throws Exception {
1603           // Update the desiredBundleSizeBytes in order to limit the
1604           // number of splits to maximumNumberOfSplits.
1605           long maximumNumberOfSplits = 4000;
1606           long sizeEstimate = getEstimatedSizeBytes(options);
1607           desiredBundleSizeBytes =
1608               Math.max(
1609                   sizeEstimate / maximumNumberOfSplits,
1610                   // BoundedReadEvaluatorFactory may provide us with a desiredBundleSizeBytes of 0
1611                   // https://github.com/apache/beam/issues/28793
1612                   Math.max(1, desiredBundleSizeBytes));
1613     
1614           // Delegate to testable helper.
1615           List<BigtableSource> splits =
1616               splitBasedOnSamples(desiredBundleSizeBytes, getSampleRowKeys(options));
1617     
1618           // Reduce the splits.
1619           List<BigtableSource> reduced = reduceSplits(splits, options, MAX_SPLIT_COUNT);
1620           // Randomize the result before returning an immutable copy of the splits, the default behavior
1621           // may lead to multiple workers hitting the same tablet.
1622           Collections.shuffle(reduced);
1623           return ImmutableList.copyOf(reduced);
1624         }

1626         /** Returns a mutable list of reduced splits. */
1627         @VisibleForTesting
1628         protected List<BigtableSource> reduceSplits(
1629             List<BigtableSource> splits, PipelineOptions options, long maxSplitCounts)
1630             throws IOException {
1631           int numberToCombine = (int) ((splits.size() + maxSplitCounts - 1) / maxSplitCounts);
1632           if (splits.size() < maxSplitCounts || numberToCombine < 2) {
1633             return new ArrayList<>(splits);
1634           }
1635           List<BigtableSource> reducedSplits = new ArrayList<>();
1636           List<ByteKeyRange> previousSourceRanges = new ArrayList<>();
1637           int counter = 0;
1638           long size = 0;
1639           for (BigtableSource source : splits) {
1640             if (counter == numberToCombine
1641                 || !checkRangeAdjacency(previousSourceRanges, source.getRanges())) {
1642               reducedSplits.add(
1643                   new BigtableSource(
1644                       factory,
1645                       configId,
1646                       config,
1647                       readOptions.withKeyRanges(previousSourceRanges),
1648                       size));
1649               counter = 0;
1650               size = 0;
1651               previousSourceRanges = new ArrayList<>();
1652             }
1653             previousSourceRanges.addAll(source.getRanges());
1654             previousSourceRanges = mergeRanges(previousSourceRanges);
1655             size += source.getEstimatedSizeBytes(options);
1656             counter++;
1657           }
1658           if (size > 0) {
1659             reducedSplits.add(
1660                 new BigtableSource(
1661                     factory, configId, config, readOptions.withKeyRanges(previousSourceRanges), size));
1662           }
1663           return reducedSplits;
1664         }

1673           ImmutableList.Builder<ByteKeyRange> mergedRanges = ImmutableList.builder();
1674           if (ranges != null) {

1683         /**
1684          * Helper to validate range Adjacency. Ranges are considered adjacent if
1685          * [1..100][100..200][200..300]
1686          */
1687         private static boolean checkRangeAdjacency(List<ByteKeyRange> ranges) {
1688           int index = 0;
1689           if (ranges.size() < 2) {
1690             return true;
1691           }
1692           ByteKey lastEndKey = ranges.get(index++).getEndKey();
1693           while (index < ranges.size()) {
1694             ByteKeyRange currentKeyRange = ranges.get(index++);
1695             if (!lastEndKey.equals(currentKeyRange.getStartKey())) {
1696               return false;
1697             }
1698             lastEndKey = currentKeyRange.getEndKey();
1699           }
1700           return true;
1701         }

1704          * Helper to combine/merge ByteKeyRange Ranges should only be merged if they are adjacent ex.
1705          * [1..100][100..200][200..300] will result in [1..300] Note: this method will not check for
1706          * adjacency see {@link #checkRangeAdjacency(List)}
1707          */
1708         private static List<ByteKeyRange> mergeRanges(List<ByteKeyRange> ranges) {
1709           List<ByteKeyRange> response = new ArrayList<>();
1710           if (ranges.size() < 2) {
1711             response.add(ranges.get(0));
1712           } else {
1713             response.add(
1714                 ByteKeyRange.of(
1715                     ranges.get(0).getStartKey(), ranges.get(ranges.size() - 1).getEndKey()));
1716           }
1717           return response;
1718         }

1720         /** Helper that splits this source into bundles based on Cloud Bigtable sampled row keys. */
1721         private List<BigtableSource> splitBasedOnSamples(
1722             long desiredBundleSizeBytes, List<KeyOffset> sampleRowKeys) {
1723           // There are no regions, or no samples available. Just scan the entire range.
1724           if (sampleRowKeys.isEmpty()) {
1725             LOG.info("Not splitting source {} because no sample row keys are available.", this);
1726             return Collections.singletonList(this);
1727           }
1728           LOG.info(
1729               "About to split into bundles of size {} with sampleRowKeys length {} first element {}",
1730               desiredBundleSizeBytes,
1731               sampleRowKeys.size(),
1732               sampleRowKeys.get(0));
1733     
1734           ImmutableList.Builder<BigtableSource> splits = ImmutableList.builder();
1735           for (ByteKeyRange range : getRanges()) {
1736             splits.addAll(splitRangeBasedOnSamples(desiredBundleSizeBytes, sampleRowKeys, range));
1737           }
1738           return splits.build();
1739         }

1741         /**
1742          * Helper that splits a {@code ByteKeyRange} into bundles based on Cloud Bigtable sampled row
1743          * keys.
1744          */
1745         private List<BigtableSource> splitRangeBasedOnSamples(
1746             long desiredBundleSizeBytes, List<KeyOffset> sampleRowKeys, ByteKeyRange range) {
1747     
1748           // Loop through all sampled responses and generate splits from the ones that overlap the
1749           // scan range. The main complication is that we must track the end range of the previous
1750           // sample to generate good ranges.
1751           ByteKey lastEndKey = ByteKey.EMPTY;
1752           long lastOffset = 0;
1753           ImmutableList.Builder<BigtableSource> splits = ImmutableList.builder();
1754           for (KeyOffset keyOffset : sampleRowKeys) {
1755             ByteKey responseEndKey = makeByteKey(keyOffset.getKey());
1756             long responseOffset = keyOffset.getOffsetBytes();
1757             checkState(
1758                 responseOffset >= lastOffset,
1759                 "Expected response byte offset %s to come after the last offset %s",
1760                 responseOffset,
1761                 lastOffset);
1762     
1763             if (!range.overlaps(ByteKeyRange.of(lastEndKey, responseEndKey))) {
1764               // This region does not overlap the scan, so skip it.
1765               lastOffset = responseOffset;
1766               lastEndKey = responseEndKey;
1767               continue;
1768             }
1769     
1770             // Calculate the beginning of the split as the larger of startKey and the end of the last
1771             // split. Unspecified start is smallest key so is correctly treated as earliest key.
1772             ByteKey splitStartKey = lastEndKey;
1773             if (splitStartKey.compareTo(range.getStartKey()) < 0) {
1774               splitStartKey = range.getStartKey();
1775             }
1776     
1777             // Calculate the end of the split as the smaller of endKey and the end of this sample. Note
1778             // that range.containsKey handles the case when range.getEndKey() is empty.
1779             ByteKey splitEndKey = responseEndKey;
1780             if (!range.containsKey(splitEndKey)) {
1781               splitEndKey = range.getEndKey();
1782             }
1783     
1784             // We know this region overlaps the desired key range, and we know a rough estimate of its
1785             // size. Split the key range into bundle-sized chunks and then add them all as splits.
1786             long sampleSizeBytes = responseOffset - lastOffset;
1787             List<BigtableSource> subSplits =
1788                 splitKeyRangeIntoBundleSizedSubranges(
1789                     sampleSizeBytes,
1790                     desiredBundleSizeBytes,
1791                     ByteKeyRange.of(splitStartKey, splitEndKey));
1792             splits.addAll(subSplits);
1793     
1794             // Move to the next region.
1795             lastEndKey = responseEndKey;
1796             lastOffset = responseOffset;
1797           }
1798     
1799           // We must add one more region after the end of the samples if both these conditions hold:
1800           //  1. we did not scan to the end yet (lastEndKey is concrete, not 0-length).
1801           //  2. we want to scan to the end (endKey is empty) or farther (lastEndKey < endKey).
1802           if (!lastEndKey.isEmpty()
1803               && (range.getEndKey().isEmpty() || lastEndKey.compareTo(range.getEndKey()) < 0)) {
1804             splits.add(this.withSingleRange(ByteKeyRange.of(lastEndKey, range.getEndKey())));

1821         /**
1822          * Computes the estimated size in bytes based on the total size of all samples that overlap the
1823          * key ranges this source will scan.
1824          */
1825         private long getEstimatedSizeBytesBasedOnSamples(List<KeyOffset> samples) {
1826           long estimatedSizeBytes = 0;
1827           long lastOffset = 0;
1828           ByteKey currentStartKey = ByteKey.EMPTY;
1829           // Compute the total estimated size as the size of each sample that overlaps the scan range.
1830           // TODO: In future, Bigtable service may provide finer grained APIs, e.g., to sample given a
1831           // filter or to sample on a given key range.
1832           for (KeyOffset keyOffset : samples) {
1833             ByteKey currentEndKey = makeByteKey(keyOffset.getKey());
1834             long currentOffset = keyOffset.getOffsetBytes();
1835             if (!currentStartKey.isEmpty() && currentStartKey.equals(currentEndKey)) {
1836               // Skip an empty region.
1837               lastOffset = currentOffset;
1838               continue;
1839             } else {
1840               for (ByteKeyRange range : getRanges()) {
1841                 if (range.overlaps(ByteKeyRange.of(currentStartKey, currentEndKey))) {
1842                   estimatedSizeBytes += currentOffset - lastOffset;
1843                   // We don't want to double our estimated size if two ranges overlap this sample
1844                   // region, so exit early.
1845                   break;
1846                 }
1847               }
1848             }
1849             currentStartKey = currentEndKey;
1850             lastOffset = currentOffset;
1851           }
1852           return estimatedSizeBytes;
1853         }

1891         /** Helper that splits the specified range in this source into bundles. */
1892         private List<BigtableSource> splitKeyRangeIntoBundleSizedSubranges(
1893             long sampleSizeBytes, long desiredBundleSizeBytes, ByteKeyRange range) {
1894           // Catch the trivial cases. Split is small enough already, or this is the last region.
1895           LOG.debug(
1896               "Subsplit for sampleSizeBytes {} and desiredBundleSizeBytes {}",
1897               sampleSizeBytes,
1898               desiredBundleSizeBytes);
1899           if (sampleSizeBytes <= desiredBundleSizeBytes) {
1900             return Collections.singletonList(
1901                 this.withSingleRange(ByteKeyRange.of(range.getStartKey(), range.getEndKey())));
1902           }
1903     
1904           checkArgument(
1905               sampleSizeBytes > 0, "Sample size %s bytes must be greater than 0.", sampleSizeBytes);
1906           checkArgument(
1907               desiredBundleSizeBytes > 0,
1908               "Desired bundle size %s bytes must be greater than 0.",
1909               desiredBundleSizeBytes);
1910     
1911           int splitCount = (int) Math.ceil(((double) sampleSizeBytes) / desiredBundleSizeBytes);
1912           List<ByteKey> splitKeys = range.split(splitCount);
1913           ImmutableList.Builder<BigtableSource> splits = ImmutableList.builder();
1914           Iterator<ByteKey> keys = splitKeys.iterator();
1915           ByteKey prev = keys.next();
1916           while (keys.hasNext()) {
1917             ByteKey next = keys.next();
1918             splits.add(
1919                 this.withSingleRange(ByteKeyRange.of(prev, next))
1920                     .withEstimatedSizeBytes(sampleSizeBytes / splitCount));
1921             prev = next;
1922           }
1923           return splits.build();
1924         }

1977               (reader.start()

1979                           true, makeByteKey(reader.getCurrentRow().getKey())))
1980                   || rangeTracker.markDone();

1982             ++recordsReturned;
1983             source.reportLineageOnce(reader);

1991         }

1996               (reader.advance()
1997                       && rangeTracker.tryReturnRecordAt(
1998                           true, makeByteKey(reader.getCurrentRow().getKey())))
1999                   || rangeTracker.markDone();
2000           if (hasRecord) {

2002           }

2034         @Override
2035         public final @Nullable synchronized BigtableSource splitAtFraction(double fraction) {
2036           ByteKey splitKey;
2037           ByteKeyRange range = rangeTracker.getRange();
2038           try {
2039             splitKey = range.interpolateKey(fraction);
2040           } catch (RuntimeException e) {
2041             LOG.info("{}: Failed to interpolate key for fraction {}.", range, fraction, e);
2042             return null;
2043           }
2044           LOG.info("Proposing to split {} at fraction {} (key {})", rangeTracker, fraction, splitKey);
2045           BigtableSource primary;
2046           BigtableSource residual;
2047           try {
2048             primary = source.withSingleRange(ByteKeyRange.of(range.getStartKey(), splitKey));
2049             residual = source.withSingleRange(ByteKeyRange.of(splitKey, range.getEndKey()));
2050           } catch (RuntimeException e) {
2051             LOG.info(
2052                 "{}: Interpolating for fraction {} yielded invalid split key {}.",
2053                 rangeTracker.getRange(),
2054                 fraction,
2055                 splitKey,
2056                 e);
2057             return null;
2058           }
2059           if (!rangeTracker.trySplitAtPosition(splitKey)) {
2060             return null;
2061           }
2062           this.source = primary;
2063           return residual;
2064         }

2455           if (backlogReplicationAdjustment == null) {
2456             backlogReplicationAdjustment = DEFAULT_BACKLOG_REPLICATION_ADJUSTMENT;
2457           }
2458     
2459           ActionFactory actionFactory = new ActionFactory();
2460           ChangeStreamMetrics metrics = new ChangeStreamMetrics();

2462               new DaoFactory(
2463                   bigtableConfig, metadataTableConfig, getTableId(), metadataTableId, changeStreamName);

2465           // Validate the configuration is correct before creating the pipeline, if required.
2466           try {
2467             MetadataTableAdminDao metadataTableAdminDao = daoFactory.getMetadataTableAdminDao();
2468             boolean validateConfig = true;
2469             if (getValidateConfig() != null) {
2470               validateConfig = getValidateConfig();

2472             // Validate app profile and create metadata table if validate is required.
2473             if (validateConfig) {
2474               createOrUpdateMetadataTable(metadataTableAdminDao, metadataTableId);
2475               validateAppProfile(metadataTableAdminDao, metadataTableConfig.getAppProfileId().get());
2476             }
2477             // Validate metadata table if validate is required. We validate metadata table after
2478             // createOrUpdateMetadataTable because if the metadata doesn't exist, we have to run
2479             // createOrUpdateMetadataTable to create the metadata table.
2480             if (metadataTableConfig.getValidate()) {
2481               checkArgument(
2482                   metadataTableAdminDao.doesMetadataTableExist(),
2483                   "Metadata table does not exist: " + metadataTableAdminDao.getTableId());
2484             }
2485           } catch (Exception e) {
2486             throw new RuntimeException(e);
2487           } finally {
2488             daoFactory.close();
2489           }
2490     
2491           InitializeDoFn initializeDoFn =
2492               new InitializeDoFn(daoFactory, startTime, existingPipelineOptions);

2494               new DetectNewPartitionsDoFn(getEndTime(), actionFactory, daoFactory, metrics);
2495           ReadChangeStreamPartitionDoFn readChangeStreamPartitionDoFn =
2496               new ReadChangeStreamPartitionDoFn(
2497                   daoFactory, actionFactory, metrics, backlogReplicationAdjustment);

2500               input
2501                   .apply(Impulse.create())
2502                   .apply("Initialize", ParDo.of(initializeDoFn))
2503                   .apply("DetectNewPartition", ParDo.of(detectNewPartitionsDoFn))
2504                   .apply("ReadChangeStreamPartition", ParDo.of(readChangeStreamPartitionDoFn));
2505     
2506           Coder<KV<ByteString, ChangeStreamRecord>> outputCoder = readChangeStreamOutput.getCoder();
2507           CoderSizeEstimator<KV<ByteString, ChangeStreamRecord>> sizeEstimator =
2508               new CoderSizeEstimator<>(outputCoder);
2509           readChangeStreamPartitionDoFn.setSizeEstimator(sizeEstimator);
2510     
2511           return readChangeStreamOutput.apply(
2512               "FilterForMutation", ParDo.of(new FilterForMutationDoFn()));
2513         }

2548        * table create permissions. This method is useful if the pipeline isn't granted permissions to
2549        * create Bigtable tables. Run this method with correct permissions to create the metadata table,
2550        * which is required to read Bigtable change streams. This method only needs to be run once, and
2551        * the metadata table can be reused for all pipelines.
2552        *
2553        * @param projectId project id of the metadata table, usually the same as the project of the table
2554        *     being streamed
2555        * @param instanceId instance id of the metadata table, usually the same as the instance of the
2556        *     table being streamed
2557        * @param tableId name of the metadata table, leave it null or empty to use default.
2558        * @return true if the table was successfully created. Otherwise, false.
2559        */
2560       public static boolean createOrUpdateReadChangeStreamMetadataTable(
2561           String projectId, String instanceId, @Nullable String tableId) throws IOException {
2562         BigtableConfig bigtableConfig =
2563             BigtableConfig.builder()
2564                 .setValidate(true)
2565                 .setProjectId(StaticValueProvider.of(projectId))
2566                 .setInstanceId(StaticValueProvider.of(instanceId))
2567                 .setAppProfileId(
2568                     StaticValueProvider.of(
2569                         "default")) // App profile is not used. It's only required for data API.
2570                 .build();
2571     
2572         if (tableId == null || tableId.isEmpty()) {
2573           tableId = MetadataTableAdminDao.DEFAULT_METADATA_TABLE_NAME;
2574         }
2575     
2576         DaoFactory daoFactory = new DaoFactory(null, bigtableConfig, null, tableId, null);
2577     
2578         try {
2579           MetadataTableAdminDao metadataTableAdminDao = daoFactory.getMetadataTableAdminDao();
2580     
2581           // Only try to create or update metadata table if option is set to true. Otherwise, just
2582           // check if the table exists.
2583           if (metadataTableAdminDao.createMetadataTable()) {
2584             LOG.info("Created metadata table: " + metadataTableAdminDao.getTableId());
