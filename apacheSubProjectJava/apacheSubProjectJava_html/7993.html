482        * Tests reading all rows using key ranges. Tests a prefix [), a suffix (], and a restricted range
483        * [] and that some properties hold across them.
484        */
485       @Test
486       public void testReadingWithKeyRange() throws Exception {
487         final String table = "TEST-KEY-RANGE-TABLE";
488         final int numRows = 1001;
489         List<Row> testRows = makeTableData(table, numRows);
490         ByteKey startKey = ByteKey.copyFrom("key000000100".getBytes(StandardCharsets.UTF_8));
491         ByteKey endKey = ByteKey.copyFrom("key000000300".getBytes(StandardCharsets.UTF_8));
492     
493         service.setupSampleRowKeys(table, numRows / 10, "key000000100".length());
494         // Test prefix: [beginning, startKey).
495         final ByteKeyRange prefixRange = ByteKeyRange.ALL_KEYS.withEndKey(startKey);
496         List<Row> prefixRows = filterToRange(testRows, prefixRange);
497         runReadTest(defaultRead.withTableId(table).withKeyRange(prefixRange), prefixRows);
498     
499         // Test suffix: [startKey, end).
500         final ByteKeyRange suffixRange = ByteKeyRange.ALL_KEYS.withStartKey(startKey);
501         List<Row> suffixRows = filterToRange(testRows, suffixRange);
502         runReadTest(defaultRead.withTableId(table).withKeyRange(suffixRange), suffixRows);
503     
504         // Test restricted range: [startKey, endKey).
505         final ByteKeyRange middleRange = ByteKeyRange.of(startKey, endKey);
506         List<Row> middleRows = filterToRange(testRows, middleRange);
507         runReadTest(defaultRead.withTableId(table).withKeyRange(middleRange), middleRows);
508     
509         //////// Size and content sanity checks //////////
510     
511         // Prefix, suffix, middle should be non-trivial (non-zero,non-all).
512         assertThat(prefixRows, allOf(hasSize(lessThan(numRows)), hasSize(greaterThan(0))));
513         assertThat(suffixRows, allOf(hasSize(lessThan(numRows)), hasSize(greaterThan(0))));
514         assertThat(middleRows, allOf(hasSize(lessThan(numRows)), hasSize(greaterThan(0))));
515     
516         // Prefix + suffix should be exactly all rows.
517         List<Row> union = Lists.newArrayList(prefixRows);
518         union.addAll(suffixRows);
519         assertThat(
520             "prefix + suffix = total", union, containsInAnyOrder(testRows.toArray(new Row[] {})));
521     
522         // Suffix should contain the middle.
523         assertThat(suffixRows, hasItems(middleRows.toArray(new Row[] {})));
524       }

1577         private final BigtableSource source;
1578         private final FakeBigtableService service;
1579         private Iterator<Map.Entry<ByteString, ByteString>> rows;

1583         public FakeBigtableReader(BigtableSource source, FakeBigtableService service) {
1584           this.source = source;
1585           this.service = service;
1586           if (source.getRowFilter() == null) {
1587             filter = Predicates.alwaysTrue();
1588           } else {
1589             ByteString keyRegex = source.getRowFilter().getRowKeyRegexFilter();
1590             checkArgument(!keyRegex.isEmpty(), "Only RowKeyRegexFilter is supported");

1606         @Override
1607         public boolean advance() throws IOException {
1608           // Loop until we find a row in range, or reach the end of the iterator.
1609           Map.Entry<ByteString, ByteString> entry = null;
1610           while (rows.hasNext()) {
1611             entry = rows.next();
1612             if (!filter.apply(entry.getKey())
1613                 || !rangesContainsKey(source.getRanges(), makeByteKey(entry.getKey()))) {
1614               // Does not match row filter or does not match source range. Skip.
1615               entry = null;
1616               continue;
1617             }
1618             // Found a row inside this source's key range, stop.
1619             break;
1620           }
1621     
1622           // Return false if no more rows.
1623           if (entry == null) {
1624             currentRow = null;
1625             return false;
1626           }
1627     
1628           // Set the current row and return true.
1629           currentRow = makeRow(entry.getKey(), entry.getValue());
1630           return true;
1631         }

1634           for (ByteKeyRange range : ranges) {

1637             }
1638           }
1639           return false;
