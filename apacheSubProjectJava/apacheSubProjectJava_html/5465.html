74     /**
75      * A {@link Pipeline} manages a directed acyclic graph of {@link PTransform PTransforms}, and the
76      * {@link PCollection PCollections} that the {@link PTransform PTransforms} consume and produce.
77      *
78      * <p>Each {@link Pipeline} is self-contained and isolated from any other {@link Pipeline}. The
79      * {@link PValue PValues} that are inputs and outputs of each of a {@link Pipeline Pipeline's}
80      * {@link PTransform PTransforms} are also owned by that {@link Pipeline}. A {@link PValue} owned by
81      * one {@link Pipeline} can be read only by {@link PTransform PTransforms} also owned by that {@link
82      * Pipeline}. {@link Pipeline Pipelines} can safely be executed concurrently.
83      *
84      * <p>Here is a typical example of use:
85      *
86      * <pre>{@code
87      * // Start by defining the options for the pipeline.
88      * PipelineOptions options = PipelineOptionsFactory.create();
89      * // Then create the pipeline. The runner is determined by the options.
90      * Pipeline p = Pipeline.create(options);
91      *
92      * // A root PTransform, like TextIO.Read or Create, gets added
93      * // to the Pipeline by being applied:
94      * PCollection<String> lines =
95      *     p.apply(TextIO.read().from("gs://bucket/dir/file*.txt"));
96      *
97      * // A Pipeline can have multiple root transforms:
98      * PCollection<String> moreLines =
99      *     p.apply(TextIO.read().from("gs://bucket/other/dir/file*.txt"));
100      * PCollection<String> yetMoreLines =
101      *     p.apply(Create.of("yet", "more", "lines").withCoder(StringUtf8Coder.of()));
102      *
103      * // Further PTransforms can be applied, in an arbitrary (acyclic) graph.
104      * // Subsequent PTransforms (and intermediate PCollections etc.) are
105      * // implicitly part of the same Pipeline.
106      * PCollection<String> allLines =
107      *     PCollectionList.of(lines).and(moreLines).and(yetMoreLines)
108      *     .apply(new Flatten<String>());
109      * PCollection<KV<String, Integer>> wordCounts =
110      *     allLines
111      *     .apply(ParDo.of(new ExtractWords()))
112      *     .apply(new Count<String>());
113      * PCollection<String> formattedWordCounts =
114      *     wordCounts.apply(ParDo.of(new FormatCounts()));
115      * formattedWordCounts.apply(TextIO.write().to("gs://bucket/dir/counts.txt"));
116      *
117      * // PTransforms aren't executed when they're applied, rather they're
118      * // just added to the Pipeline.  Once the whole Pipeline of PTransforms
119      * // is constructed, the Pipeline's PTransforms can be run using a
120      * // PipelineRunner.  The default PipelineRunner executes the Pipeline
121      * // directly, sequentially, in this one process, which is useful for
122      * // unit tests and simple experiments:
123      * p.run();
124      *
125      * }</pre>
126      */
127     @SuppressWarnings({

209       /**
210        * <b><i>For internal use only; no backwards-compatibility guarantees.</i></b>
211        *
212        * <p>Replaces all nodes that match a {@link PTransformOverride} in this pipeline. Overrides are
213        * applied in the order they are present within the list.

215       @Internal
216       public void replaceAll(List<PTransformOverride> overrides) {
217         for (PTransformOverride override : overrides) {
218           replace(override);
219         }
220         checkNoMoreMatches(overrides);
221       }
222     
223       private void checkNoMoreMatches(final List<PTransformOverride> overrides) {
224         traverseTopologically(
225             new PipelineVisitor.Defaults() {
226               SetMultimap<Node, PTransformOverride> matched = HashMultimap.create();
227     
228               @Override
229               public CompositeBehavior enterCompositeTransform(Node node) {
230                 if (!node.isRootNode()) {
231                   checkForMatches(node);
232                 }
233                 if (matched.containsKey(node)) {
234                   return CompositeBehavior.DO_NOT_ENTER_TRANSFORM;
235                 } else {
236                   return CompositeBehavior.ENTER_TRANSFORM;
237                 }
238               }
239     
240               @Override
241               public void leaveCompositeTransform(Node node) {
242                 if (node.isRootNode() && !matched.isEmpty()) {
243                   LOG.info(
244                       "Found nodes that matched overrides. Matches: {}. The match usually should be empty unless there are runner specific replacement transforms.",
245                       matched);
246                 }
247               }
248     
249               @Override
250               public void visitPrimitiveTransform(Node node) {
251                 checkForMatches(node);
252               }
253     
254               private void checkForMatches(Node node) {
255                 for (PTransformOverride override : overrides) {
256                   if (override
257                       .getMatcher()
258                       .matchesDuringValidation(node.toAppliedPTransform(getPipeline()))) {
259                     matched.put(node, override);
260                   }
261                 }
262               }
263             });
264       }
265     
266       private void replace(final PTransformOverride override) {
267         final Set<Node> matches = new HashSet<>();
268         final Set<Node> freedNodes = new HashSet<>();
269         traverseTopologically(
270             new PipelineVisitor.Defaults() {
271               @Override
272               public CompositeBehavior enterCompositeTransform(Node node) {
273                 if (!node.isRootNode() && freedNodes.contains(node.getEnclosingNode())) {
274                   // This node will be freed because its parent will be freed.
275                   freedNodes.add(node);
276                   return CompositeBehavior.ENTER_TRANSFORM;
277                 }
278                 if (!node.isRootNode()
279                     && override.getMatcher().matches(node.toAppliedPTransform(getPipeline()))) {
280                   matches.add(node);
281                   // This node will be freed. When we visit any of its children, they will also be freed
282                   freedNodes.add(node);
283                 }
284                 return CompositeBehavior.ENTER_TRANSFORM;
285               }
286     
287               @Override
288               public void visitPrimitiveTransform(Node node) {
289                 if (freedNodes.contains(node.getEnclosingNode())) {
290                   freedNodes.add(node);
291                 } else if (override.getMatcher().matches(node.toAppliedPTransform(getPipeline()))) {
292                   matches.add(node);
293                   freedNodes.add(node);
294                 }
295               }
296             });
297         for (Node freedNode : freedNodes) {
298           usedFullNames.remove(freedNode.getFullName());
299         }
300         for (Node match : matches) {
301           applyReplacement(match, override.getOverrideFactory());
302         }
303       }

313       /**
314        * Runs this {@link Pipeline} using the given {@link PipelineOptions}, using the runner specified
315        * by the options.
316        */
317       public PipelineResult run(PipelineOptions options) {
318         PipelineRunner<? extends PipelineResult> runner = PipelineRunner.fromOptions(options);
319         // Ensure all of the nodes are fully specified before a PipelineRunner gets access to the
320         // pipeline.
321         LOG.debug("Running {} via {}", this, runner);
322         try {
323           validate(options);
324           validateErrorHandlers();
325           return runner.run(this);
326         } catch (UserCodeException e) {
327           // This serves to replace the stack with one that ends here and
328           // is caused by the caught UserCodeException, thereby splicing
329           // out all the stack frames in between the PipelineRunner itself
330           // and where the worker calls into the user's code.
331           throw new PipelineExecutionException(e.getCause());
332         }
333       }

369       /**
370        * <b><i>For internal use only; no backwards-compatibility guarantees.</i></b>
371        *
372        * <p>A {@link PipelineVisitor} can be passed into {@link Pipeline#traverseTopologically} to be
373        * called for each of the transforms and values in the {@link Pipeline}.
374        */
375       @Internal
376       public interface PipelineVisitor {
377         /**
378          * Called before visiting anything values or transforms, as many uses of a visitor require
379          * access to the {@link Pipeline} object itself.
380          */
381         void enterPipeline(Pipeline p);
382     
383         /**
384          * Called for each composite transform after all topological predecessors have been visited but
385          * before any of its component transforms.
386          *
387          * <p>The return value controls whether or not child transforms are visited.
388          */
389         CompositeBehavior enterCompositeTransform(TransformHierarchy.Node node);
390     
391         /**
392          * Called for each composite transform after all of its component transforms and their outputs
393          * have been visited.
394          */
395         void leaveCompositeTransform(TransformHierarchy.Node node);
396     
397         /**
398          * Called for each primitive transform after all of its topological predecessors and inputs have
399          * been visited.
400          */
401         void visitPrimitiveTransform(TransformHierarchy.Node node);
402     
403         /** Called for each value after the transform that produced the value has been visited. */
404         void visitValue(PValue value, TransformHierarchy.Node producer);
405     
406         /** Called when all values and transforms in a {@link Pipeline} have been visited. */
407         void leavePipeline(Pipeline pipeline);
408     
409         /**
410          * Control enum for indicating whether or not a traversal should process the contents of a
411          * composite transform or not.
412          */
413         enum CompositeBehavior {
414           ENTER_TRANSFORM,
415           DO_NOT_ENTER_TRANSFORM
416         }
417     
418         /**
419          * Default no-op {@link PipelineVisitor} that enters all composite transforms. User
420          * implementations can override just those methods they are interested in.
421          */
422         class Defaults implements PipelineVisitor {
423     
424           private @Nullable Pipeline pipeline;
425     
426           protected Pipeline getPipeline() {
427             if (pipeline == null) {
428               throw new IllegalStateException(
429                   "Illegal access to pipeline after visitor traversal was completed");
430             }
431             return pipeline;
432           }
433     
434           @Override
435           public void enterPipeline(Pipeline pipeline) {
436             this.pipeline = checkNotNull(pipeline);
437           }
438     
439           @Override
440           public CompositeBehavior enterCompositeTransform(TransformHierarchy.Node node) {
441             return CompositeBehavior.ENTER_TRANSFORM;
442           }
443     
444           @Override
445           public void leaveCompositeTransform(TransformHierarchy.Node node) {}
446     
447           @Override
448           public void visitPrimitiveTransform(TransformHierarchy.Node node) {}
449     
450           @Override
451           public void visitValue(PValue value, TransformHierarchy.Node producer) {}
452     
453           @Override
454           public void leavePipeline(Pipeline pipeline) {
455             this.pipeline = null;
456           }
457         }
458       }

460       /**
461        * <b><i>For internal use only; no backwards-compatibility guarantees.</i></b>
462        *
463        * <p>Invokes the {@link PipelineVisitor PipelineVisitor's} {@link
464        * PipelineVisitor#visitPrimitiveTransform} and {@link PipelineVisitor#visitValue} operations on
465        * each of this {@link Pipeline Pipeline's} transform and value nodes, in forward topological
466        * order.

476         visitor.enterPipeline(this);
477         transforms.visit(visitor);
478         visitor.leavePipeline(this);
479       }

542       /**
543        * Applies a {@link PTransform} to the given {@link PInput}.
544        *
545        * @see Pipeline#apply
546        */
547       private <InputT extends PInput, OutputT extends POutput> OutputT applyInternal(
548           String name, InputT input, PTransform<? super InputT, OutputT> transform) {
549         String namePrefix = transforms.getCurrent().getFullName();
550         String uniqueName = uniquifyInternal(namePrefix, name);
551     
552         final String builtName = buildName(namePrefix, name);
553         instancePerName.put(builtName, transform);
554     
555         LOG.debug("Adding {} to {}", transform, this);
556         transforms.pushNode(uniqueName, input, transform);
557         try {
558           transforms.finishSpecifyingInput();
559           OutputT output = transform.expand(input);
560           transforms.setOutput(output);
561     
562           return output;
563         } finally {
564           transforms.popNode();
565         }
566       }

568       private <
569               InputT extends PInput,
570               OutputT extends POutput,
571               TransformT extends PTransform<? super InputT, OutputT>>
572           void applyReplacement(
573               Node original,
574               PTransformOverrideFactory<InputT, OutputT, TransformT> replacementFactory) {
575         PTransformReplacement<InputT, OutputT> replacement =
576             replacementFactory.getReplacementTransform(
577                 (AppliedPTransform<InputT, OutputT, TransformT>) original.toAppliedPTransform(this));
578         if (replacement.getTransform() == original.getTransform()) {
579           return;
580         }
581         InputT originalInput = replacement.getInput();
582         Map<TupleTag<?>, PCollection<?>> originalOutputs = original.getOutputs();
583     
584         LOG.debug("Replacing {} with {}", original, replacement);
585         transforms.replaceNode(original, originalInput, replacement.getTransform());
586         try {
587           OutputT newOutput = replacement.getTransform().expand(originalInput);
588           Map<PCollection<?>, ReplacementOutput> originalToReplacement =
589               replacementFactory.mapOutputs(original.getOutputs(), newOutput);
590           // Ensure the internal TransformHierarchy data structures are consistent.
591           transforms.setOutput(newOutput);
592           transforms.replaceOutputs(originalToReplacement);
593           checkState(
594               ImmutableSet.copyOf(originalOutputs.values())
595                   .equals(ImmutableSet.copyOf(transforms.getCurrent().getOutputs().values())),
596               "After replacing %s with %s, outputs were not rewired correctly:"
597                   + " Original outputs %s became %s.",
598               original,
599               transforms.getCurrent(),
600               originalOutputs,
601               transforms.getCurrent().getOutputs());
602         } finally {
603           transforms.popNode();
604         }
605       }

607       @VisibleForTesting
608       void validate(PipelineOptions options) {
609         this.traverseTopologically(new ValidateVisitor(options));
610         final Collection<Map.Entry<String, Collection<PTransform<?, ?>>>> errors =
611             Collections2.filter(instancePerName.asMap().entrySet(), Predicates.not(new IsUnique<>()));
612         if (!errors.isEmpty()) {
613           switch (options.getStableUniqueNames()) {
614             case OFF:
615               break;
616             case WARNING:
617               LOG.warn(
618                   "The following transforms do not have stable unique names: {}",
619                   Joiner.on(", ").join(transform(errors, new KeysExtractor())));
620               break;
621             case ERROR: // be very verbose here since it will just fail the execution
622               throw new IllegalStateException(
623                   String.format(
624                           "Pipeline update will not be possible because the following transforms do"
625                               + " not have stable unique names: %s.",
626                           Joiner.on(", ").join(transform(errors, new KeysExtractor())))
627                       + "\n\n"
628                       + "Conflicting instances:\n"
629                       + Joiner.on("\n")
630                           .join(transform(errors, new UnstableNameToMessage(instancePerName)))
631                       + "\n\nYou can fix it adding a name when you call apply(): "
632                       + "pipeline.apply(<name>, <transform>).");
633             default:
634               throw new IllegalArgumentException(
635                   "Unrecognized value for stable unique names: " + options.getStableUniqueNames());
636           }
637         }
638       }
639     
640       /**
641        * Returns a unique name for a transform with the given prefix (from enclosing transforms) and
642        * initial name.
643        */
644       private String uniquifyInternal(String namePrefix, String origName) {
645         String name = origName;
646         int suffixNum = 2;
647         while (true) {
648           String candidate = buildName(namePrefix, name);
649           if (usedFullNames.add(candidate)) {
650             return candidate;
651           }
652           // A duplicate!  Retry.
653           name = origName + suffixNum++;
654         }
655       }
656     
657       /** Builds a name from a "/"-delimited prefix and a name. */
658       private String buildName(String namePrefix, String name) {
659         return namePrefix.isEmpty() ? name : namePrefix + "/" + name;
660       }

662       private static class ValidateVisitor extends PipelineVisitor.Defaults {
663     
664         private final PipelineOptions options;
665     
666         public ValidateVisitor(PipelineOptions options) {
667           this.options = options;
668         }
669     
670         @Override
671         public CompositeBehavior enterCompositeTransform(Node node) {
672           if (node.getTransform() != null) {
673             node.getTransform().validate(options, node.getInputs(), node.getOutputs());
674           }
675           return CompositeBehavior.ENTER_TRANSFORM;
676         }
677     
678         @Override
679         public void visitPrimitiveTransform(Node node) {
680           node.getTransform().validate(options, node.getInputs(), node.getOutputs());
681         }

691       private static class UnstableNameToMessage
692           implements Function<Map.Entry<String, Collection<PTransform<?, ?>>>, String> {
693         private final Multimap<String, PTransform<?, ?>> instances;
694     
695         private UnstableNameToMessage(final Multimap<String, PTransform<?, ?>> instancePerName) {
696           this.instances = instancePerName;
697         }
698     
699         @SuppressFBWarnings(
700             value = "NP_METHOD_PARAMETER_TIGHTENS_ANNOTATION",
701             justification = "https://github.com/google/guava/issues/920")
702         @Override
703         public String apply(@Nonnull final Map.Entry<String, Collection<PTransform<?, ?>>> input) {
704           final Collection<PTransform<?, ?>> values = instances.get(input.getKey());
705           return "- name="
706               + input.getKey()
707               + ":\n"
708               + Joiner.on("\n").join(transform(values, new TransformToMessage()));
709         }
710       }
