261        * Construct a runner from the provided options.
262        *
263        * @param options Properties that configure the runner.
264        * @return The newly created runner.
265        */
266       public static DataflowRunner fromOptions(PipelineOptions options) {
267         DataflowPipelineOptions dataflowOptions =
268             PipelineOptionsValidator.validate(DataflowPipelineOptions.class, options);
269         ArrayList<String> missing = new ArrayList<>();
270     
271         if (dataflowOptions.getAppName() == null) {
272           missing.add("appName");
273         }
274     
275         if (Strings.isNullOrEmpty(dataflowOptions.getRegion())
276             && isServiceEndpoint(dataflowOptions.getDataflowEndpoint())) {
277           missing.add("region");
278         }
279         if (missing.size() > 0) {
280           throw new IllegalArgumentException(
281               "Missing required pipeline options: " + Joiner.on(',').join(missing));
282         }
283     
284         validateWorkerSettings(
285             PipelineOptionsValidator.validate(DataflowPipelineWorkerPoolOptions.class, options));
286     
287         PathValidator validator = dataflowOptions.getPathValidator();
288         String gcpTempLocation;
289         try {
290           gcpTempLocation = dataflowOptions.getGcpTempLocation();
291         } catch (Exception e) {
292           throw new IllegalArgumentException(
293               "DataflowRunner requires gcpTempLocation, "
294                   + "but failed to retrieve a value from PipelineOptions",
295               e);
296         }
297         validator.validateOutputFilePrefixSupported(gcpTempLocation);
298     
299         String stagingLocation;
300         try {
301           stagingLocation = dataflowOptions.getStagingLocation();
302         } catch (Exception e) {
303           throw new IllegalArgumentException(
304               "DataflowRunner requires stagingLocation, "
305                   + "but failed to retrieve a value from PipelineOptions",
306               e);
307         }
308         validator.validateOutputFilePrefixSupported(stagingLocation);
309     
310         if (!isNullOrEmpty(dataflowOptions.getSaveProfilesToGcs())) {
311           validator.validateOutputFilePrefixSupported(dataflowOptions.getSaveProfilesToGcs());
312         }
313     
314         if (dataflowOptions.getFilesToStage() != null) {
315           // The user specifically requested these files, so fail now if they do not exist.
316           // (automatically detected classpath elements are permitted to not exist, so later
317           // staging will not fail on nonexistent files)
318           dataflowOptions.getFilesToStage().stream()
319               .forEach(
320                   stagedFileSpec -> {
321                     File localFile;
322                     if (stagedFileSpec.contains("=")) {
323                       String[] components = stagedFileSpec.split("=", 2);
324                       localFile = new File(components[1]);
325                     } else {
326                       localFile = new File(stagedFileSpec);
327                     }
328                     if (!localFile.exists()) {
329                       // should be FileNotFoundException, but for build-time backwards compatibility
330                       // cannot add checked exception
331                       throw new RuntimeException(
332                           String.format("Non-existent files specified in filesToStage: %s", localFile));
333                     }
334                   });
335         } else {
336           dataflowOptions.setFilesToStage(
337               detectClassPathResourcesToStage(DataflowRunner.class.getClassLoader(), options));
338           if (dataflowOptions.getFilesToStage().isEmpty()) {
339             throw new IllegalArgumentException("No files to stage has been found.");
340           } else {
341             LOG.info(
342                 "PipelineOptions.filesToStage was not specified. "
343                     + "Defaulting to files from the classpath: will stage {} files. "
344                     + "Enable logging at DEBUG level to see which files will be staged.",
345                 dataflowOptions.getFilesToStage().size());
346             LOG.debug("Classpath elements: {}", dataflowOptions.getFilesToStage());
347           }
348         }
349     
350         // Verify jobName according to service requirements, truncating converting to lowercase if
351         // necessary.
352         String jobName = dataflowOptions.getJobName().toLowerCase();
353         checkArgument(
354             jobName.matches("[a-z]([-a-z0-9]*[a-z0-9])?"),
355             "JobName invalid; the name must consist of only the characters "
356                 + "[-a-z0-9], starting with a letter and ending with a letter "
357                 + "or number");
358         if (!jobName.equals(dataflowOptions.getJobName())) {
359           LOG.info(
360               "PipelineOptions.jobName did not match the service requirements. "
361                   + "Using {} instead of {}.",
362               jobName,
363               dataflowOptions.getJobName());
364         }
365         dataflowOptions.setJobName(jobName);
366     
367         // Verify project
368         String project = dataflowOptions.getProject();
369         if (project.matches("[0-9]*")) {
370           throw new IllegalArgumentException(
371               "Project ID '"
372                   + project
373                   + "' invalid. Please make sure you specified the Project ID, not project number.");
374         } else if (!project.matches(PROJECT_ID_REGEXP)) {
375           throw new IllegalArgumentException(
376               "Project ID '"
377                   + project
378                   + "' invalid. Please make sure you specified the Project ID, not project"
379                   + " description.");
380         }
381     
382         DataflowPipelineDebugOptions debugOptions =
383             dataflowOptions.as(DataflowPipelineDebugOptions.class);
384         // Verify the number of worker threads is a valid value
385         if (debugOptions.getNumberOfWorkerHarnessThreads() < 0) {
386           throw new IllegalArgumentException(
387               "Number of worker harness threads '"
388                   + debugOptions.getNumberOfWorkerHarnessThreads()
389                   + "' invalid. Please make sure the value is non-negative.");
390         }
391     
392         // Verify that if recordJfrOnGcThrashing is set, the pipeline is at least on java 11
393         if (dataflowOptions.getRecordJfrOnGcThrashing()
394             && Environments.getJavaVersion() == Environments.JavaVersion.java8) {
395           throw new IllegalArgumentException(
396               "recordJfrOnGcThrashing is only supported on java 9 and up.");
397         }
398     
399         if (dataflowOptions.isStreaming() && dataflowOptions.getGcsUploadBufferSizeBytes() == null) {
400           dataflowOptions.setGcsUploadBufferSizeBytes(GCS_UPLOAD_BUFFER_SIZE_BYTES_DEFAULT);
401         }
402     
403         // Adding the Java version to the SDK name for user's and support convenience.
404         String agentJavaVer = "(JRE 8 environment)";
405         if (Environments.getJavaVersion() != Environments.JavaVersion.java8) {
406           agentJavaVer =
407               String.format("(JRE %s environment)", Environments.getJavaVersion().specification());
408         }
409     
410         DataflowRunnerInfo dataflowRunnerInfo = DataflowRunnerInfo.getDataflowRunnerInfo();
411         String userAgentName = dataflowRunnerInfo.getName();
412         Preconditions.checkArgument(
413             !userAgentName.equals(""), "Dataflow runner's `name` property cannot be empty.");
414         String userAgentVersion = dataflowRunnerInfo.getVersion();
415         Preconditions.checkArgument(
416             !userAgentVersion.equals(""), "Dataflow runner's `version` property cannot be empty.");
417         String userAgent =
418             String.format("%s/%s%s", userAgentName, userAgentVersion, agentJavaVer).replace(" ", "_");
419         dataflowOptions.setUserAgent(userAgent);
420     
421         return new DataflowRunner(dataflowOptions);
422       }

428       static void validateSdkContainerImageOptions(DataflowPipelineWorkerPoolOptions workerOptions) {
429         // Check against null - empty string value for workerHarnessContainerImage
430         // must be preserved for legacy dataflowWorkerJar to work.
431         String sdkContainerOption = workerOptions.getSdkContainerImage();
432         String workerHarnessOption = workerOptions.getWorkerHarnessContainerImage();
433         Preconditions.checkArgument(
434             sdkContainerOption == null
435                 || workerHarnessOption == null
436                 || sdkContainerOption.equals(workerHarnessOption),
437             "Cannot use legacy option workerHarnessContainerImage with sdkContainerImage. Prefer sdkContainerImage.");
438     
439         // Default to new option, which may be null.
440         String containerImage = workerOptions.getSdkContainerImage();
441         if (workerOptions.getWorkerHarnessContainerImage() != null
442             && workerOptions.getSdkContainerImage() == null) {
443           // Set image to old option if old option was set but new option is not set.
444           LOG.warn(
445               "Prefer --sdkContainerImage over deprecated legacy option --workerHarnessContainerImage.");
446           containerImage = workerOptions.getWorkerHarnessContainerImage();
447         }
448     
449         // Make sure both options have same value.
450         workerOptions.setSdkContainerImage(containerImage);
451         workerOptions.setWorkerHarnessContainerImage(containerImage);
452       }

481             !hasExperimentWorkerRegion || gcpOptions.getWorkerRegion() == null,
482             "Experiment worker_region and option workerRegion are mutually exclusive.");
483         Preconditions.checkArgument(
484             !hasExperimentWorkerRegion || gcpOptions.getWorkerZone() == null,
485             "Experiment worker_region and option workerZone are mutually exclusive.");
486     
487         if (gcpOptions.getZone() != null) {
488           LOG.warn("Option --zone is deprecated. Please use --workerZone instead.");
489           gcpOptions.setWorkerZone(gcpOptions.getZone());
490           gcpOptions.setZone(null);

522       private List<PTransformOverride> getOverrides(boolean streaming) {
523         ImmutableList.Builder<PTransformOverride> overridesBuilder = ImmutableList.builder();
524     
525         // Create is implemented in terms of a Read, so it must precede the override to Read in
526         // streaming
527         overridesBuilder
528             .add(
529                 PTransformOverride.of(
530                     PTransformMatchers.flattenWithDuplicateInputs(),
531                     DeduplicatedFlattenFactory.create()))
532             .add(
533                 PTransformOverride.of(
534                     PTransformMatchers.emptyFlatten(), EmptyFlattenAsCreateFactory.instance()));
535     
536         if (streaming) {
537           // For update compatibility, always use a Read for Create in streaming mode.
538           overridesBuilder.add(
539               PTransformOverride.of(
540                   PTransformMatchers.classEqualTo(Create.Values.class), new AlwaysCreateViaRead()));
541         }
542     
543         // By default Dataflow runner replaces single-output ParDo with a ParDoSingle override.
544         // However, we want a different expansion for single-output splittable ParDo.
545         overridesBuilder
546             .add(
547                 PTransformOverride.of(
548                     PTransformMatchers.splittableParDoSingle(),
549                     new ReflectiveOneToOneOverrideFactory(
550                         SplittableParDoOverrides.ParDoSingleViaMulti.class, this)))
551             .add(
552                 PTransformOverride.of(
553                     PTransformMatchers.splittableParDoMulti(),
554                     new SplittableParDoOverrides.SplittableParDoOverrideFactory()));

557           if (!hasExperiment(options, ENABLE_CUSTOM_PUBSUB_SOURCE)) {
558             overridesBuilder.add(
559                 PTransformOverride.of(
560                     PTransformMatchers.classEqualTo(PubsubUnboundedSource.class),

562           }
563           if (!hasExperiment(options, ENABLE_CUSTOM_PUBSUB_SINK)) {
564             overridesBuilder.add(
565                 PTransformOverride.of(
566                     PTransformMatchers.classEqualTo(PubsubUnboundedSink.class),
567                     new StreamingPubsubIOWriteOverrideFactory(this)));
568           }
569     
570           try {
571             overridesBuilder.add(KafkaIO.Read.KAFKA_READ_OVERRIDE);
572           } catch (NoClassDefFoundError e) {
573             // Do nothing. io-kafka is an optional dependency of runners-google-cloud-dataflow-java
574             // and only needed when KafkaIO is used in the pipeline.
575           }
576     
577           overridesBuilder.add(SubscribeTransform.V1_READ_OVERRIDE);
578     
579           if (!hasExperiment(options, "enable_file_dynamic_sharding")) {
580             overridesBuilder.add(
581                 PTransformOverride.of(
582                     PTransformMatchers.writeWithRunnerDeterminedSharding(),
583                     new StreamingShardedWriteFactory(options)));
584           }

587               PTransformOverride.of(
588                   PTransformMatchers.groupIntoBatches(),
589                   new GroupIntoBatchesOverride.StreamingGroupIntoBatchesOverrideFactory(this)));
590     
591           overridesBuilder.add(
592               PTransformOverride.of(

594                   new GroupIntoBatchesOverride.StreamingGroupIntoBatchesWithShardedKeyOverrideFactory(
595                       this)));
596

599                   // Streaming Bounded Read is implemented in terms of Streaming Unbounded Read, and
600                   // must precede it

602                       PTransformMatchers.classEqualTo(Read.Bounded.class),
603                       new StreamingBoundedReadOverrideFactory()))
604               .add(

606                       PTransformMatchers.classEqualTo(Read.Unbounded.class),
607                       new StreamingUnboundedReadOverrideFactory()));

610               PTransformOverride.of(
611                   PTransformMatchers.classEqualTo(View.CreatePCollectionView.class),
612                   new StreamingCreatePCollectionViewFactory()));
613     
614           // Dataflow Streaming runner overrides the SPLITTABLE_PROCESS_KEYED transform
615           // natively in the Dataflow service.
616         } else {

618           overridesBuilder
619               // Replace GroupIntoBatches before the state/timer replacements below since
620               // GroupIntoBatches internally uses a stateful DoFn.

622                   PTransformOverride.of(
623                       PTransformMatchers.classEqualTo(GroupIntoBatches.class),

628                       new GroupIntoBatchesOverride.BatchGroupIntoBatchesWithShardedKeyOverrideFactory<>(
629                           this)));
630     
631           overridesBuilder
632               // State and timer pardos are implemented by expansion to GBK-then-ParDo
633               .add(
634                   PTransformOverride.of(
635                       PTransformMatchers.stateOrTimerParDoMulti(),
636                       BatchStatefulParDoOverrides.multiOutputOverrideFactory(options)))
637               .add(
638                   PTransformOverride.of(
639                       PTransformMatchers.stateOrTimerParDoSingle(),
640                       BatchStatefulParDoOverrides.singleOutputOverrideFactory()));
641           // Dataflow Batch runner uses the naive override of the SPLITTABLE_PROCESS_KEYED transform
642           // for now, but eventually (when liquid sharding is implemented) will also override it
643           // natively in the Dataflow service.
644           overridesBuilder.add(
645               PTransformOverride.of(
646                   PTransformMatchers.splittableProcessKeyedBounded(),
647                   new SplittableParDoNaiveBounded.OverrideFactory()));
648     
649           overridesBuilder
650               .add(
651                   PTransformOverride.of(
652                       PTransformMatchers.classEqualTo(View.AsMap.class),
653                       new ReflectiveViewOverrideFactory(BatchViewOverrides.BatchViewAsMap.class, this)))
654               .add(
655                   PTransformOverride.of(
656                       PTransformMatchers.classEqualTo(View.AsMultimap.class),
657                       new ReflectiveViewOverrideFactory(
658                           BatchViewOverrides.BatchViewAsMultimap.class, this)))
659               .add(
660                   PTransformOverride.of(
661                       PTransformMatchers.classEqualTo(Combine.GloballyAsSingletonView.class),
662                       new CombineGloballyAsSingletonViewOverrideFactory(this)))
663               .add(
664                   PTransformOverride.of(
665                       PTransformMatchers.classEqualTo(View.AsList.class),
666                       new ReflectiveViewOverrideFactory(
667                           BatchViewOverrides.BatchViewAsList.class, this)))
668               .add(
669                   PTransformOverride.of(
670                       PTransformMatchers.classEqualTo(View.AsIterable.class),
671                       new ReflectiveViewOverrideFactory(
672                           BatchViewOverrides.BatchViewAsIterable.class, this)));
673         }
674         /* TODO(Beam-4684): Support @RequiresStableInput on Dataflow in a more intelligent way
675         Use Reshuffle might cause an extra and unnecessary shuffle to be inserted. To enable this, we
676         should make sure that we do not add extra shuffles for transforms whose input is already stable.
677         // Uses Reshuffle, so has to be before the Reshuffle override
678         overridesBuilder.add(
679             PTransformOverride.of(
680                 PTransformMatchers.requiresStableInputParDoSingle(),
681                 RequiresStableInputParDoOverrides.singleOutputOverrideFactory()));
682         // Uses Reshuffle, so has to be before the Reshuffle override
683         overridesBuilder.add(
684             PTransformOverride.of(
685                 PTransformMatchers.requiresStableInputParDoMulti(),
686                 RequiresStableInputParDoOverrides.multiOutputOverrideFactory()));
687         */
688         overridesBuilder
689             .add(
690                 PTransformOverride.of(
691                     PTransformMatchers.classEqualTo(Reshuffle.class), new ReshuffleOverrideFactory()))
692             // Order is important. Streaming views almost all use Combine internally.
693             .add(
694                 PTransformOverride.of(
695                     new DataflowPTransformMatchers.CombineValuesWithoutSideInputsPTransformMatcher(),
696                     new PrimitiveCombineGroupedValuesOverrideFactory()))
697             .add(
698                 PTransformOverride.of(
699                     PTransformMatchers.classEqualTo(ParDo.SingleOutput.class),
700                     new PrimitiveParDoSingleFactory()));
701     
702         overridesBuilder.add(
703             PTransformOverride.of(
704                 PTransformMatchers.classEqualTo(RedistributeByKey.class),
705                 new RedistributeByKeyOverrideFactory()));
706     
707         if (streaming) {
708           // For update compatibility, always use a Read for Create in streaming mode.
709           overridesBuilder
710               .add(
711                   PTransformOverride.of(
712                       PTransformMatchers.classEqualTo(Create.Values.class), new AlwaysCreateViaRead()))
713               // Create is implemented in terms of BoundedRead.
714               .add(
715                   PTransformOverride.of(
716                       PTransformMatchers.classEqualTo(Read.Bounded.class),
717                       new StreamingBoundedReadOverrideFactory()))
718               // Streaming Bounded Read is implemented in terms of Streaming Unbounded Read.
719               .add(
720                   PTransformOverride.of(
721                       PTransformMatchers.classEqualTo(Read.Unbounded.class),
722                       new StreamingUnboundedReadOverrideFactory()))
723               .add(
724                   PTransformOverride.of(
725                       PTransformMatchers.classEqualTo(ParDo.SingleOutput.class),
726                       new PrimitiveParDoSingleFactory()));
727         }
728     
729         return overridesBuilder.build();
730       }

764        * Replace the View.AsYYY transform with specialized view overrides for Dataflow. It is required
765        * that the new replacement transform uses the supplied PCollectionView and does not create
766        * another instance.
767        */
768       private static class ReflectiveViewOverrideFactory<InputT, ViewT>
769           implements PTransformOverrideFactory<
770               PCollection<InputT>,
771               PCollectionView<ViewT>,
772               PTransform<PCollection<InputT>, PCollectionView<ViewT>>> {
773     
774         final Class<PTransform<PCollection<InputT>, PCollectionView<ViewT>>> replacement;
775         final DataflowRunner runner;
776     
777         private ReflectiveViewOverrideFactory(
778             Class<PTransform<PCollection<InputT>, PCollectionView<ViewT>>> replacement,
779             DataflowRunner runner) {
780           this.replacement = replacement;
781           this.runner = runner;
782         }
783     
784         CreatePCollectionView<ViewT, ViewT> findCreatePCollectionView(
785             final AppliedPTransform<
786                     PCollection<InputT>,
787                     PCollectionView<ViewT>,
788                     PTransform<PCollection<InputT>, PCollectionView<ViewT>>>
789                 transform) {
790           final AtomicReference<CreatePCollectionView> viewTransformRef = new AtomicReference<>();
791           transform
792               .getPipeline()
793               .traverseTopologically(
794                   new PipelineVisitor.Defaults() {
795                     // Stores whether we have entered the expected composite view transform.
796                     private boolean tracking = false;
797     
798                     @Override
799                     public CompositeBehavior enterCompositeTransform(Node node) {
800                       if (transform.getTransform() == node.getTransform()) {
801                         tracking = true;
802                       }
803                       return super.enterCompositeTransform(node);
804                     }
805     
806                     @Override
807                     public void visitPrimitiveTransform(Node node) {
808                       if (tracking && node.getTransform() instanceof CreatePCollectionView) {
809                         checkState(
810                             viewTransformRef.compareAndSet(
811                                 null, (CreatePCollectionView) node.getTransform()),
812                             "Found more than one instance of a CreatePCollectionView when "
813                                 + "attempting to replace %s, found [%s, %s]",
814                             replacement,
815                             viewTransformRef.get(),
816                             node.getTransform());
817                       }
818                     }
819     
820                     @Override
821                     public void leaveCompositeTransform(Node node) {
822                       if (transform.getTransform() == node.getTransform()) {
823                         tracking = false;
824                       }
825                     }
826                   });
827     
828           checkState(
829               viewTransformRef.get() != null,
830               "Expected to find CreatePCollectionView contained within %s",
831               transform.getTransform());
832           return viewTransformRef.get();
833         }
834     
835         @Override
836         public PTransformReplacement<PCollection<InputT>, PCollectionView<ViewT>>
837             getReplacementTransform(
838                 final AppliedPTransform<
839                         PCollection<InputT>,
840                         PCollectionView<ViewT>,
841                         PTransform<PCollection<InputT>, PCollectionView<ViewT>>>
842                     transform) {
843     
844           PTransform<PCollection<InputT>, PCollectionView<ViewT>> rep =
845               InstanceBuilder.ofType(replacement)
846                   .withArg(DataflowRunner.class, runner)
847                   .withArg(CreatePCollectionView.class, findCreatePCollectionView(transform))
848                   .build();
849           return PTransformReplacement.of(
850               PTransformReplacements.getSingletonMainInput(transform), (PTransform) rep);
851         }
852     
853         @Override
854         public Map<PCollection<?>, ReplacementOutput> mapOutputs(
855             Map<TupleTag<?>, PCollection<?>> outputs, PCollectionView<ViewT> newOutput) {
856           /*
857           The output of View.AsXYZ is a PCollectionView that expands to the PCollection to be materialized.
858           The PCollectionView itself must have the same tag since that tag may have been embedded in serialized DoFns
859           previously and cannot easily be rewired. The PCollection may differ, so we rewire it, even if the rewiring
860           is a noop.
861           */
862           return ReplacementOutputs.singleton(outputs, newOutput);
863         }

909       protected RunnerApi.Pipeline applySdkEnvironmentOverrides(
910           RunnerApi.Pipeline pipeline, DataflowPipelineOptions options) {
911         String sdkHarnessContainerImageOverrides = options.getSdkHarnessContainerImageOverrides();
912         String[] overrides =
913             Strings.isNullOrEmpty(sdkHarnessContainerImageOverrides)
914                 ? new String[0]
915                 : sdkHarnessContainerImageOverrides.split(",", -1);
916         if (overrides.length % 2 != 0) {
917           throw new RuntimeException(
918               "invalid syntax for SdkHarnessContainerImageOverrides: "
919                   + options.getSdkHarnessContainerImageOverrides());
920         }
921         RunnerApi.Pipeline.Builder pipelineBuilder = pipeline.toBuilder();
922         RunnerApi.Components.Builder componentsBuilder = pipelineBuilder.getComponentsBuilder();
923         componentsBuilder.clearEnvironments();
924         for (Map.Entry<String, RunnerApi.Environment> entry :
925             pipeline.getComponents().getEnvironmentsMap().entrySet()) {
926           RunnerApi.Environment.Builder environmentBuilder = entry.getValue().toBuilder();
927           if (BeamUrns.getUrn(RunnerApi.StandardEnvironments.Environments.DOCKER)
928               .equals(environmentBuilder.getUrn())) {
929             RunnerApi.DockerPayload dockerPayload;
930             try {
931               dockerPayload = RunnerApi.DockerPayload.parseFrom(environmentBuilder.getPayload());
932             } catch (InvalidProtocolBufferException e) {
933               throw new RuntimeException("Error parsing environment docker payload.", e);
934             }
935             String containerImage = dockerPayload.getContainerImage();
936             boolean updated = false;
937             for (int i = 0; i < overrides.length; i += 2) {
938               containerImage = containerImage.replaceAll(overrides[i], overrides[i + 1]);
939               if (!containerImage.equals(dockerPayload.getContainerImage())) {
940                 updated = true;
941               }
942             }
943             if (containerImage.startsWith("apache/beam")
944                 && !updated
945                 // don't update if the container image is already configured by DataflowRunner
946                 && !containerImage.equals(getContainerImageForJob(options))) {
947               containerImage =
948                   DataflowRunnerInfo.getDataflowRunnerInfo().getContainerImageBaseRepository()
949                       + containerImage.substring(containerImage.lastIndexOf("/"));
950             }
951             environmentBuilder.setPayload(
952                 RunnerApi.DockerPayload.newBuilder()
953                     .setContainerImage(containerImage)
954                     .build()
955                     .toByteString());
956           }
957           componentsBuilder.putEnvironments(entry.getKey(), environmentBuilder.build());
958         }
959         return pipelineBuilder.build();
960       }

964         RunnerApi.Pipeline.Builder pipelineBuilder = pipeline.toBuilder();
965         RunnerApi.Components.Builder componentsBuilder = pipelineBuilder.getComponentsBuilder();
966         componentsBuilder.clearEnvironments();
967         for (Map.Entry<String, RunnerApi.Environment> entry :
968             pipeline.getComponents().getEnvironmentsMap().entrySet()) {
969           RunnerApi.Environment.Builder environmentBuilder = entry.getValue().toBuilder();
970           environmentBuilder.clearDependencies();
971           for (RunnerApi.ArtifactInformation info : entry.getValue().getDependenciesList()) {
972             if (!BeamUrns.getUrn(RunnerApi.StandardArtifacts.Types.FILE).equals(info.getTypeUrn())) {
973               throw new RuntimeException(
974                   String.format("unsupported artifact type %s", info.getTypeUrn()));
975             }
976             RunnerApi.ArtifactFilePayload filePayload;
977             try {
978               filePayload = RunnerApi.ArtifactFilePayload.parseFrom(info.getTypePayload());
979             } catch (InvalidProtocolBufferException e) {
980               throw new RuntimeException("Error parsing artifact file payload.", e);
981             }
982             String stagedName;
983             if (BeamUrns.getUrn(RunnerApi.StandardArtifacts.Roles.STAGING_TO)
984                 .equals(info.getRoleUrn())) {
985               try {
986                 RunnerApi.ArtifactStagingToRolePayload stagingPayload =
987                     RunnerApi.ArtifactStagingToRolePayload.parseFrom(info.getRolePayload());
988                 stagedName = stagingPayload.getStagedName();
989               } catch (InvalidProtocolBufferException e) {
990                 throw new RuntimeException("Error parsing artifact staging_to role payload.", e);
991               }
992             } else {
993               try {
994                 File source = new File(filePayload.getPath());
995                 HashCode hashCode = Files.asByteSource(source).hash(Hashing.sha256());
996                 stagedName = Environments.createStagingFileName(source, hashCode);
997               } catch (IOException e) {
998                 throw new RuntimeException(
999                     String.format("Error creating staged name for artifact %s", filePayload.getPath()),
1000                     e);
1001               }
1002             }
1003             environmentBuilder.addDependencies(
1004                 info.toBuilder()
1005                     .setTypeUrn(BeamUrns.getUrn(RunnerApi.StandardArtifacts.Types.URL))
1006                     .setTypePayload(
1007                         RunnerApi.ArtifactUrlPayload.newBuilder()
1008                             .setUrl(
1009                                 FileSystems.matchNewResource(options.getStagingLocation(), true)
1010                                     .resolve(
1011                                         stagedName, ResolveOptions.StandardResolveOptions.RESOLVE_FILE)
1012                                     .toString())
1013                             .setSha256(filePayload.getSha256())
1014                             .build()
1015                             .toByteString()));
1016           }
1017           componentsBuilder.putEnvironments(entry.getKey(), environmentBuilder.build());
1018         }
1019         return pipelineBuilder.build();

1915           PubsubUnboundedSource overriddenTransform, StepTranslationContext stepTranslationContext) {
1916         stepTranslationContext.addInput(PropertyNames.FORMAT, "pubsub");
1917         if (overriddenTransform.getTopicProvider() != null) {
1918           if (overriddenTransform.getTopicProvider().isAccessible()) {
1919             stepTranslationContext.addInput(
1920                 PropertyNames.PUBSUB_TOPIC, overriddenTransform.getTopic().getFullPath());
1921           } else {
1922             stepTranslationContext.addInput(
1923                 PropertyNames.PUBSUB_TOPIC_OVERRIDE,
1924                 ((NestedValueProvider) overriddenTransform.getTopicProvider()).propertyName());
1925           }

1927         if (overriddenTransform.getSubscriptionProvider() != null) {
1928           if (overriddenTransform.getSubscriptionProvider().isAccessible()) {
1929             stepTranslationContext.addInput(
1930                 PropertyNames.PUBSUB_SUBSCRIPTION, overriddenTransform.getSubscription().getFullPath());
1931           } else {

1933                 PropertyNames.PUBSUB_SUBSCRIPTION_OVERRIDE,
1934                 ((NestedValueProvider) overriddenTransform.getSubscriptionProvider()).propertyName());
1935           }
1936         }
1937         if (overriddenTransform.getTimestampAttribute() != null) {
1938           stepTranslationContext.addInput(
1939               PropertyNames.PUBSUB_TIMESTAMP_ATTRIBUTE, overriddenTransform.getTimestampAttribute());
1940         }
1941         if (overriddenTransform.getIdAttribute() != null) {
1942           stepTranslationContext.addInput(
1943               PropertyNames.PUBSUB_ID_ATTRIBUTE, overriddenTransform.getIdAttribute());
1944         }

1946         // the attributes or messageId, we supply an identity "parse fn" so the worker will
1947         // read PubsubMessage's from Windmill and simply pass them around; and in case it
1948         // doesn't need attributes, we're already implicitly using a "Coder" that interprets
1949         // the data as a PubsubMessage's payload.
1950         if (overriddenTransform.getNeedsAttributes() || overriddenTransform.getNeedsMessageId()) {
1951           stepTranslationContext.addInput(
1952               PropertyNames.PUBSUB_SERIALIZED_ATTRIBUTES_FN,
1953               byteArrayToJsonString(serializeToByteArray(new IdentityMessageFn())));

2234       /** Remove values with duplicate ids. */
2235       private static class Deduplicate<T>
2236           extends PTransform<PCollection<ValueWithRecordId<T>>, PCollection<T>> {
2237     
2238         // Use a finite set of keys to improve bundling.  Without this, the key space
2239         // will be the space of ids which is potentially very large, which results in much
2240         // more per-key overhead.
2241         private static final int NUM_RESHARD_KEYS = 10000;
2242     
2243         @Override
2244         public PCollection<T> expand(PCollection<ValueWithRecordId<T>> input) {
2245           return input
2246               .apply(
2247                   WithKeys.of(
2248                           (ValueWithRecordId<T> value) ->
2249                               Arrays.hashCode(value.getId()) % NUM_RESHARD_KEYS)
2250                       .withKeyType(TypeDescriptors.integers()))

2447       static class StreamingShardedWriteFactory<UserT, DestinationT, OutputT>
2448           implements PTransformOverrideFactory<
2449               PCollection<UserT>,
2450               WriteFilesResult<DestinationT>,
2451               WriteFiles<UserT, DestinationT, OutputT>> {
2452     
2453         // We pick 10 as a default, as it works well with the default number of workers started
2454         // by Dataflow.
2455         static final int DEFAULT_NUM_SHARDS = 10;
2456         DataflowPipelineWorkerPoolOptions options;
2457     
2458         StreamingShardedWriteFactory(PipelineOptions options) {
2459           this.options = options.as(DataflowPipelineWorkerPoolOptions.class);
2460         }
2461     
2462         @Override
2463         public PTransformReplacement<PCollection<UserT>, WriteFilesResult<DestinationT>>
2464             getReplacementTransform(
2465                 AppliedPTransform<
2466                         PCollection<UserT>,
2467                         WriteFilesResult<DestinationT>,
2468                         WriteFiles<UserT, DestinationT, OutputT>>
2469                     transform) {
2470           // By default, if numShards is not set WriteFiles will produce one file per bundle. In
2471           // streaming, there are large numbers of small bundles, resulting in many tiny files.
2472           // Instead we pick max workers * 2 to ensure full parallelism, but prevent too-many files.
2473           // (current_num_workers * 2 might be a better choice, but that value is not easily available
2474           // today).
2475           // If the user does not set either numWorkers or maxNumWorkers, default to 10 shards.
2476           int numShards;
2477           if (options.getMaxNumWorkers() > 0) {
2478             numShards = options.getMaxNumWorkers() * 2;
2479           } else if (options.getNumWorkers() > 0) {
2480             numShards = options.getNumWorkers() * 2;
2481           } else {
2482             numShards = DEFAULT_NUM_SHARDS;
2483           }
2484     
2485           try {
2486             List<PCollectionView<?>> sideInputs =
2487                 WriteFilesTranslation.getDynamicDestinationSideInputs(transform);
2488             FileBasedSink sink = WriteFilesTranslation.getSink(transform);
2489             WriteFiles<UserT, DestinationT, OutputT> replacement =
2490                 WriteFiles.to(sink).withSideInputs(sideInputs);
2491             if (WriteFilesTranslation.isWindowedWrites(transform)) {
2492               replacement = replacement.withWindowedWrites();
2493             }
2494     
2495             if (WriteFilesTranslation.isAutoSharded(transform)) {
2496               replacement = replacement.withAutoSharding();
2497               return PTransformReplacement.of(
2498                   PTransformReplacements.getSingletonMainInput(transform), replacement);
2499             }
2500     
2501             return PTransformReplacement.of(
2502                 PTransformReplacements.getSingletonMainInput(transform),
2503                 replacement.withNumShards(numShards));
2504           } catch (Exception e) {
2505             throw new RuntimeException(e);
2506           }
2507         }

2510         public Map<PCollection<?>, ReplacementOutput> mapOutputs(
2511             Map<TupleTag<?>, PCollection<?>> outputs, WriteFilesResult<DestinationT> newOutput) {
2512           return ReplacementOutputs.tagged(outputs, newOutput);
2513         }
