252       private void runTestWrite(
253           String[] elems, String header, String footer, int numShards, boolean skipIfEmpty)
254           throws Exception {
255         String outputName = "file.txt";
256         Path baseDir = Files.createTempDirectory(tempFolder.getRoot().toPath(), "testwrite");
257         ResourceId baseFilename =
258             FileBasedSink.convertToFileResourceIfPossible(baseDir.resolve(outputName).toString());
259     
260         PCollection<String> input =
261             p.apply("CreateInput", Create.of(Arrays.asList(elems)).withCoder(StringUtf8Coder.of()));
262     
263         TextIO.TypedWrite<String, Void> write =
264             TextIO.write().to(baseFilename).withHeader(header).withFooter(footer).withOutputFilenames();
265     
266         if (numShards == 1) {
267           write = write.withoutSharding();
268         } else if (numShards > 0) {
269           write = write.withNumShards(numShards).withShardNameTemplate(ShardNameTemplate.INDEX_OF_MAX);
270         }
271         if (skipIfEmpty) {
272           write = write.skipIfEmpty();
273         }
274     
275         input.apply(write);
276     
277         p.run();
278     
279         assertOutputFiles(
280             elems,
281             header,
282             footer,
283             numShards,
284             baseFilename,
285             firstNonNull(
286                 write.getShardTemplate(), DefaultFilenamePolicy.DEFAULT_UNWINDOWED_SHARD_TEMPLATE),
287             skipIfEmpty);

301       private static void assertOutputFiles(
302           String[] elems,
303           final String header,
304           final String footer,
305           int numShards,
306           ResourceId outputPrefix,
307           String shardNameTemplate,
308           boolean skipIfEmpty)
309           throws Exception {
310         List<File> expectedFiles = new ArrayList<>();
311         if (skipIfEmpty && elems.length == 0) {
312           String pattern = outputPrefix.toString() + "*";
313           MatchResult matches =
314               Iterables.getOnlyElement(FileSystems.match(Collections.singletonList(pattern)));
315           assertEquals(NOT_FOUND, matches.status());
316         } else if (numShards == 0) {
317           String pattern = outputPrefix.toString() + "*";
318           List<MatchResult> matches = FileSystems.match(Collections.singletonList(pattern));
319           for (Metadata expectedFile : Iterables.getOnlyElement(matches).metadata()) {
320             expectedFiles.add(new File(expectedFile.resourceId().toString()));
321           }
322         } else {
323           for (int i = 0; i < numShards; i++) {
324             expectedFiles.add(
325                 new File(
326                     DefaultFilenamePolicy.constructName(
327                             outputPrefix, shardNameTemplate, "", i, numShards, null, null)
328                         .toString()));
329           }
330         }
331     
332         List<List<String>> actual = new ArrayList<>();
333     
334         for (File tmpFile : expectedFiles) {
335           List<String> currentFile = readLinesFromFile(tmpFile);
336           actual.add(currentFile);
337         }
338     
339         List<String> expectedElements = new ArrayList<>(elems.length);
340         for (String elem : elems) {
341           byte[] encodedElem = CoderUtils.encodeToByteArray(StringUtf8Coder.of(), elem);
342           String line = new String(encodedElem, StandardCharsets.UTF_8);
343           expectedElements.add(line);
344         }
345     
346         List<String> actualElements =
347             Lists.newArrayList(
348                 Iterables.concat(
349                     FluentIterable.from(actual)
350                         .transform(removeHeaderAndFooter(header, footer))
351                         .toList()));
352     
353         assertThat(actualElements, containsInAnyOrder(expectedElements.toArray()));
354         assertTrue(actual.stream().allMatch(haveProperHeaderAndFooter(header, footer)::apply));
355       }

357       private static List<String> readLinesFromFile(File f) throws IOException {
358         List<String> currentFile = new ArrayList<>();
359         try (BufferedReader reader = Files.newBufferedReader(f.toPath(), StandardCharsets.UTF_8)) {
360           while (true) {
361             String line = reader.readLine();
362             if (line == null) {
363               break;
364             }
365             currentFile.add(line);
366           }
367         }
368         return currentFile;
369       }
370     
371       private static Function<List<String>, List<String>> removeHeaderAndFooter(
372           final String header, final String footer) {
373         return new Function<List<String>, List<String>>() {
374           @Override
375           public @Nullable List<String> apply(List<String> lines) {
376             ArrayList<String> newLines = Lists.newArrayList(lines);
377             if (header != null) {
378               newLines.remove(0);
379             }
380             if (footer != null) {
381               int last = newLines.size() - 1;
382               newLines.remove(last);
383             }
384             return newLines;
385           }
386         };
387       }
388     
389       private static Predicate<List<String>> haveProperHeaderAndFooter(
390           final String header, final String footer) {
391         return fileLines -> {
392           int last = fileLines.size() - 1;
393           return (header == null || fileLines.get(0).equals(header))
394               && (footer == null || fileLines.get(last).equals(footer));
395         };
396       }

421       public void testWindowedWritesWithOnceTrigger() throws Throwable {
422         p.enableAbandonedNodeEnforcement(false);
423         expectedException.expect(IllegalArgumentException.class);
424         expectedException.expectMessage("Unsafe trigger");
425     
426         // Tests for https://issues.apache.org/jira/browse/BEAM-3169
427         PCollection<String> data =
428             p.apply(Create.of("0", "1", "2"))
429                 .apply(
430                     Window.<String>into(FixedWindows.of(Duration.standardSeconds(1)))
431                         // According to this trigger, all data should be written.
432                         // However, the continuation of this trigger is elementCountAtLeast(1),
433                         // so with a buggy implementation that used a GBK before renaming files,
434                         // only 1 file would be renamed.
435                         .triggering(AfterPane.elementCountAtLeast(3))
436                         .withAllowedLateness(Duration.standardMinutes(1))
437                         .discardingFiredPanes());
438         data.apply(
439                 TextIO.write()
440                     .to(new File(tempFolder.getRoot(), "windowed-writes").getAbsolutePath())
441                     .withNumShards(2)
442                     .withWindowedWrites()
443                     .<Void>withOutputFilenames())
444             .getPerDestinationOutputFilenames()
445             .apply(Values.create());
