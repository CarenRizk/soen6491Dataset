87     /**
88      * General-purpose transforms for working with files: listing files (matching), reading and writing.
89      *
90      * <h2>Matching filepatterns</h2>
91      *
92      * <p>{@link #match} and {@link #matchAll} match filepatterns (respectively either a single
93      * filepattern or a {@link PCollection} thereof) and return the files that match them as {@link
94      * PCollection PCollections} of {@link MatchResult.Metadata}. Configuration options for them are in
95      * {@link MatchConfiguration} and include features such as treatment of filepatterns that don't
96      * match anything and continuous incremental matching of filepatterns (watching for new files).
97      *
98      * <h3>Example: Watching a single filepattern for new files</h3>
99      *
100      * <p>This example matches a single filepattern repeatedly every 30 seconds, continuously returns
101      * new matched files as an unbounded {@code PCollection<Metadata>} and stops if no new files appear
102      * for 1 hour.
103      *
104      * <pre>{@code
105      * PCollection<Metadata> matches = p.apply(FileIO.match()
106      *     .filepattern("...")
107      *     .continuously(
108      *       Duration.standardSeconds(30), afterTimeSinceNewOutput(Duration.standardHours(1))));
109      * }</pre>
110      *
111      * <h3>Example: Matching a PCollection of filepatterns arriving from Kafka</h3>
112      *
113      * <p>This example reads filepatterns from Kafka and matches each one as it arrives, producing again
114      * an unbounded {@code PCollection<Metadata>}, and failing in case the filepattern doesn't match
115      * anything.
116      *
117      * <pre>{@code
118      * PCollection<String> filepatterns = p.apply(KafkaIO.read()...);
119      *
120      * PCollection<Metadata> matches = filepatterns.apply(FileIO.matchAll()
121      *     .withEmptyMatchTreatment(DISALLOW));
122      * }</pre>
123      *
124      * <h2>Reading files</h2>
125      *
126      * <p>{@link #readMatches} converts each result of {@link #match} or {@link #matchAll} to a {@link
127      * ReadableFile} that is convenient for reading a file's contents, optionally decompressing it.
128      *
129      * <h3>Example: Returning filenames and contents of compressed files matching a filepattern</h3>
130      *
131      * <p>This example matches a single filepattern and returns {@code KVs} of filenames and their
132      * contents as {@code String}, decompressing each file with GZIP.
133      *
134      * <pre>{@code
135      * PCollection<KV<String, String>> filesAndContents = p
136      *     .apply(FileIO.match().filepattern("hdfs://path/to/*.gz"))
137      *     // withCompression can be omitted - by default compression is detected from the filename.
138      *     .apply(FileIO.readMatches().withCompression(GZIP))
139      *     .apply(MapElements
140      *         // uses imports from TypeDescriptors
141      *         .into(kvs(strings(), strings()))
142      *         .via((ReadableFile f) -> {
143      *           try {
144      *             return KV.of(
145      *                 f.getMetadata().resourceId().toString(), f.readFullyAsUTF8String());
146      *           } catch (IOException ex) {
147      *             throw new RuntimeException("Failed to read the file", ex);
148      *           }
149      *         }));
150      * }</pre>
151      *
152      * <h2>Writing files</h2>
153      *
154      * <p>{@link #write} and {@link #writeDynamic} write elements from a {@link PCollection} of a given
155      * type to files, using a given {@link Sink} to write a set of elements to each file. The collection
156      * can be bounded or unbounded - in either case, writing happens by default per window and pane, and
157      * the amount of data in each window and pane is finite, so a finite number of files ("shards") are
158      * written for each window and pane. There are several aspects to this process:
159      *
160      * <ul>
161      *   <li><b>How many shards are generated per pane:</b> This is controlled by <i>sharding</i>, using
162      *       {@link Write#withNumShards} or {@link Write#withSharding}. The default is runner-specific,
163      *       so the number of shards will vary based on runner behavior, though at least 1 shard will
164      *       always be produced for every non-empty pane. Note that setting a fixed number of shards can
165      *       hurt performance: it adds an additional {@link GroupByKey} to the pipeline. However, it is
166      *       required to set it when writing an unbounded {@link PCollection} due to <a
167      *       href="https://issues.apache.org/jira/browse/BEAM-1438">BEAM-1438</a> and similar behavior
168      *       in other runners.
169      *   <li><b>How the shards are named:</b> This is controlled by a {@link Write.FileNaming}:
170      *       filenames can depend on a variety of inputs, e.g. the window, the pane, total number of
171      *       shards, the current file's shard index, and compression. Controlling the file naming is
172      *       described in the section <i>File naming</i> below.
173      *   <li><b>Which elements go into which shard:</b> Elements within a pane get distributed into
174      *       different shards created for that pane arbitrarily, though {@link FileIO.Write} attempts to
175      *       make shards approximately evenly sized. For more control over which elements go into which
176      *       files, consider using <i>dynamic destinations</i> (see below).
177      *   <li><b>How a given set of elements is written to a shard:</b> This is controlled by the {@link
178      *       Sink}, e.g. {@link AvroIO#sink} will generate Avro files. The {@link Sink} controls the
179      *       format of a single file: how to open a file, how to write each element to it, and how to
180      *       close the file - but it does not control the set of files or which elements go where.
181      *       Elements are written to a shard in an arbitrary order. {@link FileIO.Write} can
182      *       additionally compress the generated files using {@link FileIO.Write#withCompression}.
183      *   <li><b>How all of the above can be element-dependent:</b> This is controlled by <i>dynamic
184      *       destinations</i>. It is possible to have different groups of elements use different
185      *       policies for naming files and for configuring the {@link Sink}. See "dynamic destinations"
186      *       below.
187      * </ul>
188      *
189      * <h3>File naming</h3>
190      *
191      * <p>The names of generated files are produced by a {@link Write.FileNaming}. The default naming
192      * strategy is to name files in the format: {@code
193      * $prefix-$start-$end-$pane-$shard-of-$numShards$suffix$compressionSuffix}, where:
194      *
195      * <ul>
196      *   <li>$prefix is set by {@link Write#withPrefix}, the default is "output".
197      *   <li>$start and $end are boundaries of the window of data being written, formatted in ISO 8601
198      *       format (YYYY-mm-ddTHH:MM:SSZZZ). The window is omitted in case this is the global window.
199      *   <li>$pane is the index of the pane within the window. The pane is omitted in case it is known
200      *       to be the only pane for this window.
201      *   <li>$shard is the index of the current shard being written, out of the $numShards total shards
202      *       written for the current pane. Both are formatted using 5 digits (or more if necessary
203      *       according to $numShards) and zero-padded.
204      *   <li>$suffix is set by {@link Write#withSuffix}, the default is empty.
205      *   <li>$compressionSuffix is based on the default extension for the chosen {@link
206      *       Write#withCompression compression type}.
207      * </ul>
208      *
209      * <p>For example: {@code data-2017-12-01T19:00:00Z-2017-12-01T20:00:00Z-2-00010-of-00050.txt.gz}
210      *
211      * <p>Alternatively, one can specify a custom naming strategy using {@link
212      * Write#withNaming(Write.FileNaming)}.
213      *
214      * <p>If {@link Write#to} is specified, then the filenames produced by the {@link Write.FileNaming}
215      * are resolved relative to that directory.
216      *
217      * <p>When using dynamic destinations via {@link #writeDynamic} (see below), specifying a custom
218      * naming strategy is required, using {@link Write#withNaming(SerializableFunction)} or {@link
219      * Write#withNaming(Contextful)}. In those, pass a function that creates a {@link Write.FileNaming}
220      * for the requested group ("destination"). You can either implement a custom {@link
221      * Write.FileNaming}, or use {@link Write#defaultNaming} to configure the default naming strategy
222      * with a prefix and suffix as per above.
223      *
224      * <h3>Dynamic destinations</h3>
225      *
226      * <p>If the elements in the input collection can be partitioned into groups that should be treated
227      * differently, {@link FileIO.Write} supports different treatment per group ("destination"). It can
228      * use different file naming strategies for different groups, and can differently configure the
229      * {@link Sink}, e.g. write different elements to Avro files in different directories with different
230      * schemas.
231      *
232      * <p>This feature is supported by {@link #writeDynamic}. Use {@link Write#by} to specify how too
233      * partition the elements into groups ("destinations"). Then elements will be grouped by
234      * destination, and {@link Write#withNaming(Contextful)} and {@link Write#via(Contextful)} will be
235      * applied separately within each group, i.e. different groups will be written using the file naming
236      * strategies returned by {@link Write#withNaming(Contextful)} and using sinks returned by {@link
237      * Write#via(Contextful)} for the respective destinations. Note that currently sharding can not be
238      * destination-dependent: every window/pane for every destination will use the same number of shards
239      * specified via {@link Write#withNumShards} or {@link Write#withSharding}.
240      *
241      * <h3>Handling Errors</h3>
242      *
243      * <p>When using dynamic destinations, or when using a formatting function to format a record for
244      * writing, it's possible for an individual record to be malformed, causing an exception. By
245      * default, these exceptions are propagated to the runner causing the bundle to fail. These are
246      * usually retried, though this depends on the runner. Alternately, these errors can be routed to
247      * another {@link PTransform} by using {@link Write#withBadRecordErrorHandler(ErrorHandler)}. The
248      * ErrorHandler is registered with the pipeline (see below). See {@link ErrorHandler} for more
249      * documentation. Of note, this error handling only handles errors related to specific records. It
250      * does not handle errors related to connectivity, authorization, etc. as those should be retried by
251      * the runner.
252      *
253      * <pre>{@code
254      * PCollection<> records = ...;
255      * PTransform<PCollection<BadRecord>,?> alternateSink = ...;
256      * try (BadRecordErrorHandler<?> handler = pipeline.registerBadRecordErrorHandler(alternateSink) {
257      *    records.apply("Write", FileIO.writeDynamic().otherConfigs()
258      *        .withBadRecordErrorHandler(handler));
259      * }
260      * }</pre>
261      *
262      * <h3>Writing custom types to sinks</h3>
263      *
264      * <p>Normally, when writing a collection of a custom type using a {@link Sink} that takes a
265      * different type (for example, writing a {@code PCollection<Event>} to a text-based {@code
266      * Sink<String>}), one can simply apply a {@code ParDo} or {@code MapElements} to convert the custom
267      * type to the sink's <i>output type</i>.
268      *
269      * <p>However, when using dynamic destinations, in many such cases the destination needs to be
270      * extracted from the original type, so such a conversion is not possible. For example, one might
271      * write events of a custom class {@code Event} to a text sink, using the event's "type" as a
272      * destination. In that case, specify an <i>output function</i> in {@link Write#via(Contextful,
273      * Contextful)} or {@link Write#via(Contextful, Sink)}.
274      *
275      * <h3>Example: Writing CSV files</h3>
276      *
277      * <pre>{@code
278      * class CSVSink implements FileIO.Sink<List<String>> {
279      *   private String header;
280      *   private PrintWriter writer;
281      *
282      *   public CSVSink(List<String> colNames) {
283      *     this.header = Joiner.on(",").join(colNames);
284      *   }
285      *
286      *   public void open(WritableByteChannel channel) throws IOException {
287      *     writer = new PrintWriter(Channels.newOutputStream(channel));
288      *     writer.println(header);
289      *   }
290      *
291      *   public void write(List<String> element) throws IOException {
292      *     writer.println(Joiner.on(",").join(element));
293      *   }
294      *
295      *   public void flush() throws IOException {
296      *     writer.flush();
297      *   }
298      * }
299      *
300      * PCollection<BankTransaction> transactions = ...;
301      * // Convert transactions to strings before writing them to the CSV sink.
302      * transactions.apply(MapElements
303      *         .into(TypeDescriptors.lists(TypeDescriptors.strings()))
304      *         .via(tx -> Arrays.asList(tx.getUser(), tx.getAmount())))
305      *     .apply(FileIO.<List<String>>write()
306      *         .via(new CSVSink(Arrays.asList("user", "amount")))
307      *         .to(".../path/to/")
308      *         .withPrefix("transactions")
309      *         .withSuffix(".csv"));
310      * }</pre>
311      *
312      * <h3>Example: Writing CSV files to different directories and with different headers</h3>
313      *
314      * <pre>{@code
315      * enum TransactionType {
316      *   DEPOSIT,
317      *   WITHDRAWAL,
318      *   TRANSFER,
319      *   ...
320      *
321      *   List<String> getFieldNames();
322      *   List<String> getAllFields(BankTransaction tx);
323      * }
324      *
325      * PCollection<BankTransaction> transactions = ...;
326      * transactions.apply(FileIO.<TransactionType, Transaction>writeDynamic()
327      *     .by(Transaction::getTypeName)
328      *     .via(tx -> tx.getTypeName().toFields(tx),  // Convert the data to be written to CSVSink
329      *          type -> new CSVSink(type.getFieldNames()))
330      *     .to(".../path/to/")
331      *     .withNaming(type -> defaultNaming(type + "-transactions", ".csv"));
332      * }</pre>
333      */

415       /** A utility class for accessing a potentially compressed file. */
416       public static final class ReadableFile {
417         private final MatchResult.Metadata metadata;
418         private final Compression compression;
419     
420         ReadableFile(MatchResult.Metadata metadata, Compression compression) {
421           this.metadata = metadata;
422           this.compression = compression;
423         }
424     
425         /** Returns the {@link MatchResult.Metadata} of the file. */
426         public MatchResult.Metadata getMetadata() {
427           return metadata;
428         }
429     
430         /** Returns the method with which this file will be decompressed in {@link #open}. */
431         public Compression getCompression() {
432           return compression;
433         }
434     
435         /**
436          * Returns a {@link ReadableByteChannel} reading the data from this file, potentially
437          * decompressing it using {@link #getCompression}.
438          */
439         public ReadableByteChannel open() throws IOException {
440           return compression.readDecompressed(FileSystems.open(metadata.resourceId()));
441         }
442     
443         /**
444          * Returns a {@link SeekableByteChannel} equivalent to {@link #open}, but fails if this file is
445          * not {@link MatchResult.Metadata#isReadSeekEfficient seekable}.
446          */
447         public SeekableByteChannel openSeekable() throws IOException {
448           checkState(
449               getMetadata().isReadSeekEfficient(),
450               "The file %s is not seekable",
451               metadata.resourceId());
452           return (SeekableByteChannel) open();
453         }
454     
455         /** Returns the full contents of the file as bytes. */
456         public byte[] readFullyAsBytes() throws IOException {
457           try (InputStream stream = Channels.newInputStream(open())) {
458             return StreamUtils.getBytesWithoutClosing(stream);
459           }
460         }
461     
462         /** Returns the full contents of the file as a {@link String} decoded as UTF-8. */
463         public String readFullyAsUTF8String() throws IOException {
464           return new String(readFullyAsBytes(), StandardCharsets.UTF_8);
465         }
466     
467         @Override
468         public String toString() {
469           return "ReadableFile{metadata=" + metadata + ", compression=" + compression + '}';
470         }
471     
472         @Override
473         public boolean equals(@Nullable Object o) {
474           if (this == o) {
475             return true;
476           }
477           if (o == null || getClass() != o.getClass()) {
478             return false;
479           }
480           ReadableFile that = (ReadableFile) o;
481           return Objects.equal(metadata, that.metadata) && compression == that.compression;
482         }

485         public int hashCode() {
486           return Objects.hashCode(metadata, compression);
487         }

887                   .setIsReadSeekEfficient(

983               }

1386         @Override
1387         public WriteFilesResult<DestinationT> expand(PCollection<UserT> input) {
1388           Write.Builder<DestinationT, UserT> resolvedSpec = new AutoValue_FileIO_Write.Builder<>();
1389     
1390           resolvedSpec.setDynamic(getDynamic());
1391     
1392           checkArgument(getSinkFn() != null, ".via() is required");
1393           resolvedSpec.setSinkFn(getSinkFn());
1394     
1395           checkArgument(getOutputFn() != null, "outputFn should have been set by .via()");
1396           resolvedSpec.setOutputFn(getOutputFn());
1397     
1398           // Resolve destinationFn
1399           if (getDynamic()) {
1400             checkArgument(getDestinationFn() != null, "when using writeDynamic(), .by() is required");
1401             resolvedSpec.setDestinationFn(getDestinationFn());
1402             resolvedSpec.setDestinationCoder(resolveDestinationCoder(input));
1403           } else {
1404             checkArgument(getDestinationFn() == null, ".by() requires writeDynamic()");
1405             checkArgument(
1406                 getDestinationCoder() == null, ".withDestinationCoder() requires writeDynamic()");
1407             resolvedSpec.setDestinationFn(fn(SerializableFunctions.constant(null)));
1408             resolvedSpec.setDestinationCoder((Coder) VoidCoder.of());
1409           }
1410     
1411           resolvedSpec.setFileNamingFn(resolveFileNamingFn());
1412           resolvedSpec.setEmptyWindowDestination(getEmptyWindowDestination());
1413           if (getTempDirectory() == null) {
1414             checkArgument(
1415                 getOutputDirectory() != null, "must specify either .withTempDirectory() or .to()");
1416             resolvedSpec.setTempDirectory(getOutputDirectory());
1417           } else {
1418             resolvedSpec.setTempDirectory(getTempDirectory());
1419           }
1420     
1421           resolvedSpec.setCompression(getCompression());
1422           resolvedSpec.setNumShards(getNumShards());
1423           resolvedSpec.setSharding(getSharding());
1424           resolvedSpec.setIgnoreWindowing(getIgnoreWindowing());
1425           resolvedSpec.setAutoSharding(getAutoSharding());
1426           resolvedSpec.setNoSpilling(getNoSpilling());
1427     
1428           Write<DestinationT, UserT> resolved = resolvedSpec.build();
1429           WriteFiles<UserT, DestinationT, ?> writeFiles =
1430               WriteFiles.to(new ViaFileBasedSink<>(resolved))
1431                   .withSideInputs(Lists.newArrayList(resolved.getAllSideInputs()));
1432           if (getNumShards() != null) {
1433             writeFiles = writeFiles.withNumShards(getNumShards());
1434           } else if (getSharding() != null) {
1435             writeFiles = writeFiles.withSharding(getSharding());
1436           } else {
1437             writeFiles = writeFiles.withRunnerDeterminedSharding();
1438           }
1439           if (!getIgnoreWindowing()) {
1440             writeFiles = writeFiles.withWindowedWrites();
1441           }
1442           if (getAutoSharding()) {
1443             writeFiles = writeFiles.withAutoSharding();
1444           }
1445           if (getNoSpilling()) {
1446             writeFiles = writeFiles.withNoSpilling();
1447           }
1448           if (getBadRecordErrorHandler() != null) {
1449             writeFiles = writeFiles.withBadRecordErrorHandler(getBadRecordErrorHandler());
1450           }
1451           return input.apply(writeFiles);
1452         }
