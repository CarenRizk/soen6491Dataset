119       private static File writeToFile(
120           List<String> lines, TemporaryFolder folder, String fileName, Compression compression)
121           throws IOException {
122         File file = folder.getRoot().toPath().resolve(fileName).toFile();
123         OutputStream output = new FileOutputStream(file);
124         switch (compression) {
125           case UNCOMPRESSED:
126             break;
127           case GZIP:
128             output = new GZIPOutputStream(output);
129             break;
130           case BZIP2:
131             output = new BZip2CompressorOutputStream(output);
132             break;
133           case ZIP:
134             ZipOutputStream zipOutput = new ZipOutputStream(output);
135             zipOutput.putNextEntry(new ZipEntry("entry"));
136             output = zipOutput;
137             break;
138           case DEFLATE:
139             output = new DeflateCompressorOutputStream(output);
140             break;
141           default:
142             throw new UnsupportedOperationException(compression.toString());
143         }
144         writeToStreamAndClose(lines, output);
145         return file;
146       }

153         try (PrintStream writer = new PrintStream(outputStream)) {

169       /**
170        * Helper method that runs a variety of ways to read a single file using TextIO and checks that
171        * they all match the given expected output.
172        *
173        * <p>The transforms being verified are:
174        *
175        * <ul>
176        *   <li>TextIO.read().from(filename).withCompression(compressionType)
177        *   <li>TextIO.read().from(filename).withCompression(compressionType) .withHintMatchesManyFiles()
178        *   <li>TextIO.readFiles().withCompression(compressionType)
179        *   <li>TextIO.readAll().withCompression(compressionType)
180        * </ul>
181        */
182       private static void assertReadingCompressedFileMatchesExpected(
183           File file, Compression compression, List<String> expected, Pipeline p) {
184     
185         TextIO.Read read = TextIO.read().from(file.getPath()).withCompression(compression);
186     
187         PAssert.that(p.apply("Read_" + file + "_" + compression.toString(), read))
188             .containsInAnyOrder(expected);
189         PAssert.that(
190                 p.apply(
191                     "Read_" + file + "_" + compression.toString() + "_many",
192                     read.withHintMatchesManyFiles()))
193             .containsInAnyOrder(expected);
194     
195         PAssert.that(
196                 p.apply("Create_Paths_ReadFiles_" + file, Create.of(file.getPath()))
197                     .apply("Match_" + file, FileIO.matchAll())
198                     .apply("ReadMatches_" + file, FileIO.readMatches().withCompression(compression))
199                     .apply("ReadFiles_" + compression.toString(), TextIO.readFiles()))
200             .containsInAnyOrder(expected);
201     
202         PAssert.that(
203                 p.apply("Create_Paths_ReadAll_" + file, Create.of(file.getPath()))
204                     .apply(
205                         "ReadAll_" + compression.toString(),
206                         TextIO.readAll().withCompression(compression)))
207             .containsInAnyOrder(expected);
208       }

210       /**
211        * Create a zip file with the given lines.
212        *
213        * @param expected A list of expected lines, populated in the zip file.
214        * @param folder A temporary folder used to create files.
215        * @param filename Optionally zip file name (can be null).
216        * @param fieldsEntries Fields to write in zip entries.
217        * @return The zip filename.
218        * @throws Exception In case of a failure during zip file creation.
219        */
220       private static File createZipFile(
221           List<String> expected, TemporaryFolder folder, String filename, String[]... fieldsEntries)
222           throws Exception {
223         File tmpFile = folder.getRoot().toPath().resolve(filename).toFile();
224     
225         ZipOutputStream out = new ZipOutputStream(new FileOutputStream(tmpFile));
226         PrintStream writer = new PrintStream(out, true /* auto-flush on write */);
227     
228         int index = 0;
229         for (String[] entry : fieldsEntries) {
230           out.putNextEntry(new ZipEntry(Integer.toString(index)));
231           for (String field : entry) {
232             writer.println(field);
233             expected.add(field);
234           }
235           out.closeEntry();
236           index++;
237         }
238     
239         writer.close();
240         out.close();
241     
242         return tmpFile;
243       }

317           int readCount = 0;
318           InputStream stream;
319           ReadableByteChannel channel;
320     
321           public SlowReadChannel(FileBasedSource source) throws IOException {
322             channel =
323                 FileSystems.open(
324                     FileSystems.matchSingleFileSpec(source.getFileOrPatternSpec()).resourceId());
325             stream = Channels.newInputStream(channel);
326           }
327     
328           // Data is read at most one byte at a time from line parameter.
329           @Override
330           public int read(ByteBuffer dst) throws IOException {
331             if (++readCount % 3 == 0) {
332               if (dst.hasRemaining()) {
333                 int value = stream.read();
334                 if (value == -1) {
335                   return -1;
336                 }
337                 dst.put((byte) value);
338                 return 1;
339               }
340             }
341             return 0;
342           }

344           @Override
345           public boolean isOpen() {
346             return channel.isOpen();
347           }
348     
349           @Override
350           public void close() throws IOException {

352           }

696           String file = "line1\nline2\nline3";
697           BoundedSource<String> source = prepareSource(file.getBytes(StandardCharsets.UTF_8));
698           BoundedSource<String> remainder;
699     
700           // Create the remainder, verifying properties pre- and post-splitting.
701           try (BoundedSource.BoundedReader<String> readerOrig =
702               source.createReader(PipelineOptionsFactory.create())) {
703             // Preconditions.
704             assertEquals(0.0, readerOrig.getFractionConsumed(), 1e-6);
705             assertEquals(0, readerOrig.getSplitPointsConsumed());
706             assertEquals(
707                 BoundedSource.BoundedReader.SPLIT_POINTS_UNKNOWN, readerOrig.getSplitPointsRemaining());
708     
709             // First record, before splitting.
710             assertTrue(readerOrig.start());
711             assertEquals(0, readerOrig.getSplitPointsConsumed());
712             assertEquals(
713                 BoundedSource.BoundedReader.SPLIT_POINTS_UNKNOWN, readerOrig.getSplitPointsRemaining());
714     
715             // Split. 0.1 is in line1, so should now be able to detect last record.
716             remainder = readerOrig.splitAtFraction(0.1);
717             assertNotNull(remainder);
718     
719             // First record, after splitting.
720             assertEquals(0, readerOrig.getSplitPointsConsumed());
721             assertEquals(1, readerOrig.getSplitPointsRemaining());
722     
723             // Finish and postconditions.

726             assertEquals(1, readerOrig.getSplitPointsConsumed());
727             assertEquals(0, readerOrig.getSplitPointsRemaining());
728           }
729     
730           // Check the properties of the remainder.
731           try (BoundedSource.BoundedReader<String> reader =
732               remainder.createReader(PipelineOptionsFactory.create())) {
733             // Preconditions.
734             assertEquals(0.0, reader.getFractionConsumed(), 1e-6);

736             assertEquals(

742             assertEquals(
743                 BoundedSource.BoundedReader.SPLIT_POINTS_UNKNOWN, reader.getSplitPointsRemaining());
744     
745             // Second record is line 3
746             assertTrue(reader.advance());
747             assertEquals(1, reader.getSplitPointsConsumed());

844         private static class FileReadDoFn extends DoFn<FileIO.ReadableFile, String> {
845     
846           @ProcessElement
847           public void processElement(ProcessContext c) {
848             FileIO.ReadableFile file = c.element();
849             ValueProvider<String> filenameProvider =
850                 ValueProvider.StaticValueProvider.of(file.getMetadata().resourceId().getFilename());
851             // Create a TextSource, passing null as the delimiter to use the default
852             // delimiters ('\n', '\r', or '\r\n').
853             TextSource textSource = new TextSource(filenameProvider, null, null, 0);
854             try {
855               BoundedSource.BoundedReader<String> reader =
856                   textSource
857                       .createForSubrangeOfFile(file.getMetadata(), 0, file.getMetadata().sizeBytes())
858                       .createReader(c.getPipelineOptions());
859               for (boolean more = reader.start(); more; more = reader.advance()) {
860                 c.output(reader.getCurrent());
861               }
862             } catch (IOException e) {
863               throw new RuntimeException(
864                   "Unable to readFile: " + file.getMetadata().resourceId().toString());
865             }
866           }
867         }

869         /** A transform that reads CSV file records. */
870         private static class TextFileReadTransform
871             extends PTransform<PCollection<String>, PCollection<String>> {
872           public TextFileReadTransform() {}
873     
874           @Override
875           public PCollection<String> expand(PCollection<String> files) {
876             return files
877                 // PCollection<String>
878                 .apply(FileIO.matchAll().withEmptyMatchTreatment(EmptyMatchTreatment.DISALLOW))
879                 // PCollection<Match.Metadata>
880                 .apply(FileIO.readMatches())
881                 // PCollection<FileIO.ReadableFile>
882                 .apply("Read lines", ParDo.of(new TextIOReadTest.TextSourceTest.FileReadDoFn()));
883             // PCollection<String>: line
884           }
885         }
