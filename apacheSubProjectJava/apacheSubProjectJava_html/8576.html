326                             UUID.randomUUID().toString()))
327                 .collect(Collectors.toList());

329         clientFactory = PubsubTestClient.createFactoryForPull(CLOCK, SUBSCRIPTION, 60, messages);

456       @Test
457       public void testWriteMalformedMessagesWithErrorHandler() throws Exception {
458         OutgoingMessage msg =
459             OutgoingMessage.of(
460                 com.google.pubsub.v1.PubsubMessage.newBuilder()
461                     .setData(ByteString.copyFromUtf8("foo"))
462                     .build(),
463                 0,
464                 null,
465                 "projects/project/topics/topic1");
466     
467         try (PubsubTestClientFactory factory =
468             PubsubTestClient.createFactoryForPublish(null, ImmutableList.of(msg), ImmutableList.of())) {
469           TimestampedValue<PubsubMessage> pubsubMsg =
470               TimestampedValue.of(
471                   new PubsubMessage(
472                           msg.getMessage().getData().toByteArray(),
473                           Collections.emptyMap(),
474                           msg.recordId())
475                       .withTopic(msg.topic()),
476                   Instant.ofEpochMilli(msg.getTimestampMsSinceEpoch()));
477     
478           TimestampedValue<PubsubMessage> failingPubsubMsg =
479               TimestampedValue.of(
480                   new PubsubMessage(
481                           "foo".getBytes(StandardCharsets.UTF_8),
482                           Collections.emptyMap(),
483                           msg.recordId())
484                       .withTopic("badTopic"),
485                   Instant.ofEpochMilli(msg.getTimestampMsSinceEpoch()));
486     
487           PCollection<PubsubMessage> messages =
488               pipeline.apply(
489                   Create.timestamped(ImmutableList.of(pubsubMsg, failingPubsubMsg))
490                       .withCoder(PubsubMessageWithTopicCoder.of()));
491           messages.setIsBoundedInternal(PCollection.IsBounded.BOUNDED);
492           ErrorHandler<BadRecord, PCollection<Long>> badRecordErrorHandler =
493               pipeline.registerBadRecordErrorHandler(new ErrorSinkTransform());
494           // The most straightforward method to simulate a bad message is to have a format function that
495           // deterministically fails based on some value
496           messages.apply(
497               PubsubIO.writeMessages()
498                   .toBuilder()
499                   .setFormatFn(
500                       (ValueInSingleWindow<PubsubMessage> messageAndWindow) -> {
501                         if (messageAndWindow.getValue().getTopic().equals("badTopic")) {
502                           throw new RuntimeException("expected exception");
503                         }
504                         return messageAndWindow.getValue();
505                       })
506                   .build()
507                   .to("projects/project/topics/topic1")
508                   .withClientFactory(factory)
509                   .withErrorHandler(badRecordErrorHandler));
510           badRecordErrorHandler.close();
511           PAssert.thatSingleton(badRecordErrorHandler.getOutput()).isEqualTo(1L);
512           pipeline.run();
513         }
514       }
